import { useEffect, useState, useCallback } from "react";
import {
  View,
  Text,
  TextInput,
  TouchableOpacity,
  StyleSheet,
  ScrollView,
  Alert,
  ActivityIndicator,
} from "react-native";
import { useRouter } from "expo-router";
import { getProfile, updateProfile, type UserProfile } from "../../api/profile";
import { logout } from "../../api/auth";
import { useAuthStore } from "../../store/auth";
import { AvatarPicker } from "../../components/AvatarPicker";
import { PhotosGrid } from "../../components/PhotosGrid";

function useProfileScreen() {
  const router = useRouter();
  const setUser = useAuthStore((s) => s.setUser);
  const [profile, setProfile] = useState<UserProfile | null>(null);
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);
  const [draft, setDraft] = useState({ name: "", bio: "", age: "", gender: "male" as UserProfile["gender"] });

  useEffect(() => {
    getProfile()
      .then((p) => {
        setProfile(p);
        setDraft({ name: p.name, bio: p.bio, age: String(p.age), gender: p.gender });
      })
      .finally(() => setLoading(false));
  }, []);

  const handleSave = useCallback(async () => {
    setSaving(true);
    try {
      const updated = await updateProfile({
        name: draft.name,
        bio: draft.bio,
        age: Number(draft.age) || 0,
        gender: draft.gender,
      });
      setProfile(updated);
      Alert.alert("Saved", "Profile updated successfully");
    } catch {
      Alert.alert("Error", "Failed to update profile");
    } finally {
      setSaving(false);
    }
  }, [draft]);

  const handleLogout = useCallback(async () => {
    await logout();
    setUser(null);
    router.replace("/login");
  }, [setUser, router]);

  return { profile, loading, saving, draft, setDraft, handleSave, handleLogout };
}

export default function MeScreen() {
  const { profile, loading, saving, draft, setDraft, handleSave, handleLogout } = useProfileScreen();

  if (loading) {
    return (
      <View style={s.loader}>
        <ActivityIndicator size="large" color="#E84D7A" />
      </View>
    );
  }

  return (
    <ScrollView style={s.container} contentContainerStyle={s.content}>
      <AvatarPicker uri={profile?.avatar ?? null} size={100} />
      <FieldLabel label="Photos" />
      <PhotosGrid photos={profile?.photos ?? []} />
      <ProfileForm draft={draft} onDraftChange={setDraft} />
      <SaveButton saving={saving} onSave={handleSave} />
      <LogoutButton onLogout={handleLogout} />
    </ScrollView>
  );
}

interface DraftState {
  name: string;
  bio: string;
  age: string;
  gender: UserProfile["gender"];
}

function ProfileForm({ draft, onDraftChange }: { draft: DraftState; onDraftChange: (d: DraftState) => void }) {
  return (
    <View style={s.form}>
      <FieldLabel label="Gender" />
      <GenderSelector value={draft.gender} onChange={(g) => onDraftChange({ ...draft, gender: g })} />
      <FieldLabel label="Name" />
      <TextInput
        style={s.input}
        value={draft.name}
        onChangeText={(v) => onDraftChange({ ...draft, name: v })}
        placeholderTextColor="#555"
        placeholder="Your name"
      />
      <FieldLabel label="Age" />
      <TextInput
        style={s.input}
        value={draft.age}
        onChangeText={(v) => onDraftChange({ ...draft, age: v })}
        placeholderTextColor="#555"
        placeholder="Your age"
        keyboardType="number-pad"
      />
      <FieldLabel label="Bio" />
      <TextInput
        style={[s.input, s.bioInput]}
        value={draft.bio}
        onChangeText={(v) => onDraftChange({ ...draft, bio: v })}
        placeholderTextColor="#555"
        placeholder="Tell us about yourself"
        multiline
        numberOfLines={3}
        textAlignVertical="top"
      />
    </View>
  );
}

function FieldLabel({ label }: { label: string }) {
  return <Text style={s.label}>{label}</Text>;
}

function SaveButton({ saving, onSave }: { saving: boolean; onSave: () => void }) {
  return (
    <TouchableOpacity style={s.saveBtn} onPress={onSave} disabled={saving}>
      {saving ? <ActivityIndicator color="#fff" /> : <Text style={s.saveBtnText}>Save Changes</Text>}
    </TouchableOpacity>
  );
}

function LogoutButton({ onLogout }: { onLogout: () => void }) {
  const confirmLogout = () => {
    Alert.alert("Log Out", "Are you sure you want to log out?", [
      { text: "Cancel", style: "cancel" },
      { text: "Log Out", style: "destructive", onPress: onLogout },
    ]);
  };

  return (
    <TouchableOpacity style={s.logoutBtn} onPress={confirmLogout}>
      <Text style={s.logoutBtnText}>Log Out</Text>
    </TouchableOpacity>
  );
}

const GENDER_OPTIONS: { value: UserProfile["gender"]; label: string }[] = [
  { value: "male", label: "Male" },
  { value: "female", label: "Female" },
  { value: "other", label: "Other" },
];

function GenderSelector({ value, onChange }: { value: UserProfile["gender"]; onChange: (g: UserProfile["gender"]) => void }) {
  return (
    <View style={s.genderRow}>
      {GENDER_OPTIONS.map((opt) => (
        <TouchableOpacity
          key={opt.value}
          style={[s.genderBtn, value === opt.value && s.genderBtnActive]}
          onPress={() => onChange(opt.value)}
        >
          <Text style={[s.genderBtnText, value === opt.value && s.genderBtnTextActive]}>{opt.label}</Text>
        </TouchableOpacity>
      ))}
    </View>
  );
}

  genderBtnTextActive: { color: "#E84D7A" },
});
