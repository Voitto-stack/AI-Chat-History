# taskMode

```ts
import type { UserInfo } from "@sitin/api-proto/gen/archat_api/user_api";

export const PROFILE_TASK_PATHS = ["/tags", "/bio", "/profession"] as const;

type TaskStep = 0 | 1 | 2;

const STEP_BY_PATH: Record<string, TaskStep> = PROFILE_TASK_PATHS.reduce(
  (acc, path, index) => {
    acc[path] = index as TaskStep;
    return acc;
  },
  {} as Record<string, TaskStep>,
);

export const PERCENT_BY_STEP: Record<TaskStep, number> = {
  0: 33.33,
  1: 66.66,
  2: 100,
};

export function getTaskStep(pathname: string): TaskStep {
  return STEP_BY_PATH[pathname] ?? 0;
}

export function getProfileTaskEntryPath(): (typeof PROFILE_TASK_PATHS)[number] {
  return PROFILE_TASK_PATHS[0];
}

export function getTaskNextPath(pathname: string): string {
  const currentStep = getTaskStep(pathname);
  return PROFILE_TASK_PATHS[currentStep + 1] ?? "/";
}

export function getTaskPrevPath(pathname: string): string {
  const currentStep = getTaskStep(pathname);
  return PROFILE_TASK_PATHS[currentStep - 1] ?? "/";
}

export function isCompleteProfileDone(userInfo: UserInfo | null | undefined) {
  if (!userInfo) return false;
  const hasBio = (userInfo.bio ?? "").trim().length > 0;
  const hasProfession = (userInfo.profession ?? "").trim().length > 0;
  const hasTags = (userInfo.interestTabs ?? []).some((item) => (item.tags?.length ?? 0) > 0);
  return hasBio && hasProfession && hasTags;
}
```
