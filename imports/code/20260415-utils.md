export const formatAmount = (amount: number): string => {
  return amount >= 0 ? `+$${amount.toFixed(2)}` : `-$${Math.abs(amount).toFixed(2)}`;
};

export const formatAmountInCents = (amount: number): string => {
  return amount >= 0 ? `+¢${Math.floor(amount * 100)}` : `-¢${Math.floor(Math.abs(amount) * 100)}`;
};

export const formatDuration = (seconds: number): string => {
  const minutes = Math.floor(seconds / 60);

  return `${minutes}:${remainingSeconds.toString().padStart(2, "0")}`;
};

export function isCallChangeType(type: number): boolean {
  return (
    type === PwaUserBalanceChangeType.PWA_USER_BALANCE_CHANGE_TYPE_VIDEO_CALL ||
    type === PwaUserBalanceChangeType.PWA_USER_BALANCE_CHANGE_TYPE_VOICE_CALL ||
    type === PwaUserBalanceChangeType.PWA_USER_BALANCE_CHANGE_TYPE_MOCK_VIDEO
  );
}

/** 将单条 API 余额变动记录转换为 RecordItem */
export function buildRecordItem(h: PwaUserBalanceChangeHistory, orders: CallOrderInfo[], dateKey: string): RecordItem {
  const changeTimeSec = h.changeTime ?? 0;
  const amount = parseFloat(h.changeBalance ?? "0") || 0;
  const changeType = h.type ?? 0;
  const recordType = CHANGE_TYPE_TO_RECORD_TYPE[changeType] ?? RecordType.Other;
  const isCall = isCallChangeType(changeType);
  const matchingOrders = isCall ? orders : [];

  const record: RecordItem = {
    id: `${dateKey}_${recordType}_${changeTimeSec}`,
    type: recordType,
    date: dateKey,
    amount,
    time: changeTimeSec,
    orders: isCall ? matchingOrders : undefined,
  };

  if (isCall && matchingOrders.length > 0) {
    const totalCallDuration = matchingOrders.reduce((s, o) => s + (o.callDuration ?? 0), 0);
    const totalFreeAmount = matchingOrders.reduce((s, o) => s + (parseFloat(o.freeAmount ?? "0") || 0), 0);
    record.duration = totalCallDuration;
    record.freeAmount = totalFreeAmount.toFixed(2);
  }

  return record;
}
