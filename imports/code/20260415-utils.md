/**
 * Lovia 模块工具函数
 */

/**
 * 图片 URL 解析结果
 */
export interface ParsedImageUrl {
  url: string;
  width?: number;
  height?: number;
  minUrl?: string;
  minWidth?: number;
  minHeight?: number;
  midUrl?: string;
  midWidth?: number;
  midHeight?: number;
}

/**
 * 解析图片 URL
 * 后端返回的 url 字段可能是 JSON 字符串，需要解析后获取真实的图片地址
 *
 * @param url - 可能是 JSON 字符串或普通 URL
 * @param preferSize - 优先返回的尺寸：'mid' 中等尺寸，'min' 缩略图，'original' 原图
 * @returns 解析后的图片 URL
 */
export const parseImageUrl = (
  url: string,
  preferSize: 'mid' | 'min' | 'original' = 'mid'
): string => {
  if (!url) return '';

  // 如果是 JSON 字符串，解析它
  if (url.startsWith('{')) {
    try {
      const parsed: ParsedImageUrl = JSON.parse(url);

      // 根据偏好尺寸返回对应的 URL
      switch (preferSize) {
        case 'mid':
          return parsed.midUrl || parsed.minUrl || parsed.url || '';
        case 'min':
          return parsed.minUrl || parsed.midUrl || parsed.url || '';
        case 'original':
          return parsed.url || parsed.midUrl || parsed.minUrl || '';
        default:
          return parsed.midUrl || parsed.minUrl || parsed.url || '';
      }
    } catch {
      return url;
    }
  }

  // 已经是普通 URL，直接返回
  return url;
};

/**
 * 解析完整的图片信息
 * @param url - 可能是 JSON 字符串或普通 URL
 * @returns 解析后的完整图片信息
 */
export const parseImageInfo = (url: string): ParsedImageUrl => {
  if (!url) return { url: '' };

  if (url.startsWith('{')) {
    try {
      return JSON.parse(url);
    } catch {
      return { url };
    }
  }

  return { url };
};
