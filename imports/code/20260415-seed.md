import { PrismaClient } from '../src/generated/prisma';
import { randomBytes } from 'crypto';

const prisma = new PrismaClient();

/**
 * Generate a random password for system users
 */
function generateRandomPassword(length: number = 16): string {
  return randomBytes(length).toString('base64').slice(0, length);
}

/**
 * Initial permissions for RBAC system
 * Format: { domain }.{ resource }.{ action }
 */
const INITIAL_PERMISSIONS = [
  // RBAC management permissions
  { key: 'rbac.user.read', description: 'View admin users' },
  { key: 'rbac.user.write', description: 'Create and update admin users' },
  { key: 'rbac.user.delete', description: 'Delete admin users' },
  { key: 'rbac.role.read', description: 'View roles' },
  { key: 'rbac.role.write', description: 'Create and update roles' },
  { key: 'rbac.role.delete', description: 'Delete roles' },
  { key: 'rbac.permission.read', description: 'View permissions' },
  { key: 'rbac.permission.write', description: 'Create and update permissions' },
  { key: 'rbac.permission.delete', description: 'Delete permissions' },

  // PWA Dashboard permissions (example for future panels)
  { key: 'pwa.dashboard.read', description: 'View PWA dashboard' },
  { key: 'pwa.dashboard.export', description: 'Export PWA dashboard data' },

  // Lovia (Story Management) permissions
  { key: 'lovia.story.create', description: 'Create stories' },
  { key: 'lovia.story.read', description: 'View stories' },
  { key: 'lovia.story.update', description: 'Update stories' },
  { key: 'lovia.story.delete', description: 'Delete stories' },

  // General admin permissions
  { key: 'admin.audit.read', description: 'View audit logs' },
];

async function main() {
  console.log('🌱 Starting seed...');

  // Create permissions
  console.log('📝 Creating permissions...');
  for (const perm of INITIAL_PERMISSIONS) {
    await prisma.permission.upsert({
      where: { key: perm.key },
      update: { description: perm.description },
      create: perm,
    });
  }
  console.log(`   Created ${INITIAL_PERMISSIONS.length} permissions`);

  // Create Super Admin role with all permissions
  console.log('👑 Creating Super Admin role...');
  const allPermissions = await prisma.permission.findMany();

  const superAdminRole = await prisma.role.upsert({
    where: { name: 'Super Admin' },
    update: { description: 'Full system access' },
    create: {
      name: 'Super Admin',
      description: 'Full system access',
    },
  });

  // Assign all permissions to Super Admin role
  for (const perm of allPermissions) {
    await prisma.rolePermission.upsert({
      where: {
        roleId_permissionId: {
          roleId: superAdminRole.id,
          permissionId: perm.id,
        },
      },
      update: {},
      create: {
        roleId: superAdminRole.id,
        permissionId: perm.id,
      },
    });
  }
  console.log(`   Assigned ${allPermissions.length} permissions to Super Admin`);

  // Create Admin role (All read permissions)
  console.log('👤 Creating Admin role...');
  const adminRole = await prisma.role.upsert({
    where: { name: 'Admin' },
    update: { description: 'Standard admin access' },
    create: {
      name: 'Admin',
      description: 'Standard admin access',
    },
  });

  const adminPermissionKeys = [
    'rbac.user.read',
    'rbac.role.read',
    'rbac.permission.read',
    'pwa.dashboard.read',
    'pwa.dashboard.export',
    'admin.audit.read',
  ];

  const adminPermissions = await prisma.permission.findMany({
    where: { key: { in: adminPermissionKeys } },
  });

  for (const perm of adminPermissions) {
    await prisma.rolePermission.upsert({
      where: {
        roleId_permissionId: {
          roleId: adminRole.id,
          permissionId: perm.id,
        },
      },
      update: {},
      create: {
        roleId: adminRole.id,
        permissionId: perm.id,
      },
    });
  }
  console.log(`   Assigned ${adminPermissions.length} permissions to Admin`);

  // Create Viewer role (All PWA permissions)
  console.log('👀 Creating Viewer role...');
  const viewerRole = await prisma.role.upsert({
    where: { name: 'Viewer' },
    update: { description: 'Full access to PWA resources' },
    create: {
      name: 'Viewer',
      description: 'Full access to PWA resources',
    },
  });

  const viewerPermissions = await prisma.permission.findMany({
    where: { key: { startsWith: 'pwa.' } },
  });

  for (const perm of viewerPermissions) {
    await prisma.rolePermission.upsert({
      where: {
        roleId_permissionId: {
          roleId: viewerRole.id,
          permissionId: perm.id,
        },
      },
      update: {},
      create: {
        roleId: viewerRole.id,
        permissionId: perm.id,
      },
    });
  }
  console.log(`   Assigned ${viewerPermissions.length} permissions to Viewer`);

  // Create initial super_admin user
  console.log('🔐 Creating super_admin user...');

  // Generate random password for system user
  const superAdminPassword = generateRandomPassword();

  const superAdminUser = await prisma.adminUser.upsert({
    where: { sub: 'super_admin' },
    update: {},
    create: {
      sub: 'super_admin',
      displayName: 'Super Administrator',
      password: superAdminPassword,
      userType: 'SYSTEM',
    },
  });

  // Assign Super Admin role to super_admin user
  await prisma.userRole.upsert({
    where: {
      userId_roleId: {
        userId: superAdminUser.id,
        roleId: superAdminRole.id,
      },
    },
    update: {},
    create: {
      userId: superAdminUser.id,
      roleId: superAdminRole.id,
    },
  });

  console.log('   Created super_admin user with Super Admin role');
  console.log('');
  console.log('✅ Seed completed!');
  console.log('');
  console.log('🔑 Initial system account:');
  console.log('   Sub: super_admin');
  console.log('   Password:', superAdminPassword);
  console.log('');
  console.log('⚠️  Please save this password securely! It will not be shown again.');
  console.log('⚠️  Feishu OAuth users have default password "presence" but password login is disabled.');
}

main()
  .catch((e) => {
    console.error('❌ Seed failed:', e);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });
