# userInfo

```ts
import WebsocketManager from "./WebsocketManager";
import { useUserStore } from "@/stores/userStore";
import { useChatListStore } from "@/stores/chatListStore";
import { S2CUserInfoUpdate, S2CMaleUserInfoUpdate } from "@heyhru/business-pwa-proto/gen/ai_tcp/chat_match";
import { OnWsMessageListener } from "./WebSocketConfig";

const TAG = "UserInfoWsHandler";

/**
 * 用户信息 WebSocket 处理器
 * 负责用户信息相关的 WebSocket 监听和更新逻辑
 * - S2C_UserInfoUpdate: 当前登录用户自身信息变更
 * - S2C_MaleUserInfoUpdate: 聊天列表中对方用户信息变更
 */
class UserInfoWsHandler {
  private wsManager: WebsocketManager | null = null;

  // 获取 WebsocketManager 实例（懒加载）
  private getWsManager(): WebsocketManager {
    if (!this.wsManager) {
      this.wsManager = WebsocketManager.getInstance();
    }
    return this.wsManager;
  }

  // 初始化用户信息处理（在 WebSocket 连接成功时调用）
  public init(): void {
    this.registerUserInfoListener();
    this.registerMaleUserInfoListener();
  }

  // 注册当前用户信息更新监听器
  private registerUserInfoListener(): void {
    const onMessage: OnWsMessageListener<S2CUserInfoUpdate> = {
      onMessage: (update: S2CUserInfoUpdate) => {
        console.log(TAG, "收到用户信息更新推送:", update);

        const userStore = useUserStore.getState();
        const currentUserId = userStore.userInfo?.userId;

        // 只处理当前用户的信息更新
        if (currentUserId && update.userId === currentUserId && update.user) {
          // 直接更新完整的用户信息
          userStore.setUserInfo(update.user);
          console.log(TAG, "用户信息已更新:", {
            userId: update.userId,
            updateTime: update.updateTime,
          });
        } else {
          console.log(TAG, "忽略其他用户的信息更新:", update.userId);
        }
      },
      resType: () => S2CUserInfoUpdate,
    };

    // 注册消息监听器
    this.getWsManager().addMessageListener("S2C_UserInfoUpdate", onMessage);
  }

  // 注册对方用户信息更新监听器（聊天列表中的用户）
  private registerMaleUserInfoListener(): void {
    const onMessage: OnWsMessageListener<S2CMaleUserInfoUpdate> = {
      onMessage: (update: S2CMaleUserInfoUpdate) => {
        console.log(TAG, "收到对方用户信息更新推送:", update);

        if (update.userId && update.user) {
          // 更新用户信息 只修改该 userId 对应的缓存条目
          useChatListStore.getState().setUserInfo(update.user);
          console.log(TAG, "对方用户信息已更新:", {
            userId: update.userId,
            updateTime: update.updateTime,
          });
        }
      },
      resType: () => S2CMaleUserInfoUpdate,
    };

    this.getWsManager().addMessageListener("S2C_MaleUserInfoUpdate", onMessage);
  }

  // 移除用户信息更新监听器
  public destroy(): void {
    if (this.wsManager) {
      this.wsManager.removeMessageListener("S2C_UserInfoUpdate");
      this.wsManager.removeMessageListener("S2C_MaleUserInfoUpdate");
      console.log(TAG, "用户信息监听器已移除");
    }
  }
}

// 导出单例
export const userInfoHandler = new UserInfoWsHandler();
```
