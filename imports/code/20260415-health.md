# health

```ts
import Router from "@koa/router";
import { prisma } from "../utils/prisma.js";
import { success } from "../utils/response.js";

export const healthRouter = new Router();

/**
 * GET /health
 * 最简存活探针 —— 仅验证 HTTP 服务可达，不依赖任何外部资源。
 * 适用于负载均衡器 / K8s liveness probe。
 */
healthRouter.get("/", (ctx) => {
  success(ctx, { status: "ok" });
});

/**
 * GET /health/ready
 * 就绪探针 —— 验证服务及其关键依赖均可正常工作。
 * 逐项检查数据库连接和核心业务表的可查询性。
 */
healthRouter.get("/ready", async (ctx) => {
  const checks: Record<
    string,
    { status: "ok" | "fail"; latencyMs: number; error?: string }
  > = {};

  // 1. 数据库连通性
  const dbStart = Date.now();
  try {
    await prisma.$queryRaw`SELECT 1`;
    checks.database = { status: "ok", latencyMs: Date.now() - dbStart };
  } catch (err) {
    checks.database = {
      status: "fail",
      latencyMs: Date.now() - dbStart,
      error: (err as Error).message,
    };
  }

  // 2. RBAC 核心表（admin_users / admin_roles / admin_permissions）
  const rbacStart = Date.now();
  try {
    await Promise.all([
      prisma.adminUser.count(),
      prisma.role.count(),
      prisma.permission.count(),
    ]);
    checks.rbac = {
      status: "ok",
      latencyMs: Date.now() - rbacStart,
    };
  } catch (err) {
    checks.rbac = {
      status: "fail",
      latencyMs: Date.now() - rbacStart,
      error: (err as Error).message,
    };
  }

  // 3. Lovia 核心表（story）
  const loviaStart = Date.now();
  try {
    await prisma.story.count();
    checks.lovia = {
      status: "ok",
      latencyMs: Date.now() - loviaStart,
    };
  } catch (err) {
    checks.lovia = {
      status: "fail",
      latencyMs: Date.now() - loviaStart,
      error: (err as Error).message,
    };
  }

  const allOk = Object.values(checks).every((c) => c.status === "ok");

  ctx.status = allOk ? 200 : 503;
  ctx.body = {
    success: allOk,
    data: {
      status: allOk ? "ready" : "degraded",
      uptime: Math.floor(process.uptime()),
      checks,
    },
  };
});

```
