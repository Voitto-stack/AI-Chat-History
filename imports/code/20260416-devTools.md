# devTools

```ts
/**
 * 开发环境调试工具
 * 在浏览器控制台通过 window.__DEV__ 访问
 *
 * 业务阶段说明：
 * - StageOne ($0.5): 注册奖励（不算正式提现）
 * - StageTwo ($4): 第1个提现阶段
 * - StageThree ($10): 第2个提现阶段
 * - StageFour ($12): 第3个提现阶段
 * - StageFive ($25): 第4个提现阶段
 * - StageSix ($50): 第5个提现阶段
 * - StageSeven ($50): 第5个提现阶段（循环重复）
 */

/* eslint-disable no-restricted-imports */
// devTools 需要在 React 外部使用，必须直接访问 Store
import { useCashoutStore } from "@/stores/cashoutStore";
import { useTaskStore } from "@/stores/taskStore";
import { useUserStore } from "@/stores/userStore";
/* eslint-enable no-restricted-imports */
import { CashoutStage, cashoutStagesList, stageAmountMap, USPaypalAccountType } from "@/types/cashout";
import { getNextStage, getStageIndex } from "@/utils/cashoutUtils";
import { UserCloudStorage, UserCloudKey } from "@/services/userCloudStorage";
import { reportPwaWithdrawPhase, normalMediumWithdraw } from "@/http/cashoutApi";
import { formatNumber } from "@/utils/format";

// 业务阶段映射（技术阶段 -> 业务阶段）
const businessStageMap: Record<CashoutStage, string> = {
  [CashoutStage.StageOne]: "注册奖励 ($0.5)",
  [CashoutStage.StageTwo]: "第1个提现 ($4)",
  [CashoutStage.StageThree]: "第2个提现 ($10)",
  [CashoutStage.StageFour]: "第3个提现 ($12)",
  [CashoutStage.StageFive]: "第4个提现 ($25)",
  [CashoutStage.StageSix]: "第5个提现 ($50)",
  [CashoutStage.StageSeven]: "第5个提现-循环 ($50)",
};

// 获取业务阶段描述
function getBusinessStageDesc(stage: CashoutStage): string {
  return businessStageMap[stage] || stage;
}

// 调试工具类型定义
interface DevTools {
  cashout: {
    /** 完成当前提现阶段，进入下一阶段 */
    finishWillCashoutStage: () => Promise<void>;
    /** 完成当前任务阶段，进入下一阶段 */
    finishCurrentTaskStage: () => Promise<void>;
    /** 设置提现阶段 */
    setWillCashoutStage: (stage: CashoutStage) => Promise<void>;
    /** 设置任务阶段 */
    setCurrentTaskStage: (stage: CashoutStage) => Promise<void>;
    /** 同步当前阶段到后端 */
    syncBackend: () => Promise<void>;
    /** 重置视频通话时长 */
    resetVideoCallTime: () => Promise<void>;
    /** 设置视频通话时长 */
    setVideoCallTime: (seconds: number) => Promise<void>;
    /** 测试提现请求 */
    testWithdraw: (params?: {
      amount?: string;
      allowPayment?: boolean;
      isLbsAmerica?: boolean;
      usPaypalAccountType?: USPaypalAccountType;
    }) => Promise<void>;
    /** 获取当前状态 */
    getState: () => {
      currentTaskStage: CashoutStage;
      willCashoutStage: CashoutStage;
      videoCallTotalSeconds: number;
    };
  };
  stores: {
    /** 获取 Cashout Store 状态 */
    cashout: () => ReturnType<typeof useCashoutStore.getState>;
    /** 获取 Task Store 状态 */
    task: () => ReturnType<typeof useTaskStore.getState>;
    /** 获取 User Store 状态 */
    user: () => ReturnType<typeof useUserStore.getState>;
  };
  cloud: {
    /** 获取云端数据 */
    get: (key: UserCloudKey) => Promise<string | null>;
    /** 设置云端数据 */
    set: (key: UserCloudKey, value: string) => Promise<void>;
    /** 清空指定的云端数据 */
    clear: (key: UserCloudKey) => Promise<void>;
  };
  /** 列出所有可用的提现阶段 */
  stages: CashoutStage[];
}

// 扩展 window 类型
declare global {
  interface Window {
    __DEV__?: DevTools;
  }
}

// 仅在开发环境中初始化
if (process.env.NODE_ENV === "development") {
  const devTools: DevTools = {
    cashout: {
      finishWillCashoutStage: async () => {
        const { willCashoutStage, setWillCashoutStage } = useCashoutStore.getState();
        const nextStage = getNextStage(willCashoutStage);
        if (nextStage) {
          setWillCashoutStage(nextStage);
          await UserCloudStorage.setValue(UserCloudKey.CashoutSuccessStage, nextStage);
          console.log(`✅ willCashoutStage: ${willCashoutStage} → ${nextStage}`);
        } else {
          console.warn("⚠️ 已经是最后一个阶段");
        }
      },

      finishCurrentTaskStage: async () => {
        const { currentTaskStage, setCurrentTaskStage, setVideoCallTotalSeconds } = useCashoutStore.getState();
        const nextStage = getNextStage(currentTaskStage);
        if (nextStage) {
          setCurrentTaskStage(nextStage);
          setVideoCallTotalSeconds(0);
          await UserCloudStorage.setValue(UserCloudKey.CashoutStage, nextStage);
          await UserCloudStorage.setValue(UserCloudKey.VideoCallTime, "0");

          // 同步到后端
          const nextPhaseIndex = getStageIndex(nextStage);
          await reportPwaWithdrawPhase(nextPhaseIndex);

          console.log(`✅ currentTaskStage: ${currentTaskStage} → ${nextStage}`);
          console.log(`✅ 已同步到后端 (phase: ${nextPhaseIndex})`);
          console.log("✅ 视频通话时长已重置");
        } else {
          console.warn("⚠️ 已经是最后一个阶段");
        }
      },

      setWillCashoutStage: async (stage: CashoutStage) => {
        if (!cashoutStagesList.includes(stage)) {
          console.error("❌ 无效的阶段:", stage);
          console.log("可用阶段:", cashoutStagesList);
          return;
        }
        const { setWillCashoutStage } = useCashoutStore.getState();
        setWillCashoutStage(stage);
        await UserCloudStorage.setValue(UserCloudKey.CashoutSuccessStage, stage);
        console.log(`✅ willCashoutStage 已设置为: ${stage}`);
      },

      setCurrentTaskStage: async (stage: CashoutStage) => {
        if (!cashoutStagesList.includes(stage)) {
          console.error("❌ 无效的阶段:", stage);
          console.log("可用阶段:", cashoutStagesList);
          return;
        }
        const { setCurrentTaskStage } = useCashoutStore.getState();
        setCurrentTaskStage(stage);
        await UserCloudStorage.setValue(UserCloudKey.CashoutStage, stage);

        // 同步到后端
        const phaseIndex = getStageIndex(stage);
        await reportPwaWithdrawPhase(phaseIndex);

        console.log(`✅ currentTaskStage 已设置为: ${stage}`);
        console.log(`✅ 已同步到后端 (phase: ${phaseIndex})`);
      },

      syncBackend: async () => {
        const { currentTaskStage } = useCashoutStore.getState();
        const phaseIndex = getStageIndex(currentTaskStage);
        await reportPwaWithdrawPhase(phaseIndex);
        console.log(`✅ 已同步当前阶段到后端: ${currentTaskStage} (phase: ${phaseIndex})`);
      },

      resetVideoCallTime: async () => {
        const { setVideoCallTotalSeconds } = useCashoutStore.getState();
        setVideoCallTotalSeconds(0);
        await UserCloudStorage.setValue(UserCloudKey.VideoCallTime, "0");
        console.log("✅ 视频通话时长已重置为 0");
      },

      setVideoCallTime: async (seconds: number) => {
        if (typeof seconds !== "number" || seconds < 0) {
          console.error("❌ 请输入有效的秒数");
          return;
        }
        const { setVideoCallTotalSeconds } = useCashoutStore.getState();
        setVideoCallTotalSeconds(seconds);
        await UserCloudStorage.setValue(UserCloudKey.VideoCallTime, String(seconds));
        console.log(`✅ 视频通话时长已设置为: ${seconds}秒 (${Math.floor(seconds / 60)}分钟)`);
      },

      testWithdraw: async (params) => {
        const { willCashoutStage } = useCashoutStore.getState();

        // 默认参数：使用当前阶段金额和测试配置
        const amount = params?.amount || formatNumber(stageAmountMap[willCashoutStage], 2);
        const allowPayment = params?.allowPayment ?? true;
        const isLbsAmerica = params?.isLbsAmerica ?? true;
        const usPaypalAccountType = params?.usPaypalAccountType ?? USPaypalAccountType.US_PAYPAL_ACCOUNT;

        console.log("🚀 测试提现请求...");
        console.log("参数:", {
          amount,
          allowPayment,
          isLbsAmerica,
          usPaypalAccountType,
        });

        try {
          const response = await normalMediumWithdraw({
            amount,
            allowPayment,
            isLbsAmerica,
            usPaypalAccountType,
          });

          console.log("✅ 提现请求成功");
          console.log("响应:", response);

          if (response.code === 0 || response.code === 1) {
            console.log("💰 提现处理成功");
          } else {
            console.warn("⚠️ 提现可能失败，code:", response.code);
          }
        } catch (error) {
          console.error("❌ 提现请求失败:", error);
        }
      },

      getState: () => {
        const { currentTaskStage, willCashoutStage, videoCallTotalSeconds } = useCashoutStore.getState();
        const state = {
          currentTaskStage,
          willCashoutStage,
          videoCallTotalSeconds,
        };
        console.table(state);

        // 显示业务阶段信息
        console.log("\n📊 业务阶段信息:");
        console.log(`   当前任务阶段: ${getBusinessStageDesc(currentTaskStage)}`);
        console.log(`   将要提现阶段: ${getBusinessStageDesc(willCashoutStage)}`);
        console.log(`   视频时长: ${Math.floor(videoCallTotalSeconds / 60)}分${videoCallTotalSeconds % 60}秒\n`);

        return state;
      },
    },

    stores: {
      cashout: () => {
        const state = useCashoutStore.getState();
        console.log("Cashout Store:", state);
        return state;
      },
      task: () => {
        const state = useTaskStore.getState();
        console.log("Task Store:", state);
        return state;
      },
      user: () => {
        const state = useUserStore.getState();
        console.log("User Store:", state);
        return state;
      },
    },

    cloud: {
      get: async (key: UserCloudKey) => {
        const value = await UserCloudStorage.getValue(key);
        console.log(`☁️ ${key}:`, value);
        return value;
      },
      set: async (key: UserCloudKey, value: string) => {
        await UserCloudStorage.setValue(key, value);
        console.log(`✅ 已设置 ${key} = ${value}`);
      },
      clear: async (key: UserCloudKey) => {
        await UserCloudStorage.setValue(key, "");
        console.log(`✅ 已清空 ${key}`);
      },
    },

    stages: cashoutStagesList,
  };

  // 暴露到全局
  window.__DEV__ = devTools;

  // 打印使用提示
  console.log("%c🛠️ 开发工具已加载", "color: #4CAF50; font-size: 14px; font-weight: bold;");
  console.log("%c使用 window.__DEV__ 访问调试工具", "color: #2196F3; font-size: 12px;");

  console.log("\n💰 业务阶段说明（共5个提现阶段）:");
  console.log("  StageOne   - 注册奖励 ($0.5)");
  console.log("  StageTwo   - 第1个提现 ($4)");
  console.log("  StageThree - 第2个提现 ($10)");
  console.log("  StageFour  - 第3个提现 ($12)");
  console.log("  StageFive  - 第4个提现 ($25)");
  console.log("  StageSix   - 第5个提现 ($50)");
  console.log("  StageSeven - 第5个提现-循环 ($50)");

  console.log("\n📝 可用命令:");
  console.log("  __DEV__.cashout.getState()                   - 查看当前状态（含业务阶段）");
  console.log("  __DEV__.cashout.testWithdraw()               - 测试提现请求（使用当前阶段金额）");
  console.log("  __DEV__.cashout.testWithdraw({amount:'10'})  - 测试提现请求（自定义金额）");
  console.log("  __DEV__.cashout.finishWillCashoutStage()     - 完成提现阶段");
  console.log("  __DEV__.cashout.finishCurrentTaskStage()     - 完成任务阶段");
  console.log("  __DEV__.cashout.syncBackend()                - 同步当前阶段到后端");
  console.log("  __DEV__.cashout.setVideoCallTime(600)        - 设置视频时长(秒)");
  console.log("  __DEV__.stores.cashout()                     - 查看 Cashout Store");
  console.log("  __DEV__.stages                               - 查看所有阶段\n");
}

export {};
```
