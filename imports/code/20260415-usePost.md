# usePost

```ts
/**
 * 帖子管理 Hook
 */

import { useCallback, useState } from "react";
import { PostServiceCommonCode } from "@heyhru/business-pwa-proto/gen/archat_api/post_api";
import { usePostStore } from "@/stores/postStore";
import { useUserStore } from "@/stores/userStore";
import { getPostsByUserId } from "@/http/postApi";

export const usePost = () => {
  const { myPosts, cursor, hasMore, isLoading, setPosts, appendPosts, setLoading } = usePostStore();
  const { userInfo } = useUserStore();
  const [error, setError] = useState<string | null>(null);
  const userId = userInfo?.userId;

  /** 拉取我的帖子列表。refresh=true 时重置游标从头拉取 */
  const fetchMyPosts = useCallback(
    async (refresh = false) => {
      if (!userId) return;
      if (isLoading) return;
      if (!refresh && !hasMore) return;

      const nextCursor = refresh ? 0 : cursor;
      setLoading(true);
      setError(null);

      try {
        const res = await getPostsByUserId({ userId, cursor: nextCursor });

        if (res.code === PostServiceCommonCode.Success) {
          const posts = res.posts ?? [];
          const opts = { cursor: res.nextCursor ?? 0, hasMore: posts.length > 0 };

          if (refresh) {
            setPosts(posts, opts);
          } else {
            appendPosts(posts, opts);
          }
        } else {
          setError(res.message || "Failed to load posts");
        }
      } catch {
        setError("Network error");
      } finally {
        setLoading(false);
      }
    },
    [userId, isLoading, hasMore, cursor, setPosts, appendPosts, setLoading, setError],
  );

  return {
    myPosts,
    isLoading,
    hasMore,
    error,
    fetchMyPosts,
  };
};
```
