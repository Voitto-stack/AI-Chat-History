import { UserInfo, Gender, UserType } from "@sitin/api-proto/gen/archat_api/user_api";
import defaultAvatarImg from "@/assets/images/common/ic_avatar_placeholder.webp";

/**
 * 默认头像占位图
 */
export const DEFAULT_AVATAR = defaultAvatarImg;
const FEMALE_BEAUTIFIED_AVATAR =
  "https://file.archat.us/cai/beautified_avatars/female/graciechen_A_color_cartoon_headshot_of_a_beautiful_woman_in_a_adcc31bb-c77d-4345-a810-d0e19325d88b_0.png";

export enum CustomAvatarType {
  Normal,
  Min,
  Mid,
}

/**
 * 获取用户头像 URL
 * 优先级：customAvatar > beautifiedAvatar > avatarUrl > 默认头像
 */
export function getAvatarUrl(
  userInfo?: UserInfo | null,
  customAvatarType: CustomAvatarType = CustomAvatarType.Mid,
): string {
  if (!userInfo) return DEFAULT_AVATAR;

  // 旧版兼容逻辑？目前原样保留此逻辑
  if (userInfo.gender === Gender.FEMALE && (userInfo.customAvatar?.midUrl ?? "").includes("/male/")) {
    return FEMALE_BEAUTIFIED_AVATAR;
  }

  // 优先使用自定义头像
  if (userInfo.customAvatar?.midUrl) {
    switch (customAvatarType) {
      case CustomAvatarType.Normal:
        return userInfo.customAvatar.url ?? DEFAULT_AVATAR;
      case CustomAvatarType.Min:
        return userInfo.customAvatar.minUrl ?? DEFAULT_AVATAR;
      case CustomAvatarType.Mid:
        return userInfo.customAvatar.midUrl ?? DEFAULT_AVATAR;
    }
  }

  // 其次使用美化头像
  if (userInfo.beautifiedAvatar) {
    return userInfo.beautifiedAvatar;
  }

  // 最后使用原始头像
  if (userInfo.avatarUrl) {
    return userInfo.avatarUrl;
  }

  return DEFAULT_AVATAR;
}

/**
 * 获取用户标签（最多返回指定数量）
 */
export function getUserTags(userInfo?: UserInfo | null, maxCount: number = 3): string[] {
  if (!userInfo?.interestTabs) return [];

  return (
    userInfo.interestTabs
      .flatMap((tab) => tab.tags?.map((tag) => tag.key))
      .filter((name): name is string => name !== undefined)
      .slice(0, maxCount) || []
  );
}

/**
 * 判断用户是否为真人
 */
export function isRealHuman(userInfo?: UserInfo | null): boolean {
  return userInfo?.userType === UserType.User;
}
