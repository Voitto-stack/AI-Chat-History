# auth

```ts
/**
 * Auth API
 * 用户认证相关接口
 */
import {
  changePasswordRequestSchema,
  changePasswordResponseSchema,
  currentUserSchema,
  loginRequestSchema,
  loginResponseSchema,
} from '@sitin/minerva-schemas';
import type {
  ChangePasswordRequest,
  ChangePasswordResponse,
  CurrentUser,
  LoginRequest,
  LoginResponse,
} from '@sitin/minerva-schemas';
import { request } from '@/request/client';

// Re-export types for external use
export type {
  LoginRequest,
  LoginResponse,
  CurrentUser,
  ChangePasswordRequest,
  ChangePasswordResponse,
};

/**
 * 用户登录
 */
export async function login(params: LoginRequest): Promise<LoginResponse> {
  const body = loginRequestSchema.parse(params);
  return request<LoginResponse>(
    '/api/auth/login',
    {
      method: 'POST',
      body,
      skipAuth: true,
    },
    loginResponseSchema,
  );
}

/**
 * 获取当前用户信息（含角色和权限）
 */
export async function getCurrentUser(): Promise<CurrentUser> {
  return request<CurrentUser>('/api/auth/me', {}, currentUserSchema);
}

/**
 * 修改密码
 */
export async function changePassword(
  params: ChangePasswordRequest,
): Promise<ChangePasswordResponse> {
  const body = changePasswordRequestSchema.parse(params);
  return request<ChangePasswordResponse>(
    '/api/auth/change-password',
    {
      method: 'POST',
      body,
    },
    changePasswordResponseSchema,
  );
}

```
