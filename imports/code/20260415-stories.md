import Router from '@koa/router';
import {
  createStoryRequestSchema,
  updateStoryRequestSchema,
  deleteStoryRequestSchema,
  getStoryListRequestSchema,
  type StoryResponse,
} from '@sitin/minerva-schemas';
import { prisma } from '../../utils/prisma.js';
import { success } from '../../utils/response.js';
import { AppError } from '../../utils/AppError.js';
import { authMiddleware, requirePermission } from '../../middlewares/auth.js';
import {
  uploadPicturesToOSS,
  parsePicturesFromDB,
  serializePicturesForDB,
} from '../../utils/oss.js';

export const storiesRouter = new Router();

// Apply auth middleware to all routes
storiesRouter.use(authMiddleware);

/**
 * 将数据库 story 记录转换为 API 响应格式
 */
function transformStoryToResponse(story: {
  id: bigint;
  from_user_id: number;
  to_user_id: number;
  title: string | null;
  content: string | null;
  pictures: string | null;
  created_at: Date;
}): StoryResponse {
  return {
    id: Number(story.id),
    fromUserId: story.from_user_id,
    toUserId: story.to_user_id,
    title: story.title,
    content: story.content,
    pictures: parsePicturesFromDB(story.pictures),
    createdAt: story.created_at.getTime(),
  };
}

/**
 * POST /api/lovia/stories
 * 创建故事
 * 权限: lovia.story.create
 */
storiesRouter.post(
  '/',
  requirePermission('lovia.story.create'),
  async (ctx) => {
    const input = createStoryRequestSchema.parse(ctx.request.body);

    // 上传图片到 OSS
    const uploadedPictures = await uploadPicturesToOSS(input.pictures);

    // 创建故事记录
    const story = await prisma.story.create({
      data: {
        from_user_id: input.from_user_id,
        to_user_id: input.to_user_id,
        title: input.title || null,
        content: input.content,
        pictures: serializePicturesForDB(uploadedPictures),
      },
    });

    success(ctx, transformStoryToResponse(story), 201);
  }
);

/**
 * GET /api/lovia/stories
 * 获取故事列表（分页）
 * 权限: lovia.story.read
 */
storiesRouter.get('/', requirePermission('lovia.story.read'), async (ctx) => {
  const { limit, offset } = getStoryListRequestSchema.parse(ctx.query);

  // 查询故事列表（排除软删除的记录）
  const stories = await prisma.story.findMany({
    where: {
      deleted_at: null,
    },
    orderBy: {
      created_at: 'desc',
    },
    skip: offset,
    take: limit + 1, // 多取一条用于判断是否还有更多数据
  });

  // 判断是否有更多数据
  const hasMore = stories.length > limit;
  const resultStories = hasMore ? stories.slice(0, limit) : stories;

  // 计算下一页 offset
  const nextOffset = hasMore ? String(offset + limit) : '-1';

  success(ctx, {
    storyList: resultStories.map(transformStoryToResponse),
    nextOffset,
  });
});

/**
 * PATCH /api/lovia/stories
 * 更新故事（需要完整主键）
 * 权限: lovia.story.update
 */
storiesRouter.patch(
  '/',
  requirePermission('lovia.story.update'),
  async (ctx) => {
    const input = updateStoryRequestSchema.parse(ctx.request.body);

    // 检查故事是否存在
    const existing = await prisma.story.findUnique({
      where: {
        id_from_user_id_to_user_id: {
          id: BigInt(input.id),
          from_user_id: input.from_user_id,
          to_user_id: input.to_user_id,
        },
      },
    });

    if (!existing || existing.deleted_at !== null) {
      throw AppError.notFound('故事不存在');
    }

    // 准备更新数据
    const updateData: {
      title?: string | null;
      content?: string;
      pictures?: string;
      updated_at: Date;
    } = {
      updated_at: new Date(),
    };

    if (input.title !== undefined) {
      updateData.title = input.title || null;
    }

    if (input.content !== undefined) {
      updateData.content = input.content;
    }

    if (input.pictures !== undefined) {
      // 上传新图片到 OSS
      const uploadedPictures = await uploadPicturesToOSS(input.pictures);
      updateData.pictures = serializePicturesForDB(uploadedPictures);
    }

    // 执行更新
    const story = await prisma.story.update({
      where: {
        id_from_user_id_to_user_id: {
          id: BigInt(input.id),
          from_user_id: input.from_user_id,
          to_user_id: input.to_user_id,
        },
      },
      data: updateData,
    });

    success(ctx, transformStoryToResponse(story));
  }
);

/**
 * DELETE /api/lovia/stories
 * 批量删除故事（软删除，需要完整主键）
 * 权限: lovia.story.delete
 */
storiesRouter.delete(
  '/',
  requirePermission('lovia.story.delete'),
  async (ctx) => {
    const input = deleteStoryRequestSchema.parse(ctx.request.body);

    // 软删除：设置 deleted_at
    const deleteTime = new Date();

    // 使用事务批量软删除
    await prisma.$transaction(
      input.stories.map((item) =>
        prisma.story.update({
          where: {
            id_from_user_id_to_user_id: {
              id: BigInt(item.id),
              from_user_id: item.from_user_id,
              to_user_id: item.to_user_id,
            },
          },
          data: {
            deleted_at: deleteTime,
          },
        })
      )
    );

    success(ctx, { message: '删除成功' });
  }
);
