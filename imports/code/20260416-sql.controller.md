# sql.controller

```ts
import { type FastifyInstance } from "fastify";
import { postSqlParse, postSqlExecute, postSqlDatabases, postSqlTables } from "./sql.service.js";

export function sqlController(app: FastifyInstance) {
  app.post("/sql/parse", postSqlParse);
  app.post("/sql/execute", postSqlExecute);
  app.post("/sql/databases", postSqlDatabases);
  app.post("/sql/tables", postSqlTables);
}
```
