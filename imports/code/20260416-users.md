# users

```ts
import { client } from "./client";
import { AuthUser, Role } from "../store/auth";

export interface CreateUserPayload {
  username: string;
  email: string;
  password: string;
  role: Role;
}

export interface UpdateUserPayload {
  id: string;
  email?: string;
  role?: Role;
}

export const listUsers = () => client.post<AuthUser[]>("/users/list");

export const createUser = (payload: CreateUserPayload) => client.post<AuthUser>("/users/create", payload);

export const updateUser = (payload: UpdateUserPayload) => client.post<AuthUser>("/users/update", payload);

export const deleteUser = (id: string) => client.post("/users/delete", { id });

```
