# call

```ts
/**
 * 通话相关通用类型定义
 * 被 webCallManager / nativeCallManager / mockCallManager 等共享
 */

import { MockVideoSource, UserInfo } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";

/**
 * 通话状态枚举
 */
export enum CallState {
  /** 空闲状态/初始化状态 - 无通话 */
  Idle = "Idle",
  /** 来电中 - 等待接听 */
  Incoming = "Incoming",
  /** 已接听 - 接听 */
  Accept = "Accept",
  /** 已拒绝 - 通话被拒绝 */
  Rejected = "Rejected",
  /** 已连接 - 通话进行中 */
  Connected = "Connected",
}

/**
 * 扩展的通话类型
 * 值与 TUICallType 保持一致（AUDIO_CALL=1, VIDEO_CALL=2），避免 SDK 耦合
 */
export enum CallType {
  /** 语音通话 */
  AUDIO_CALL = 1,
  /** 视频通话 */
  VIDEO_CALL = 2,
  /** Mock通话 */
  MOCK_CALL = 3,
}

/**
 * 通话会话信息接口
 */
export interface ConnectSession {
  /** 房间ID */
  roomId: string;
  /** 远程用户ID */
  remoteUserId: string;
  /** 当前用户ID */
  userId: string;
  /** 通话类型 (1=语音, 2=视频 3=mock) */
  callType: CallType;
  /** 通话状态 */
  callState: CallState;
  /** 开始呼叫时间戳 */
  callInvitedTime: number;
  /** 连接成功时间戳 */
  callBeginTime: number;
  /** 通话结束时间戳 */
  callEndTime: number;
  /** 上行网络质量 (0-6) */
  uplinkNetQuality: number;
  /** 下行网络质量 (0-6) */
  downlinkNetQuality: number;
}

/**
 * Mock 通话类型枚举
 */
export enum MockCallType {
  /** 引导模式 - 先展示视频规则引导，再发起 mock 通话 */
  Guide = "Guide",
  /** 派发模式 - 仅响铃，接听后提示连接失败并关闭 */
  Dispatch = "Dispatch",
  /** 正常模式 - 标准 mock 视频流程 */
  Normal = "Normal",
}

/**
 * Mock 通话状态枚举
 */
export enum MockCallState {
  /** 空闲状态 */
  Idle = "Idle",
  /** 连接中 - 正在获取视频资源 + 权限检查 */
  Connecting = "Connecting",
  /** 已连接 - Mock 通话进行中 */
  Connected = "Connected",
  /** 已结束 */
  Ended = "Ended",
}

/**
 * Mock 通话会话信息
 */
export interface MockConnectSession {
  /** 远端视频 URL */
  mockVideoUrl: string;
  /** 视频 ID */
  mockVideoId: number;
  /** 关联用户信息 */
  remoteUserInfo: UserInfo;
  /** Mock 通话状态 */
  callState: MockCallState;
  /** 通话开始时间戳 */
  callBeginTime: number;
  /** 通话结束时间戳 */
  callEndTime: number;
  /** 剩余测试视频数量 */
  remainingCount: number;
  /** 订单 ID（创建订单后获得） */
  orderId: number;
  /** Mock 通话类型 */
  mockCallType: MockCallType;
  /** 通话单价（美元） */
  price: number;
  /** 获取视频资源的来源 */
  source: MockVideoSource;
}

```
