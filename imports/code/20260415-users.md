import Router from '@koa/router';
import {
  assignUserRolesRequestSchema,
  createUserRequestSchema,
  resetUserPasswordRequestSchema,
  updateUserRequestSchema,
} from '@sitin/minerva-schemas';
import { prisma } from '../../utils/prisma.js';
import { success } from '../../utils/response.js';
import { AppError } from '../../utils/AppError.js';
import { authMiddleware, requirePermission } from '../../middlewares/auth.js';

export const usersRouter = new Router();

// Apply auth middleware to all routes
usersRouter.use(authMiddleware);

/**
 * GET /api/rbac/users
 * List all admin users
 */
usersRouter.get('/', requirePermission('rbac.user.read'), async (ctx) => {
  const users = await prisma.adminUser.findMany({
    select: {
      id: true,
      sub: true,
      displayName: true,
      avatarUrl: true,
      tenantKey: true,
      unionId: true,
      mobile: true,
      userType: true,
      createdAt: true,
      userRoles: {
        include: {
          role: {
            select: {
              id: true,
              name: true,
            },
          },
        },
      },
    },
    orderBy: { id: 'desc' },
  });

  const result = users.map((user) => ({
    ...user,
    roles: user.userRoles.map((ur) => ur.role),
    userRoles: undefined,
  }));

  success(ctx, result);
});

/**
 * GET /api/rbac/users/:id
 * Get user by ID
 */
usersRouter.get('/:id', requirePermission('rbac.user.read'), async (ctx) => {
  const { id } = ctx.params;

  const user = await prisma.adminUser.findUnique({
    where: { id },
    select: {
      id: true,
      sub: true,
      displayName: true,
      avatarUrl: true,
      tenantKey: true,
      unionId: true,
      mobile: true,
      userType: true,
      createdAt: true,
      updatedAt: true,
      userRoles: {
        include: {
          role: {
            include: {
              rolePermissions: {
                include: {
                  permission: true,
                },
              },
            },
          },
        },
      },
    },
  });

  if (!user) {
    throw AppError.notFound('User not found');
  }

  // Extract roles and permissions
  const roles = user.userRoles.map((ur) => ({
    id: ur.role.id,
    name: ur.role.name,
    permissions: ur.role.rolePermissions.map((rp) => rp.permission.key),
  }));

  success(ctx, {
    ...user,
    roles,
    userRoles: undefined,
  });
});

/**
 * POST /api/rbac/users
 * Create a new admin user
 */
usersRouter.post('/', requirePermission('rbac.user.write'), async (ctx) => {
  const body = createUserRequestSchema.parse(ctx.request.body);

  // Check if sub already exists
  const existing = await prisma.adminUser.findUnique({
    where: { sub: body.sub },
  });

  if (existing) {
    throw AppError.conflict('User with this sub already exists', 'SUB_EXISTS');
  }

  // Create user with roles (SYSTEM type for manually created users)
  const user = await prisma.adminUser.create({
    data: {
      sub: body.sub,
      password: body.password,
      displayName: body.displayName,
      userType: 'SYSTEM',
      userRoles: body.roleIds
        ? {
            create: body.roleIds.map((roleId) => ({
              roleId,
            })),
          }
        : undefined,
    },
    include: {
      userRoles: {
        include: {
          role: true,
        },
      },
    },
  });

  success(
    ctx,
    {
      id: user.id,
      sub: user.sub,
      displayName: user.displayName,
      userType: user.userType,
      createdAt: user.createdAt,
      roles: user.userRoles.map((ur) => ({
        id: ur.role.id,
        name: ur.role.name,
      })),
    },
    201
  );
});

/**
 * PATCH /api/rbac/users/:id
 * Update an admin user
 */
usersRouter.patch('/:id', requirePermission('rbac.user.write'), async (ctx) => {
  const { id } = ctx.params;
  const body = updateUserRequestSchema.parse(ctx.request.body);

  const user = await prisma.adminUser.update({
    where: { id },
    data: body,
    select: {
      id: true,
      sub: true,
      displayName: true,
      avatarUrl: true,
      mobile: true,
      userType: true,
      updatedAt: true,
    },
  });

  success(ctx, user);
});

/**
 * DELETE /api/rbac/users/:id
 * Delete an admin user
 */
usersRouter.delete(
  '/:id',
  requirePermission('rbac.user.delete'),
  async (ctx) => {
    const { id } = ctx.params;

    // Prevent deleting yourself
    if (id === ctx.state.user.userId) {
      throw AppError.badRequest('Cannot delete your own account');
    }

    // Delete user roles first, then user
    await prisma.$transaction([
      prisma.userRole.deleteMany({ where: { userId: id } }),
      prisma.adminUser.delete({ where: { id } }),
    ]);

    success(ctx, { message: 'User deleted successfully' });
  }
);

/**
 * PUT /api/rbac/users/:id/roles
 * Assign roles to user (replaces existing roles)
 */
usersRouter.put(
  '/:id/roles',
  requirePermission('rbac.user.write'),
  async (ctx) => {
    const { id } = ctx.params;
    const body = assignUserRolesRequestSchema.parse(ctx.request.body);

    // Verify user exists
    const user = await prisma.adminUser.findUnique({ where: { id } });
    if (!user) {
      throw AppError.notFound('User not found');
    }

    // Replace roles in a transaction
    await prisma.$transaction([
      prisma.userRole.deleteMany({ where: { userId: id } }),
      prisma.userRole.createMany({
        data: body.roleIds.map((roleId) => ({
          userId: id,
          roleId,
        })),
      }),
    ]);

    // Fetch updated user with roles
    const updatedUser = await prisma.adminUser.findUnique({
      where: { id },
      include: {
        userRoles: {
          include: {
            role: true,
          },
        },
      },
    });

    success(ctx, {
      id: updatedUser!.id,
      sub: updatedUser!.sub,
      roles: updatedUser!.userRoles.map((ur) => ({
        id: ur.role.id,
        name: ur.role.name,
      })),
    });
  }
);

/**
 * POST /api/rbac/users/:id/reset-password
 * Reset user's password (admin action)
 */
usersRouter.post(
  '/:id/reset-password',
  requirePermission('rbac.user.write'),
  async (ctx) => {
    const { id } = ctx.params;
    const body = resetUserPasswordRequestSchema.parse(ctx.request.body);

    const user = await prisma.adminUser.findUnique({ where: { id } });
    if (!user) {
      throw AppError.notFound('User not found');
    }

    await prisma.adminUser.update({
      where: { id },
      data: { password: body.newPassword },
    });

    success(ctx, { message: 'Password reset successfully' });
  }
);
