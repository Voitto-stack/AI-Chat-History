# client

```ts
import { create } from "@heyhru/common-util-fetch";
import { getStoredToken, removeStoredToken } from "./token";

export const client = create({
  baseURL: process.env.EXPO_PUBLIC_API_BASE_URL ?? "http://localhost:3001",
  getToken: () => getStoredToken(),
  onUnauthorized: () => {
    removeStoredToken();
  },
});

```
