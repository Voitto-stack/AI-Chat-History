/**
 * 上传用户录制视频
 * @param params.videoBlob 录制的视频 Blob
 * @param params.verifyPass 视频验证是否通过
 * @param params.failReason 验证失败原因（可选）
 * @param params.orderId 关联的订单 ID（可选）
 */
export async function uploadUserVideo(params: {
  videoBlob: Blob;
  verifyPass: boolean;
  failReason?: string;
  orderId?: number;
}): Promise<UploadUserVideoResponse> {
  const arrayBuffer = await params.videoBlob.arrayBuffer();
  const file = new Uint8Array(arrayBuffer);

  const request: UploadUserVideoRequest = {
    file,
    fileType: "webm",
    verifyPass: params.verifyPass,
    failReason: params.failReason,
    orderId: params.orderId,
  };

  return await httpClient.requestPost2(UploadUserVideoRequest, request, UploadUserVideoResponse);
}
