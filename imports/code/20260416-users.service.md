# users.service

```ts
import { type FastifyRequest, type FastifyReply } from "fastify";
import { hashPassword } from "@heyhru/server-util-crypto";
import {
  listUsers as listUserRows,
  getUserById as getUserRowById,
  getUserByUsername as getUserByUsernameRow,
  createUserRow,
  updateUserRow,
  deleteUser as deleteUserRow,
  updatePasswordHash as updatePasswordHashRow,
} from "./users.model.js";

export function getUserByUsername(username: string) {
  return getUserByUsernameRow(username);
}

export function updateUserPassword(id: string, hash: string) {
  return updatePasswordHashRow(id, hash);
}

export async function userList(_req: FastifyRequest, reply: FastifyReply) {
  return reply.send(await listUserRows());
}

export async function userGet(req: FastifyRequest, reply: FastifyReply) {
  const { id } = (req.body ?? {}) as Record<string, string>;
  const user = await getUserRowById(id);
  if (!user) return reply.code(404).send({ error: "未找到" });
  return reply.send(user);
}

export async function userCreate(req: FastifyRequest, reply: FastifyReply) {
  const body = (req.body ?? {}) as {
    username: string;
    email: string;
    password: string;
    role: string;
  };
  const hash = await hashPassword(body.password);
  const user = await createUserRow(body.username, body.email, hash, body.role);
  return reply.code(201).send(user);
}

export async function userUpdate(req: FastifyRequest, reply: FastifyReply) {
  const { id, ...rest } = (req.body ?? {}) as { id: string; email?: string; role?: string };
  const user = await updateUserRow(id, rest);
  if (!user) return reply.code(404).send({ error: "未找到" });
  return reply.send(user);
}

export async function userDelete(req: FastifyRequest, reply: FastifyReply) {
  const { id } = (req.body ?? {}) as Record<string, string>;
  await deleteUserRow(id);
  return reply.code(204).send();
}
```
