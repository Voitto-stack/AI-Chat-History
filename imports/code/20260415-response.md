# response

```ts
import type { Context } from 'koa';
import type { ApiResponse, ApiSuccessResponse, ApiErrorResponse } from '@sitin/minerva-schemas';

// Re-export types from shared schemas for convenience
export type { ApiResponse, ApiSuccessResponse, ApiErrorResponse };

/**
 * 统一成功响应辅助函数
 *
 * 用法：success(ctx, data) 或 success(ctx, data, 201)
 *
 * 错误响应请直接 throw AppError，由全局 errorHandler 中间件统一拦截处理。
 */
export function success<T>(ctx: Context, data: T, statusCode: number = 200) {
  ctx.status = statusCode;
  ctx.body = {
    success: true,
    data,
  } satisfies ApiSuccessResponse<T>;
}

```
