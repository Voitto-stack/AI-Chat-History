# referral

```ts
/**
 * 邀请码（referral code）相关工具函数
 */

import { STORAGE_KEYS } from "@/constants/storageKeys";

/**
 * 从 URL 参数中获取邀请码
 * 支持的参数名：c
 * @returns 邀请码，未找到返回 undefined
 */
export function getInviteCodeFromUrl(): string | undefined {
  const params = new URLSearchParams(window.location.search);
  return params.get("c") || undefined;
}

/**
 * 保存邀请码到 localStorage（防止页面刷新丢失）
 * @param inviteCode 邀请码
 */
export function saveInviteCodeToStorage(inviteCode: string): void {
  if (inviteCode) {
    localStorage.setItem(STORAGE_KEYS.BFF_INVITE_CODE, inviteCode);
  }
}

/**
 * 获取已保存的邀请码（从 localStorage 或 URL）
 * 优先从 URL 读取，如果 URL 中没有则从 localStorage 读取
 * @returns 邀请码，未找到返回 undefined
 */
export function getSavedInviteCode(): string | undefined {
  // 优先从 URL 获取（如果 URL 中有，说明是新的邀请链接）
  const urlCode = getInviteCodeFromUrl();
  if (urlCode) {
    // 保存到 localStorage，防止刷新丢失
    saveInviteCodeToStorage(urlCode);
    return urlCode;
  }

  // URL 中没有，尝试从 localStorage 读取
  const storedCode = localStorage.getItem(STORAGE_KEYS.BFF_INVITE_CODE);
  return storedCode || undefined;
}

/**
 * 获取用于注册的邀请码（仅首次使用）
 * 只有新用户首次注册时才返回邀请码，已使用过的返回 undefined
 * @returns 邀请码，如果已使用过或不存在则返回 undefined
 */
export function getReferralCodeForRegistration(): string | undefined {
  // 检查是否已经使用过邀请码
  const hasUsed = localStorage.getItem(STORAGE_KEYS.REFERRAL_CODE_USED);
  if (hasUsed === "true") {
    return undefined;
  }

  // 获取邀请码（优先从 URL，否则从 localStorage）
  const inviteCode = getSavedInviteCode();
  if (!inviteCode) {
    return undefined;
  }

  // 标记为已使用（设备级，防止同一��备多次注册都算邀请）
  localStorage.setItem(STORAGE_KEYS.REFERRAL_CODE_USED, "true");

  return inviteCode;
}
```
