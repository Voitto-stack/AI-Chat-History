# auth.service

```ts
import { type FastifyRequest, type FastifyReply } from "fastify";
import { signToken } from "@heyhru/server-plugin-jwt";
import { hashPassword, verifyPassword } from "@heyhru/server-util-crypto";
import { config } from "../config.js";
import { getUserByUsername, updateUserPassword } from "@heyhru/business-dms-user";
import { authLoginTotal } from "../metrics.js";

export async function authLogin(req: FastifyRequest, reply: FastifyReply) {
  const { username, password } = (req.body ?? {}) as Record<string, string>;
  if (!username || !password) {
    return reply.code(400).send({ error: "用户名和密码不能为空" });
  }

  const user = (await getUserByUsername(username)) as Record<string, unknown> | undefined;
  if (!user || !(await verifyPassword(password, user.password_hash as string))) {
    req.log.warn("Login failed for user: %s", username);
    authLoginTotal.inc({ status: "fail" });
    return reply.code(401).send({ error: "用户名或密码错误" });
  }

  const payload = { id: user.id, username: user.username, role: user.role };
  const token = signToken(payload, config.jwtSecret, config.jwtExpiresIn);

  authLoginTotal.inc({ status: "success" });
  req.log.info("User logged in: %s", username);
  return reply.send({ ...payload, token });
}

export function authLogout(_req: FastifyRequest, reply: FastifyReply) {
  return reply.send({ ok: true });
}

export function authMe(req: FastifyRequest, reply: FastifyReply) {
  const { id, username, role } = req.user;
  return reply.send({ id, username, role });
}

export async function authChangePassword(req: FastifyRequest, reply: FastifyReply) {
  const { currentPassword, newPassword } = (req.body ?? {}) as Record<string, string>;
  if (!currentPassword || !newPassword) {
    return reply.code(400).send({ error: "当前密码和新密码不能为空" });
  }

  const user = (await getUserByUsername(req.user.username)) as Record<string, unknown> | undefined;
  if (!user || !(await verifyPassword(currentPassword, user.password_hash as string))) {
    return reply.code(400).send({ error: "当前密码不正确" });
  }

  const hash = await hashPassword(newPassword);
  await updateUserPassword(user.id as string, hash);

  req.log.info("Password changed for user: %s", req.user.username);
  return reply.send({ ok: true });
}
```
