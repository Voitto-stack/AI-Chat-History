# Skeleton

```tsx
import Spinner from "./Spinner";

interface SkeletonBoxProps {
  className?: string;
  width?: string | number;
  height?: string | number;
  rounded?: string;
}

function Box({ className = "", width, height, rounded = "rounded-lg" }: SkeletonBoxProps) {
  return (
    <div
      className={`skeleton-shimmer ${rounded} ${className}`}
      style={{ width: width ?? "100%", height: height ?? 16 }}
    />
  );
}

function Circle({ size = 48, className = "" }: { size?: number; className?: string }) {
  return (
    <div className={`skeleton-shimmer rounded-full shrink-0 ${className}`} style={{ width: size, height: size }} />
  );
}

function Line({ width = "100%", className = "" }: { width?: string | number; className?: string }) {
  return <div className={`skeleton-shimmer rounded-full ${className}`} style={{ width, height: 12 }} />;
}

/** 聊天列表骨架 */
function ChatListSkeleton({ count = 6 }: { count?: number }) {
  return (
    <div className="flex flex-col px-4">
      {Array.from({ length: count }).map((_, i) => (
        <div key={i} className="flex items-center gap-3 py-[10px] border-b border-[#f0f0f0]">
          <Circle size={60} />
          <div className="flex-1 flex flex-col gap-2">
            <Line width="40%" />
            <Line width="70%" />
          </div>
          <Line width={50} />
        </div>
      ))}
    </div>
  );
}

/** 记录列表骨架 */
function RecordListSkeleton({ count = 4 }: { count?: number }) {
  return (
    <div className="flex flex-col gap-3 px-4">
      {Array.from({ length: count }).map((_, i) => (
        <div key={i} className="flex flex-col gap-2 p-4 rounded-2xl bg-white">
          <Line width="30%" />
          <div className="flex items-center gap-3">
            <Circle size={40} />
            <div className="flex-1 flex flex-col gap-1.5">
              <Line width="50%" />
              <Line width="35%" />
            </div>
            <Line width={60} />
          </div>
        </div>
      ))}
    </div>
  );
}

/** 通用页面骨架（Suspense fallback） */
function PageSkeleton() {
  return (
    <div className="flex h-full w-full items-center justify-center bg-white">
      <Spinner />
    </div>
  );
}

/** Home 页骨架：TopBar + 任务卡片 */
function HomeSkeleton() {
  return (
    <div className="h-full bg-white animate-fade-in pt-[var(--sat,0px)]">
      {/* TopBar */}
      <div className="flex items-center justify-between h-[58px] px-4 py-3.5 border-b border-gray-200">
        <Box width={100} height={30} rounded="rounded-[20px]" />
        <Box width={120} height={16} rounded="rounded-full" />
      </div>
      {/* 任务卡片区域 */}
      <div className="px-4 pt-5 pb-8">
        <Box height={280} rounded="rounded-[20px]" className="mb-4" />
        <Box height={160} rounded="rounded-[20px]" className="mb-4" />
        <Box height={120} rounded="rounded-[20px]" />
      </div>
    </div>
  );
}

/** Earnings 页骨架：TopBar + 图表卡片 + 明细 */
function EarningsSkeleton() {
  return (
    <div className="h-full bg-[#EBF3FE] animate-fade-in pt-[var(--sat,0px)]">
      {/* TopBar */}
      <div className="flex items-center justify-between h-[58px] px-4 py-3.5 border-b border-gray-200 bg-white">
        <Box width={100} height={30} rounded="rounded-[20px]" />
        <Box width={120} height={16} rounded="rounded-full" />
      </div>
      {/* 图表卡片 */}
      <div className="m-4 p-3.5 rounded-xl bg-white">
        <Line width="40%" className="mb-3" />
        <Line width={160} className="mb-4" />
        <div className="flex items-end justify-between gap-2 h-[160px] px-2 pb-4">
          {Array.from({ length: 7 }).map((_, i) => (
            <div key={i} className="flex-1 flex flex-col items-center gap-1">
              <Box width={24} height={40 + i * 12} rounded="rounded" />
              <Box width={20} height={10} rounded="rounded" />
            </div>
          ))}
        </div>
      </div>
      {/* 明细卡片 */}
      <div className="mx-4 flex flex-col gap-3">
        <Box height={80} rounded="rounded-xl" />
        <Box height={80} rounded="rounded-xl" />
      </div>
    </div>
  );
}

/** Chats 页骨架：标题 + 聊天列表 */
function ChatsSkeleton() {
  return (
    <div className="flex h-full flex-col bg-white px-4 animate-fade-in pt-[var(--sat,0px)]">
      <div className="flex items-center justify-between pt-[25px] pb-[16px]">
        <Box width={100} height={34} rounded="rounded-lg" />
        <Box width={88} height={30} rounded="rounded-[18px]" />
      </div>
      <ChatListSkeleton />
    </div>
  );
}

/** Me 页骨架：头部背景 + 头像 + 帖子 */
function MeSkeleton() {
  return (
    <div className="flex flex-col w-full h-full bg-white animate-fade-in pt-[var(--sat,0px)]">
      {/* 头部背景 */}
      <Box height={280} rounded="rounded-none" className="bg-gray-200" />
      {/* 帖子网格 */}
      <div className="grid grid-cols-3 gap-1 p-1 mt-4">
        {Array.from({ length: 6 }).map((_, i) => (
          <Box key={i} height={120} rounded="rounded" />
        ))}
      </div>
    </div>
  );
}

/** Live 页骨架：全屏相机占位 */
function LiveSkeleton() {
  return (
    <div className="relative w-full h-full bg-black animate-fade-in">
      <div className="absolute inset-0 flex items-center justify-center">
        <div className="w-12 h-12 border-4 border-white/20 border-t-white rounded-full animate-spin" />
      </div>
    </div>
  );
}

const Skeleton = {
  Box,
  Circle,
  Line,
  ChatList: ChatListSkeleton,
  RecordList: RecordListSkeleton,
  Page: PageSkeleton,
  Home: HomeSkeleton,
  Earnings: EarningsSkeleton,
  Chats: ChatsSkeleton,
  Me: MeSkeleton,
  Live: LiveSkeleton,
};

export default Skeleton;
```
