# PostList

```tsx
import { useEffect, useRef } from "react";
import { usePost } from "@/hooks/usePost";
import EmptyView from "@/components/EmptyView";
import PostCard from "@/components/PostCard";
import Spinner from "@/components/Spinner";

export default function PostList() {
  const { myPosts, isLoading, hasMore, fetchMyPosts } = usePost();
  const sentinelRef = useRef<HTMLDivElement>(null);
  const hasFetchedOnMountRef = useRef(false);

  // 每次进入页面时都刷新一次
  useEffect(() => {
    if (hasFetchedOnMountRef.current) return;
    hasFetchedOnMountRef.current = true;
    fetchMyPosts(true);
  }, [fetchMyPosts]);

  // 滚动到底部加载更多
  useEffect(() => {
    if (!sentinelRef.current) return;

    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting && hasMore && !isLoading) {
          fetchMyPosts();
        }
      },
      { rootMargin: "200px" },
    );

    observer.observe(sentinelRef.current);
    return () => observer.disconnect();
  }, [hasMore, isLoading, fetchMyPosts]);

  // 首次加载态
  if (isLoading && myPosts.length === 0) {
    return (
      <div className="flex justify-center py-12">
        <Spinner size={24} />
      </div>
    );
  }

  // 空态
  if (!isLoading && myPosts.length === 0) {
    return <EmptyView className="py-12" />;
  }

  return (
    <div className="flex flex-col gap-3 px-4">
      {myPosts.map((post) => (
        <PostCard key={post.id} post={post} />
      ))}

      {/* 加载更多哨兵 */}
      <div ref={sentinelRef} className="h-px" />

      {isLoading && (
        <div className="flex justify-center py-4">
          <Spinner size={20} />
        </div>
      )}

      {!hasMore && myPosts.length > 0 && <p className="text-center text-xs text-gray-300 py-4">No more posts</p>}
    </div>
  );
}

```
