# TaskItem

```tsx
// 任务项组件

import { useTask } from "@/hooks/useTask";
import { TaskId } from "@/types/task";
import checkIcon from "@/assets/images/cash_v2/cashout_check.webp";
import noCheckIcon from "@/assets/images/cash_v2/cashout_no_check.webp";

// 提现页面任务文案映射
const TASK_TEXT_MAP: Partial<Record<TaskId, string>> = {
  [TaskId.Register]: "On boarding  0.5💲",
  [TaskId.CameraPermission]: "APK Camera Access",
  [TaskId.MicrophonePermission]: "APK Microphone Access",
  [TaskId.SecondEarn]: "Video calls [2.1💲]",
  [TaskId.FaceVerify]: "Verify your avatar [0.9💲]",
  [TaskId.InstallApk]: "Install Apk [1💲]",
  [TaskId.ThirdEarn]: "IG & Video calls [7💲]",
  [TaskId.LocationPermission]: "APK Location Access",
  [TaskId.FourthEarn]: "IG & Video calls [10💲]",
  [TaskId.FourthVideoDuration]: "Video calls >= 2min",
  [TaskId.FifthEarn]: "IG & Video calls [20💲]",
  [TaskId.FifthVideoDuration]: "Video calls >= 5min",
  [TaskId.SixthEarn]: "IG & Video calls [40💲]",
  [TaskId.SixthVideoDuration]: "Video calls >= 10min",
  [TaskId.SeventhEarn]: "IG & Video calls [45💲]",
  [TaskId.SeventhVideoDuration]: "Video calls >= 10min",
};

const getTitle = (taskId: TaskId) => TASK_TEXT_MAP[taskId] || "";

interface TaskItemProps {
  task: {
    taskId: TaskId;
    completed: boolean;
  };
}

export function TaskItem({ task }: TaskItemProps) {
  const { handleTaskAction } = useTask();

  const handleClick = () => {
    if (!task.completed) {
      handleTaskAction(task.taskId);
    }
  };

  return (
    <div
      className="flex items-center gap-[6px] pb-3 border-b-[0.5px] border-[rgba(240,240,240,1)] last:pb-0 last:border-b-0 text-[15px] font-[Pangram]"
      onClick={handleClick}
      style={{
        cursor: task.completed ? "default" : "pointer",
        color: task.completed ? "rgba(60, 60, 67, 0.30)" : "#000",
        textDecoration: task.completed ? "line-through" : "none",
      }}
    >
      <img
        className="w-4 h-4 shrink-0"
        src={task.completed ? checkIcon : noCheckIcon}
        alt={task.completed ? "completed" : "not completed"}
      />
      {getTitle(task.taskId)}
    </div>
  );
}

```
