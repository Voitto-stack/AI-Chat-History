# Toast

```tsx
import { createPortal } from "react-dom";
import { useToast } from "@/hooks/useToast";
import type { ToastType } from "@/stores/toastStore";

const BG_MAP: Record<ToastType, string> = {
  success: "bg-success",
  error: "bg-[#ff3b30]",
  info: "bg-brand",
  warning: "bg-warning",
};

export default function Toast() {
  const toasts = useToast((s) => s.toasts);
  const removeToast = useToast((s) => s.removeToast);

  if (toasts.length === 0) return null;

  return createPortal(
    <div className="fixed top-0 left-0 right-0 z-[10000] flex flex-col items-center gap-2 pt-[calc(var(--sat,0px)+60px)] px-4 pointer-events-none">
      {toasts.map((toast) => (
        <div
          key={toast.id}
          className={`${toast.closing ? "animate-toast-out" : "animate-toast-in"} ${BG_MAP[toast.type]} pointer-events-auto rounded-2xl px-4 py-3 shadow-lg w-full max-w-[320px]`}
          onAnimationEnd={() => {
            if (toast.closing) removeToast(toast.id);
          }}
        >
          <span className="text-white text-sm font-medium text-center block">{toast.message}</span>
        </div>
      ))}
    </div>,
    document.body,
  );
}

```
