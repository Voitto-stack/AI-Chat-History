# oss

```ts
import OSS from 'ali-oss';
import crypto from 'crypto';
import type { StoryPictureInput, StoryPictureResponse } from '@sitin/minerva-schemas';

/**
 * OSS 配置
 * 生产环境应从环境变量读取
 */
const ossConfig = {
  accessKeyId: process.env.OSS_ACCESS_KEY_ID || 'unknown',
  accessKeySecret: process.env.OSS_ACCESS_KEY_SECRET || 'unknown',
  bucket: process.env.OSS_BUCKET || 'unknown',
  endpoint: process.env.OSS_ENDPOINT || 'unknown',
  region: process.env.OSS_REGION || 'unknown',
};

/**
 * CDN 访问域名
 */
const CDN_HOST = process.env.OSS_CDN_HOST || 'unknown';

/**
 * 创建 OSS 客户端实例
 */
function createOssClient(): OSS {
  return new OSS({
    accessKeyId: ossConfig.accessKeyId,
    accessKeySecret: ossConfig.accessKeySecret,
    bucket: ossConfig.bucket,
    endpoint: ossConfig.endpoint,
    region: ossConfig.region,
  });
}

/**
 * 从文件名获取文件扩展名
 */
function getExtension(filename: string): string {
  const ext = filename.split('.').pop()?.toLowerCase();
  return ext || 'jpg';
}

/**
 * 计算 Base64 数据的 MD5 哈希
 */
function calculateMD5(base64Data: string): string {
  const buffer = Buffer.from(base64Data, 'base64');
  return crypto.createHash('md5').update(buffer).digest('hex');
}

/**
 * 构建 OSS 对象 Key
 * 格式: assets/image/{md5}.{ext}
 */
function buildObjectKey(md5: string, ext: string): string {
  return `assets/image/${md5}.${ext}`;
}

/**
 * 构建 CDN 访问 URL
 */
function buildCdnUrl(objectKey: string): string {
  return `${CDN_HOST}/${objectKey}`;
}

/**
 * 上传单张图片到 OSS
 * @param picture 图片输入对象（包含 name 和 base64 数据）
 * @returns 图片响应对象（包含 URL）
 */
export async function uploadPictureToOSS(
  picture: StoryPictureInput
): Promise<StoryPictureResponse> {
  const client = createOssClient();

  // 解析 Base64 数据
  const base64Data = picture.file;
  const buffer = Buffer.from(base64Data, 'base64');

  // 计算 MD5 用于文件名
  const md5 = calculateMD5(base64Data);
  const ext = getExtension(picture.name);
  const objectKey = buildObjectKey(md5, ext);

  // 上传到 OSS
  await client.put(objectKey, buffer, {
    headers: {
      'Content-Type': `image/${ext === 'jpg' ? 'jpeg' : ext}`,
    },
  });

  // 返回 CDN URL
  return {
    url: buildCdnUrl(objectKey),
  };
}

/**
 * 批量上传图片到 OSS
 * @param pictures 图片输入数组
 * @returns 图片响应数组
 */
export async function uploadPicturesToOSS(
  pictures: StoryPictureInput[]
): Promise<StoryPictureResponse[]> {
  // 并行上传所有图片
  const results = await Promise.all(
    pictures.map((picture) => uploadPictureToOSS(picture))
  );
  return results;
}

/**
 * 解析存储在数据库中的图片 JSON 字符串
 * @param picturesJson 数据库中的 pictures 字段值
 * @returns 图片响应数组
 */
export function parsePicturesFromDB(
  picturesJson: string | null
): StoryPictureResponse[] {
  if (!picturesJson) {
    return [];
  }

  try {
    const urls = JSON.parse(picturesJson);
    if (Array.isArray(urls)) {
      return urls.map((url: string) => ({ url }));
    }
    return [];
  } catch {
    return [];
  }
}

/**
 * 将图片 URL 数组序列化为 JSON 字符串用于存储
 * @param pictures 图片响应数组
 * @returns JSON 字符串
 */
export function serializePicturesForDB(
  pictures: StoryPictureResponse[]
): string {
  return JSON.stringify(pictures.map((p) => p.url));
}
```
