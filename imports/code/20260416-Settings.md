# Settings

```tsx
import { useRef, useEffect } from "react";
import { useNavigate } from "react-router-dom";
import Layout from "@/components/Layout";
import NavigationBar from "@/components/NavigationBar";
import { useTask } from "@/hooks/useTask";
import { TaskId } from "@/types/task";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";

interface SettingItem {
  label: string;
  statusText?: string;
  onClick: () => void;
}

export default function Settings() {
  const navigate = useNavigate();
  const { isTaskFinished, handleTaskAction } = useTask();
  const tapCountRef = useRef(0);
  const tapTimerRef = useRef<ReturnType<typeof setTimeout>>(undefined);

  // 埋点：设置/个人资料页展示
  useEffect(() => {
    bpTrack(EventName.pwa_profile_page_show);
  }, []);

  const isInsLinked = isTaskFinished(TaskId.BindInsAccount);

  const items: SettingItem[] = [
    { label: "PhoneNumber", onClick: () => navigate("/phoneNumber") },
    { label: "Profession", onClick: () => navigate("/profession") },
    { label: "Bio", onClick: () => navigate("/bio") },
    { label: "Tags", onClick: () => navigate("/tags") },
    {
      label: "Link Instagram",
      statusText: isInsLinked ? "Linked" : "Not Connected",
      onClick: () => handleTaskAction(TaskId.BindInsAccount),
    },
    ...(import.meta.env.MODE !== "production" ? [{ label: "Debug", onClick: () => navigate("/debug") }] : []),
  ];

  /** 连续点击版本号 5 次进入 Debug 页 */
  const handleVersionTap = () => {
    tapCountRef.current += 1;
    clearTimeout(tapTimerRef.current);
    if (tapCountRef.current >= 5) {
      tapCountRef.current = 0;
      navigate("/debug");
      return;
    }
    tapTimerRef.current = setTimeout(() => {
      tapCountRef.current = 0;
    }, 2000);
  };

  return (
    <Layout showTabBar={false}>
      <div className="flex flex-col h-full bg-surface">
        <NavigationBar title="Settings" onTitleClick={handleVersionTap} />

        <div className="flex flex-col mt-6 gap-0.5">
          <span className="ml-6 mb-1.5 text-black/50 text-[13px] font-normal">Account Information</span>

          {items.map((item) => (
            <button
              key={item.label}
              onClick={() => {
                bpTrack(EventName.pwa_profile_edit_click, { field: item.label });
                item.onClick();
              }}
              className="flex items-center mx-3 py-3 pl-3 rounded-xl bg-black/[0.03] text-black/95 text-left"
            >
              <span className="flex-1 min-w-0 block font-semibold text-[15px] capitalize">{item.label}</span>

              <div className="flex items-center gap-1.5 shrink-0">
                {item.statusText && <span className="text-[12px] text-black/60">{item.statusText}</span>}
                {/* chevron-right */}
                <svg className="w-4 h-6 mr-3 text-black/70" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M9 5l7 7-7 7" />
                </svg>
              </div>
            </button>
          ))}
        </div>
      </div>
    </Layout>
  );
}
```
