# device

```ts
/**
 * 设备工具类，用于获取设备信息和设备 ID
 */

import { getPhoneDeviceId, isApp } from "./bridge";
import { STORAGE_KEYS } from "@/constants/storageKeys";

/**
 * 获取或创建设备 ID
 * 1. 如果运行在原生 App 中，尝试通过 Bridge 获取
 * 2. 尝试从 localStorage 读取
 * 3. 如果都没有，则生成新的设备 ID
 */
export async function getDeviceId(): Promise<string> {
  if (isApp()) {
    const bridgeDeviceId = await getPhoneDeviceId();
    if (bridgeDeviceId) {
      localStorage.setItem(STORAGE_KEYS.MOCK_DEVICE_ID, bridgeDeviceId);
      return bridgeDeviceId;
    }
  }
  let deviceId = localStorage.getItem(STORAGE_KEYS.MOCK_DEVICE_ID);
  if (deviceId) {
    return deviceId;
  }
  // 生成新的设备 ID
  deviceId = `web_${Date.now()}_${Math.random().toString(36).substring(2)}`;
  localStorage.setItem(STORAGE_KEYS.MOCK_DEVICE_ID, deviceId);
  return deviceId;
}
```
