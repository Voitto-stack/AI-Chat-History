# lovia

```ts
import { z } from 'zod';

// ============================================
// 图片相关 Schema
// ============================================

/**
 * 图片对象（创建/更新时使用）
 * 前端上传时传递 Base64 编码的图片数据
 */
export const storyPictureInputSchema = z.object({
  name: z.string().describe('图片文件名，包含扩展名'),
  file: z.string().describe('图片 Base64 编码数据（不含 data:image/xxx;base64, 前缀）'),
});
export type StoryPictureInput = z.infer<typeof storyPictureInputSchema>;

/**
 * 已保存的图片（响应时使用）
 * 后端返回时只返回 URL
 */
export const storyPictureResponseSchema = z.object({
  url: z.string().url().describe('图片访问 URL'),
  description: z.string().optional().describe('图片描述'),
});
export type StoryPictureResponse = z.infer<typeof storyPictureResponseSchema>;

// ============================================
// 故事请求 Schema
// ============================================

/**
 * 创建故事请求
 */
export const createStoryRequestSchema = z.object({
  from_user_id: z.number().int().positive().describe('发送者用户ID'),
  to_user_id: z.number().int().positive().describe('接收者（伴侣）用户ID'),
  title: z.string().max(100, '标题长度超过限制').optional().describe('故事标题（可选），最大100字符'),
  content: z.string().min(1, '请输入故事内容').max(1000, '内容长度超过限制').describe('故事内容，最多1000字符'),
  pictures: z.array(storyPictureInputSchema).min(1, '请上传故事照片').max(9, '图片数量超过限制').describe('故事图片，1-9张'),
});
export type CreateStoryRequest = z.infer<typeof createStoryRequestSchema>;

/**
 * 更新故事请求
 * 由于使用复合主键，需要提供完整的主键信息定位记录
 */
export const updateStoryRequestSchema = z.object({
  // 复合主键的三个字段用于定位记录
  id: z.number().int().positive().describe('故事ID'),
  from_user_id: z.number().int().positive().describe('发送者用户ID（主键一部分）'),
  to_user_id: z.number().int().positive().describe('接收者用户ID（主键一部分）'),
  // 可更新的字段
  title: z.string().max(100, '标题长度超过限制').optional().describe('故事标题'),
  content: z.string().min(1, '请输入故事内容').max(1000, '内容长度超过限制').optional().describe('故事内容'),
  pictures: z.array(storyPictureInputSchema).min(1, '请上传故事照片').max(9, '图片数量超过限制').optional().describe('故事图片'),
});
export type UpdateStoryRequest = z.infer<typeof updateStoryRequestSchema>;

/**
 * 删除故事项（单条记录的主键信息）
 */
export const deleteStoryItemSchema = z.object({
  id: z.number().int().positive(),
  from_user_id: z.number().int().positive(),
  to_user_id: z.number().int().positive(),
});
export type DeleteStoryItem = z.infer<typeof deleteStoryItemSchema>;

/**
 * 批量删除故事请求
 */
export const deleteStoryRequestSchema = z.object({
  stories: z.array(deleteStoryItemSchema).min(1).describe('要删除的故事列表（需要完整主键）'),
});
export type DeleteStoryRequest = z.infer<typeof deleteStoryRequestSchema>;

/**
 * 获取故事列表请求（分页）
 */
export const getStoryListRequestSchema = z.object({
  limit: z.coerce.number().int().min(1).max(100).default(10).describe('每页数量'),
  offset: z.coerce.number().int().min(0).default(0).describe('偏移量'),
});
export type GetStoryListRequest = z.infer<typeof getStoryListRequestSchema>;

// ============================================
// 故事响应 Schema
// ============================================

/**
 * 单条故事响应
 */
export const storyResponseSchema = z.object({
  id: z.number().int().describe('故事ID'),
  fromUserId: z.number().int().describe('发送者用户ID'),
  toUserId: z.number().int().describe('接收者用户ID'),
  title: z.string().nullable().describe('故事标题（预留字段）'),
  content: z.string().nullable().describe('故事内容'),
  pictures: z.array(storyPictureResponseSchema).describe('故事图片列表'),
  createdAt: z.number().describe('创建时间戳（毫秒）'),
});
export type StoryResponse = z.infer<typeof storyResponseSchema>;

/**
 * 故事列表响应
 */
export const storyListResponseSchema = z.object({
  storyList: z.array(storyResponseSchema).describe('故事列表'),
  nextOffset: z.string().describe('下一页偏移量，"-1" 表示没有更多数据'),
});
export type StoryListResponse = z.infer<typeof storyListResponseSchema>;

```
