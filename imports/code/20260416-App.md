# app

```ts
import Fastify from "fastify";
import cors from "@fastify/cors";
import { metricsPlugin } from "@heyhru/server-plugin-metrics";
import { redisPlugin } from "@heyhru/server-plugin-redis";
import { rateLimitPlugin } from "@heyhru/server-plugin-rate-limit";
import { config } from "./config.js";
import { refreshPoolMetrics } from "./metrics.js";
import "./auth/auth.middleware.js"; // module augmentation for req.user
import { logger } from "./logger.js";
import { authHook } from "./auth/auth.permissions.js";
import { authController } from "./auth/auth.controller.js";
import { userController } from "./users/users.controller.js";
import { datasourceController } from "./datasources/datasources.controller.js";
import { sqlController } from "./sql/sql.controller.js";
import { approvalController } from "./approvals/approvals.controller.js";
import { auditController } from "./audit/audit.controller.js";
import { savedSqlController } from "./saved-sqls/saved-sqls.controller.js";

export async function buildApp() {
  const app = Fastify({ loggerInstance: logger });

  await app.register(cors, { origin: true, credentials: true });
  await app.register(metricsPlugin);
  await app.register(redisPlugin, { url: config.redisUrl });
  await app.register(rateLimitPlugin, { redis: app.redis });
  app.addHook("onResponse", () => { refreshPoolMetrics(); });

  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  app.decorateRequest("user", null as any);
  app.addHook("preHandler", authHook);

  app.addHook("onError", (req, _reply, error, done) => {
    req.log.error(error, "Unhandled request error");
    done();
  });

  app.get("/health", { logLevel: "silent" }, (_req, reply) => reply.send({ status: "ok" }));

  authController(app);
  userController(app);
  datasourceController(app);
  sqlController(app);
  approvalController(app);
  auditController(app);
  savedSqlController(app);

  return app;
}
```
