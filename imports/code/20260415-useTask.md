# useTask

```ts
/**
 * 任务管理 Hook
 * 负责数据和操作
 */

import { useCallback, useMemo } from "react";
import { useNavigate } from "react-router-dom";
import { toast } from "@/utils/toast";
import { useTaskStore } from "@/stores/taskStore";
import { useUserStore } from "@/stores/userStore";
import { useCash } from "@/hooks/useCash";
import { showRewardModalAsync } from "@/components/showRewardModal";
import { showInsModal } from "@/components/showInsModal";
import { showApkDownloadModal } from "@/components/showApkDownloadModal";
import { showVoiceTaskModal } from "@/components/showVoiceTaskModal";
import { listAllCommonTasks, getCommonFinishedTasks, finishUserCommonTask } from "@/http/api";
import { firstExchangeInsContact } from "@/http/insApi";
import { TaskId, ITask, TaskStatus, CUSTOM_TASK_CONFIGS } from "@/types/task";
import { CommonTaskKind, UserServiceCommonCode, CommonTaskInfo } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";
import { isApp } from "@/utils/bridge";
import {
  checkCameraPermission,
  checkMicrophonePermission,
  checkLocationPermission,
  checkNotificationPermission,
} from "@/utils/permissions";
import { isCustomTask, getCustomFinishedTasks, finishCustomTask } from "@/utils/customTasks";
import { STORAGE_KEYS } from "@/constants/storageKeys";
import { useLoadingStore } from "@/stores/loadingStore";
import { useCashoutStore } from "@/stores/cashoutStore";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";

// 需要显示奖励弹窗的任务白名单
const REWARD_MODAL_TASKS = new Set([
  // CompleteProfile 不在此白名单：Profession.tsx 有独立的两阶段奖励流程，弹窗由其自行管理
  TaskId.NotificationPermission,
  TaskId.BindInsAccount,
  // PostPhoto / FirstPost 不在此白名单：CreatePost.tsx 同时完成两个任务，弹窗由其自行管理
  TaskId.FaceVerify,
  TaskId.RecordFirstVoiceMessage,
  TaskId.RecordSecondVoiceMessage,
  TaskId.RecordThirdVoiceMessage,
  TaskId.RecordForthVoiceMessage,
]);

// 提现相关任务列表（用于触发提现任务完成埋点）
const WITHDRAW_RELATED_TASKS = new Set([
  TaskId.SecondEarn,
  TaskId.ThirdEarn,
  TaskId.FourthEarn,
  TaskId.FourthVideoDuration,
  TaskId.FifthEarn,
  TaskId.FifthVideoDuration,
  TaskId.SixthEarn,
  TaskId.SixthVideoDuration,
  TaskId.SeventhEarn,
  TaskId.SeventhVideoDuration,
  TaskId.FaceVerify,
  TaskId.InstallApk,
]);

export const useTask = () => {
  const { tasks, setTasks, updateTaskStatus, getTask } = useTaskStore();
  const { cash } = useUserStore();
  const { refresh: refreshCash } = useCash();
  const navigate = useNavigate();

  // 自定义任务列表
  const customTasks = useMemo(
    () =>
      CUSTOM_TASK_CONFIGS.map((config) => ({
        id: config.id,
        reward: config.reward,
        status: TaskStatus.notStart,
        type: CommonTaskKind.UNRECOGNIZED,
        estimatedTime: 0,
        reminMins: 0,
      })),
    [],
  );

  // 获取服务端任务列表
  const getServerTasks = useCallback(async (): Promise<ITask[]> => {
    try {
      const response = await listAllCommonTasks();
      if (response?.code === UserServiceCommonCode.Success && response.tasks) {
        return response.tasks
          .map((serverTask: CommonTaskInfo): ITask | null => {
            if (!serverTask.taskId) return null;

            // 默认值
            let estimatedTime = 30;
            let reminMins = 0.5;

            // 根据任务类型设置时间（与 react-haven 保持一致）
            switch (serverTask.taskId as TaskId) {
              case TaskId.LocationPermission:
                reminMins = 0.2;
                estimatedTime = 12;
                break;
              case TaskId.NotificationPermission:
                reminMins = 0.3;
                estimatedTime = 18;
                break;
              case TaskId.CompleteProfile:
                reminMins = 2;
                estimatedTime = 120;
                break;
              case TaskId.InstallApk:
              case TaskId.PostPhoto:
              case TaskId.FaceVerify:
                reminMins = 1;
                estimatedTime = 60;
                break;
            }

            return {
              id: serverTask.taskId as TaskId,
              reward: serverTask.reward || "",
              status: TaskStatus.notStart,
              type: serverTask.kind || CommonTaskKind.onboarding,
              estimatedTime: serverTask.estimatedDuration || estimatedTime,
              reminMins,
            };
          })
          .filter((task): task is ITask => task !== null);
      }
      return [];
    } catch (error) {
      return [];
    }
  }, []);

  // 获取所有已完成的任务ID列表（服务端 + 自定义）
  const getFinishedTasks = useCallback(async (): Promise<TaskId[]> => {
    const finishedTaskIds: TaskId[] = [];

    try {
      // 1. 获取服务端已完成任务
      const serverResponse = await getCommonFinishedTasks();
      if (serverResponse?.code === UserServiceCommonCode.Success && serverResponse.tasks) {
        for (const task of serverResponse.tasks) {
          if (task.taskId) {
            finishedTaskIds.push(task.taskId as TaskId);
          }
        }
      }

      // 2. 获取自定义已完成任务
      const customTaskIds = await getCustomFinishedTasks();
      finishedTaskIds.push(...customTaskIds);
    } catch (error) {
      console.error("getFinishedTasks error:", error);
    }

    return finishedTaskIds;
  }, []);

  // 更新任务完成状态
  const updateTasksStatus = useCallback(
    async (taskMap: Map<TaskId, ITask>): Promise<void> => {
      try {
        const newTaskMap = new Map(taskMap);

        // 获取所有已完成的任务ID列表
        const finishedTaskIds = await getFinishedTasks();

        // 统一更新所有已完成任务的状态
        for (const taskId of finishedTaskIds) {
          const task = newTaskMap.get(taskId);
          if (task) {
            newTaskMap.set(taskId, { ...task, status: TaskStatus.finish });
          }
        }

        setTasks(newTaskMap);
      } catch (error) {
        console.error("updateTasksStatus error:", error);
      }
    },
    [setTasks, getFinishedTasks],
  );

  // 完成任务
  const finishTask = useCallback(
    async (taskId: TaskId): Promise<boolean> => {
      try {
        const task = getTask(taskId);
        if (!task) return false;

        // 如果任务已完成，直接返回 true
        if (task.status === TaskStatus.finish) return true;

        const originBalance = cash || 0;
        const earnedAmount = parseFloat(task.reward) || 0;

        // 自定义任务：直接更新本地状态 + 同步云端
        if (isCustomTask(taskId)) {
          updateTaskStatus(taskId, TaskStatus.finish);
          await finishCustomTask(taskId);
          console.log(`[CustomTask] Finished task ${taskId}`);
        } else {
          // 服务端任务：调用服务端 API
          const response = await finishUserCommonTask(taskId);
          if (response?.code !== UserServiceCommonCode.Success) {
            toast.error("Task completion failed, please try again");
            return false;
          }

          // 更新任务状态
          updateTaskStatus(taskId, TaskStatus.finish);
        }

        // 刷新余额（从后端获取最新余额）
        await refreshCash();

        // 埋点：任务完成
        bpTrack(EventName.pwa_task_complete, {
          task_id: taskId,
          task_reward: earnedAmount,
        });

        // 埋点：提现任务完成（仅针对提现相关任务）
        if (WITHDRAW_RELATED_TASKS.has(taskId)) {
          bpTrack(EventName.pwa_withdraw_task_complete, {
            task_id: taskId,
            task_reward: earnedAmount,
            origin_balance: originBalance,
            new_balance: cash ? cash + earnedAmount : earnedAmount,
          });
        }

        // 埋点：Cashout 任务结果（UV: 3,591）
        bpTrack(EventName.pwa_home_cash_task_result, {
          task_id: taskId,
          result: "success",
          reward_amount: earnedAmount,
        });
        bpTrack(EventName.pwa_home_cash_guided_task_result, {
          task_name: String(taskId),
          earning_value: earnedAmount,
          task_stage: useCashoutStore.getState().currentTaskStage,
          total_earning_value: (cash || 0) + earnedAmount,
        });

        // 统一处理奖励弹窗（如果任务在白名单中且奖励金额大于0）
        // 显示弹窗前清除 loading，避免 loading 遮挡弹窗导致无法交互
        if (REWARD_MODAL_TASKS.has(taskId) && earnedAmount > 0) {
          const { count, hideLoading } = useLoadingStore.getState();
          for (let i = 0; i < count; i++) hideLoading();

          // 埋点：领取奖励
          bpTrack(EventName.pwa_task_claim_reward, {
            task_id: taskId,
            task_reward: earnedAmount,
          });

          if (taskId === TaskId.BindInsAccount) {
            // INS 绑定任务：弹窗关闭后给一单 INS 交换请求
            bpTrack(EventName.pwa_earning_ins_task_complete);
            await showRewardModalAsync(originBalance, earnedAmount, { source: "task" });
            await firstExchangeInsContact();
          } else {
            await showRewardModalAsync(originBalance, earnedAmount, { source: "task" });
          }
        }

        // SecondEarn 完成后自动弹 APK 下载弹窗（只弹一次）
        if (taskId === TaskId.SecondEarn && !isApp()) {
          const shown = localStorage.getItem(STORAGE_KEYS.APK_PROMPT_SECOND_EARN_SHOWN);
          if (!shown) {
            localStorage.setItem(STORAGE_KEYS.APK_PROMPT_SECOND_EARN_SHOWN, "true");
            showApkDownloadModal("task_page");
          }
        }

        return true;
      } catch (error) {
        toast.error("Task completion failed, please try again");
        return false;
      }
    },
    [getTask, updateTaskStatus, cash, refreshCash],
  );

  // 刷新任务列表
  const refreshTasks = useCallback(async (): Promise<void> => {
    const serverTasks = await getServerTasks();
    const newMap = new Map<TaskId, ITask>();
    [...serverTasks, ...customTasks].forEach((task) => {
      newMap.set(task.id, task);
    });
    await updateTasksStatus(newMap);

    // 自动完成 APK 安装任务（仅在 APK 环境中）
    if (isApp()) {
      bpTrack(EventName.pwa_apk_install_task_status, { apk_install_task_status: "installed_in_apk" });
      await finishTask(TaskId.InstallApk);
    }
  }, [getServerTasks, customTasks, updateTasksStatus, finishTask]);

  // 处理任务操作
  const handleTaskAction = useCallback(
    async (taskId: TaskId): Promise<boolean | void> => {
      const task = getTask(taskId);
      if (!task) return false;

      // 根据任务类型执行不同的操作
      switch (taskId) {
        // 注册任务 - 直接完成
        case TaskId.Register:
          await finishTask(taskId);
          break;

        // 相机权限任务
        case TaskId.CameraPermission: {
          const result = await checkCameraPermission();
          if (result === "granted") {
            await finishTask(taskId);
            return true;
          }
          return false;
        }

        // 麦克风权限任务
        case TaskId.MicrophonePermission: {
          const result = await checkMicrophonePermission();
          if (result === "granted") {
            await finishTask(taskId);
          }
          break;
        }

        // 位置权限任务
        case TaskId.LocationPermission: {
          const result = await checkLocationPermission();
          // 只有在成功获取到经纬度时才传递坐标，否则让用户手动输入
          const state =
            result.status === "granted" && result.latitude && result.longitude
              ? { lat: result.latitude, lng: result.longitude }
              : undefined;
          navigate("/editZipcode", { state });
          break;
        }

        // 通知权限任务 (H5 和 APK 通用)
        case TaskId.NotificationPermission:
        case TaskId.NotificationPermissionApk: {
          const result = await checkNotificationPermission();
          if (result === "granted") {
            await finishTask(taskId);
          }
          break;
        }

        // APK 安装任务
        case TaskId.InstallApk:
          // 埋点：APK 安装任务点击
          bpTrack(EventName.pwa_earnings_apk_installation_mission_clicked);

          if (isApp()) {
            // 如果已经在 APK 中，直接完成任务
            bpTrack(EventName.pwa_apk_install_task_status, { apk_install_task_status: "auto_complete" });
            await finishTask(TaskId.InstallApk);
          } else {
            // 打开 APK 下载弹窗
            showApkDownloadModal("task_action");
          }
          break;

        case TaskId.BindInsAccount:
          showInsModal();
          break;

        // 语音消息任务
        case TaskId.RecordFirstVoiceMessage: {
          showVoiceTaskModal("first");
          break;
        }
        case TaskId.RecordSecondVoiceMessage: {
          showVoiceTaskModal("second");
          break;
        }
        case TaskId.RecordThirdVoiceMessage: {
          showVoiceTaskModal("third");
          break;
        }
        case TaskId.RecordForthVoiceMessage: {
          showVoiceTaskModal("forth");
          break;
        }
        // 人���验证任务 - 需要先获取相机权限
        case TaskId.FaceVerify: {
          console.log("[useTask] FaceVerify task clicked, checking camera permission...");
          const granted = await handleTaskAction(TaskId.CameraPermission);
          console.log("[useTask] Camera permission result:", granted);
          if (granted) {
            // 跳转到人脸验证页面
            console.log("[useTask] Navigating to /verifyPhoto");
            navigate("/verifyPhoto");
          } else {
            console.log("[useTask] Camera permission denied, showing toast");
            toast.error("Camera permission is required for face verification");
          }
          break;
        }
        // 发布第一条动态
        case TaskId.FirstPost:
        case TaskId.PostPhoto: {
          // 跳转到创建动态页面
          navigate("/createPost");
          break;
        }

        // 完善资料任务
        case TaskId.CompleteProfile: {
          // 埋点：完善资料任务点击
          bpTrack(EventName.pwa_earnings_finish_profile_click);
          navigate("/profileTask");
          break;
        }

        // 消息任务 - 跳转到对应页面
        case TaskId.ReplyFirstMessage:
        case TaskId.ReplySomeMessage3:
        case TaskId.ReplySomeMessage4: {
          navigate("/chats");
          break;
        }

        // 游戏任务 - 跳转到游戏页面
        case TaskId.PlayFirstGame:
        case TaskId.WinSomeGame2:
        case TaskId.WinSomeGame3:
        case TaskId.WinSomeGame4:
          navigate("/game");
          break;

        // 自定义收益任务 - 跳转到直播页面
        case TaskId.SecondEarn:
        case TaskId.ThirdEarn:
        case TaskId.FourthEarn:
        case TaskId.FourthVideoDuration:
        case TaskId.FifthEarn:
        case TaskId.FifthVideoDuration:
        case TaskId.SixthEarn:
        case TaskId.SixthVideoDuration:
        case TaskId.SeventhEarn:
        case TaskId.SeventhVideoDuration: {
          // 跳转到直播页面
          navigate("/live");
          break;
        }
        default:
          console.log("Unknown task action:", taskId);
      }
    },
    [getTask, finishTask, navigate],
  );

  return {
    tasks,
    refreshTasks,
    finishTask,
    handleTaskAction,
    isTaskFinished: useCallback((taskId: TaskId) => getTask(taskId)?.status === TaskStatus.finish, [getTask]),
  };
};
```
