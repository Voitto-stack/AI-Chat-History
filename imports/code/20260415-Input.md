# Input

```tsx
import { ReactNode, InputHTMLAttributes, forwardRef } from "react";
import FormErrorMessage from "@/components/FormErrorMessage";

interface InputProps {
  title: string;
  titleClassName?: string;
  subtitle?: ReactNode;
  error?: string | null;
  hint?: ReactNode;
  children?: ReactNode;
  inputProps?: Omit<InputHTMLAttributes<HTMLInputElement>, "className">;
}

const Input = forwardRef<HTMLInputElement, InputProps>(
  ({ title, titleClassName, subtitle, error, hint, children, inputProps }, ref) => {
    return (
      <div className="flex-1 flex flex-col items-center justify-start">
        <div
          className={`mt-3 text-center ${titleClassName ?? "text-black font-normal text-base leading-[21px] tracking-[-0.23px]"}`}
        >
          <p>{title}</p>
        </div>

        {subtitle && <div className="mt-1">{subtitle}</div>}

        {children || (
          <input
            ref={ref}
            className="flex items-center w-[80%] h-14 mt-10 p-0 border-0 border-b-[0.5px] border-solid border-b-[#e5e5ea] rounded-none bg-transparent text-black font-medium text-[22px] leading-7 text-center focus:outline-none placeholder:text-black/30 placeholder:font-medium placeholder:text-[22px] placeholder:leading-7 placeholder:text-center autofill:bg-transparent autofill:shadow-[inset_0_0_0_1000px_transparent]"
            {...inputProps}
          />
        )}

        {error && (
          <FormErrorMessage
            message={error}
            className="justify-center mt-4 mx-[37px]"
            textClassName="leading-normal"
            icon={
              <svg width="12" height="12" viewBox="0 0 12 12" fill="none" xmlns="http://www.w3.org/2000/svg">
                <circle cx="6" cy="6" r="5.5" stroke="#FF0101" />
                <path d="M6 3V7M6 9V9.5" stroke="#FF0101" strokeLinecap="round" />
              </svg>
            }
          />
        )}

        {!error && hint && <div className="mt-4">{hint}</div>}
      </div>
    );
  },
);

Input.displayName = "Input";

export default Input;

```
