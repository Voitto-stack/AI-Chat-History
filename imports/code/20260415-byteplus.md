# byteplus

```ts
// BytePlus (火山引擎) 追踪
// https://www.volcengine.com/docs/6285/65992

import { BytePlusManager } from "../managers/BytePlusManager";
import type { UserProfile } from "../types";

/**
 * BytePlus 事件上报
 * @param eventName 事件名称
 * @param params 事件参数
 */
export function bpTrack(eventName: string, params: Record<string, unknown> = {}): void {
  BytePlusManager.getInstance().trackEvent(eventName, params);
}

/**
 * 设置用户 ID
 * @param userId 用户唯一标识
 */
export function bpSetUserId(userId: string): void {
  BytePlusManager.getInstance().setUserId(userId);
}

/**
 * 更新用户画像
 * @param profile 用户画像信息
 */
export function bpUpdateProfile(profile: Partial<UserProfile>): void {
  BytePlusManager.getInstance().updateProfile(profile);
}

/**
 * 获取用户画像
 */
export function bpGetProfile(): UserProfile {
  return BytePlusManager.getInstance().getProfile();
}

/**
 * 检查 BytePlus 是否已初始化
 */
export function isBytePlusInitialized(): boolean {
  return BytePlusManager.getInstance().isInitialized();
}

```
