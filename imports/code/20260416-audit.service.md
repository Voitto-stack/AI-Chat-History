# audit.service

```ts
import { type FastifyRequest, type FastifyReply } from "fastify";
import { insertAuditLog, queryAuditLogs } from "./audit.model.js";

export function writeAuditLog(entry: {
  userId: string;
  dataSourceId?: string;
  action: string;
  sqlText?: string;
  resultSummary?: string;
  ipAddress?: string;
}) {
  return insertAuditLog(entry);
}

export async function logList(req: FastifyRequest, reply: FastifyReply) {
  const { userId, dataSourceId, limit, offset } = (req.body ?? {}) as Record<string, string>;
  return reply.send(
    await queryAuditLogs({
      userId,
      dataSourceId,
      limit: limit ? Number(limit) : undefined,
      offset: offset ? Number(offset) : undefined,
    })
  );
}
```
