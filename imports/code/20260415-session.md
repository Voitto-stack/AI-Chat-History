# session

```ts
/**
 * Session 管理工具
 */

const SESSION_TOKEN_KEY = 'sessionToken';
const OLD_API_TOKEN_KEY = 'oldApiToken';
const FEISHU_AUTH_KEY = 'feishuAuth';

/**
 * 获取 session token
 */
export const getSessionToken = (): string | null => {
  // 优先从 localStorage 获取
  const token = localStorage.getItem(SESSION_TOKEN_KEY);
  if (token) {
    return token;
  }

  // 备选方案：从 cookie 获取
  const cookieToken = document.cookie.replace(
    /(?:(?:^|.*;\s*)sessionToken\s*=\s*([^;]*).*$)|^.*$/,
    '$1',
  );

  return cookieToken || null;
};

/**
 * 获取旧版 API token
 */
export const getOldApiToken = (): string | null => {
  return localStorage.getItem(OLD_API_TOKEN_KEY);
};

/**
 * 设置旧版 API token
 */
export const setOldApiToken = (token: string): void => {
  localStorage.setItem(OLD_API_TOKEN_KEY, token);
};

/**
 * 设置 session token
 */
export const setSessionToken = (token: string): void => {
  localStorage.setItem(SESSION_TOKEN_KEY, token);
};

/**
 * 获取飞书 auth
 */
export const getFeishuAuth = (): string | null => {
  return localStorage.getItem(FEISHU_AUTH_KEY);
};

/**
 * 设置飞书 auth
 */
export const setFeishuAuth = (auth: string): void => {
  localStorage.setItem(FEISHU_AUTH_KEY, auth);
};

/**
 * 移除 session cookie 和 localStorage
 */
export const removeSessionCookie = (): void => {
  // 清除 localStorage
  localStorage.removeItem(SESSION_TOKEN_KEY);
  localStorage.removeItem(OLD_API_TOKEN_KEY);
  localStorage.removeItem(FEISHU_AUTH_KEY);

  // 清除 cookie
  document.cookie =
    'sessionToken=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;';
};

```
