import { type FastifyRequest, type FastifyReply } from "fastify";
import NodeSqlParser from "node-sql-parser";
import {
  getDataSourceWithPassword,
  getDataSourceName,
  getPool,
  getPoolForDatabase,
} from "@heyhru/business-dms-datasource";

const RESTRICTED_DS_ROLES = new Set(["developer", "viewer"]);
const DML_DIRECT_ROLES = new Set(["admin", "maintainer", "developer"]);
const TEST_DS_PATTERN = "-test";
import { config } from "../config.js";
import { writeAuditLog } from "@heyhru/business-dms-audit";
import { sqlDuration } from "../metrics.js";

type SqlCategory = "select" | "dml" | "ddl";

const parser = new NodeSqlParser.Parser();
const DDL_KEYWORDS = new Set(["ALTER", "CREATE", "DROP", "TRUNCATE", "GRANT", "REVOKE"]);
const DDL_AST_TYPES = new Set(["alter", "create", "drop"]);

function sqlParse(sql: string): SqlCategory {
  let ast: ReturnType<typeof parser.astify>;
  try {
    const normalized = sql.replace(/::[a-zA-Z_][a-zA-Z0-9_[\]]*\b/g, "");
    ast = parser.astify(normalized);
  } catch (err) {
    // Fallback for DDL that node-sql-parser cannot parse (e.g. ALTER TYPE, ALTER TABLE ADD COLUMN)
    const first = sql.trimStart().split(/\s+/)[0]?.toUpperCase();
    if (first && DDL_KEYWORDS.has(first)) return "ddl";
    throw new Error(`SQL parse error: ${err instanceof Error ? err.message : String(err)}`, { cause: err });
  }
  const stmts = Array.isArray(ast) ? ast : [ast];
  const types = stmts.map((s) => s.type?.toLowerCase());
  if (types.some((t) => t && DDL_AST_TYPES.has(t))) return "ddl";
  if (types.some((t) => t && t !== "select")) return "dml";
  return "select";
}

export function postSqlParse(req: FastifyRequest, reply: FastifyReply) {
  const { sql } = (req.body ?? {}) as Record<string, string>;
  if (!sql) return reply.code(400).send({ error: "SQL is required" });
  try {
    return reply.send({ category: sqlParse(sql) });
  } catch (err) {
    return reply.code(400).send({ error: err instanceof Error ? err.message : String(err) });
  }
}

export async function postSqlExecute(req: FastifyRequest, reply: FastifyReply) {
  const body = (req.body ?? {}) as Record<string, unknown>;
  const { dataSourceId, database, sql } = body as Record<string, string>;
  if (!dataSourceId || !sql) {
    return reply.code(400).send({ error: "Data source ID and SQL are required" });
  }
  const page = typeof body.page === "number" ? body.page : 1;
  const pageSize = typeof body.pageSize === "number" ? body.pageSize : 50;
  const ip = (req.headers["x-forwarded-for"] as string) ?? req.headers["x-real-ip"] ?? "unknown";
  const role = (req.user as { id: string; role: string }).role;

  const dsName = await getDataSourceName(dataSourceId);
  if (!dsName) return reply.code(404).send({ error: "Data source not found" });
  if (RESTRICTED_DS_ROLES.has(role) && !dsName.includes(TEST_DS_PATTERN)) {
    return reply.code(403).send({ error: "Access denied: your role can only access test data sources" });
  }

  const allowDml = dsName.includes(TEST_DS_PATTERN) && DML_DIRECT_ROLES.has(role);
  try {
    const result = await sqlExecute({ dataSourceId, database, sql, userId: req.user.id, ip, page, pageSize, allowDml });
    return reply.send(result);
  } catch (err) {
    req.log.error(err, "SQL execute failed (user=%s, ds=%s)", req.user.id, dataSourceId);
    return reply.code(400).send({ error: err instanceof Error ? err.message : String(err) });
  }
}

interface SqlExecuteParams {
  dataSourceId: string;
  database: string | undefined;
  sql: string;
  userId: string;
  ip: string;
  page: number;
  pageSize: number;
  allowDml?: boolean;
}

async function fetchCountQuery(pool: { execute(sql: string): Promise<unknown[]> }, sql: string): Promise<number> {
  const countSql = `SELECT COUNT(*) AS total FROM (${sql}) AS _count_t`;
  const rows = (await pool.execute(countSql)) as Record<string, unknown>[];
  return Number(rows[0]?.total ?? 0);
}

async function sqlExecuteDml(pool: { execute(sql: string): Promise<unknown[]> }, sql: string, params: { dataSourceId: string; userId: string; ip: string }) {
  const end = sqlDuration.startTimer({ data_source_id: params.dataSourceId, sql_type: "dml" });
  try {
    await pool.execute(sql);
    end({ status: "success" });
  } catch (err) {
    end({ status: "failed" });
    throw err;
  }
  await writeAuditLog({ userId: params.userId, dataSourceId: params.dataSourceId, action: "DML", sqlText: sql, resultSummary: "executed", ipAddress: params.ip });
  return { rows: [], total: 0, page: 1, pageSize: 0 };
}

async function sqlExecute({ dataSourceId, database, sql, userId, ip, page, pageSize, allowDml = false }: SqlExecuteParams) {
  const category = sqlParse(sql); // throws if SQL cannot be parsed
  if (category === "ddl") {
    throw new Error("DDL statements must go through the approval workflow");
  }
  if (category === "dml" && !allowDml) {
    throw new Error("DML statements must go through the approval workflow");
  }

  const pool = database
    ? await getPoolForDatabase(dataSourceId, database, config.encryptionKey)
    : await (async () => {
        const ds = await getDataSourceWithPassword(dataSourceId, config.encryptionKey);
        return ds ? getPool(ds) : null;
      })();
  if (!pool) throw new Error("Data source not found");

  if (category !== "select") {
    return sqlExecuteDml(pool, sql, { dataSourceId, userId, ip });
  }

  const trimmedSql = sql.replace(/;\s*$/, "");
  const hasLimit = /\bLIMIT\b/i.test(trimmedSql);
  const dataSql = hasLimit ? trimmedSql : `${trimmedSql} LIMIT ${pageSize} OFFSET ${(page - 1) * pageSize}`;

  const end = sqlDuration.startTimer({ data_source_id: dataSourceId, sql_type: category });
  let rows: Record<string, unknown>[];
  let total: number;
  try {
    if (hasLimit) {
      rows = (await pool.execute(dataSql)) as Record<string, unknown>[];
      total = rows.length;
    } else {
      const [rawRows, count] = await Promise.all([pool.execute(dataSql), fetchCountQuery(pool, trimmedSql)]);
      rows = rawRows as Record<string, unknown>[];
      total = count;
    }
    end({ status: "success" });
  } catch (err) {
    end({ status: "failed" });
    throw err;
  }

  await writeAuditLog({
    userId,
    dataSourceId,
    action: "SELECT",
    sqlText: sql,
    resultSummary: `${total} total rows`,
    ipAddress: ip,
  });

  return { rows, total, page, pageSize };
}

export async function postSqlDatabases(req: FastifyRequest, reply: FastifyReply) {
  const { dataSourceId } = (req.body ?? {}) as Record<string, string>;
  if (!dataSourceId) {
    return reply.code(400).send({ error: "数据源 ID 不能为空" });
  }

  const ds = await getDataSourceWithPassword(dataSourceId, config.encryptionKey);
  if (!ds) return reply.code(404).send({ error: "数据源未找到" });

  const pool = getPool({
    ...ds,
    database: ds.database ?? (ds.type === "postgres" ? "postgres" : ""),
  });

  const sql =
    ds.type === "mysql"
      ? "SHOW DATABASES"
      : "SELECT datname FROM pg_database WHERE datistemplate = false AND datname NOT IN ('cloudsqladmin') ORDER BY datname";

  try {
    const rows = await pool.execute(sql);
    const names = (rows as Record<string, unknown>[]).map(
      (r) => Object.values(r)[0] as string
    );
    return reply.send(names);
  } catch (err) {
    req.log.error(err, "Failed to list databases (ds=%s)", dataSourceId);
    return reply.code(400).send({ error: err instanceof Error ? err.message : String(err) });
  }
}

export async function postSqlTables(req: FastifyRequest, reply: FastifyReply) {
  const { dataSourceId, database } = (req.body ?? {}) as Record<string, string>;
  if (!dataSourceId || !database) {
    return reply.code(400).send({ error: "数据源 ID 和数据库名不能为空" });
  }

  const pool = await getPoolForDatabase(dataSourceId, database, config.encryptionKey);
  if (!pool) return reply.code(404).send({ error: "数据源未找到" });

  const ds = await getDataSourceWithPassword(dataSourceId, config.encryptionKey);
  const sql =
    ds!.type === "mysql"
      ? `SELECT TABLE_NAME FROM information_schema.TABLES WHERE TABLE_SCHEMA = '${database}' ORDER BY TABLE_NAME`
      : "SELECT tablename FROM pg_tables WHERE schemaname = 'public' ORDER BY tablename";

  try {
    const rows = await pool.execute(sql);
    const names = (rows as Record<string, unknown>[]).map(
      (r) => Object.values(r)[0] as string
    );
    return reply.send(names);
  } catch (err) {
    req.log.error(err, "Failed to list tables (ds=%s, db=%s)", dataSourceId, database);
    return reply.code(400).send({ error: err instanceof Error ? err.message : String(err) });
  }
}
