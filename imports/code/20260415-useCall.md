# useCall

```tsx
import { useCallback, useRef } from "react";
import WebCallManager from "@/utils/webCallManager";
import NativeCallManager from "@/utils/nativeCallManager";
import { useUser } from "@/hooks/useUser";
import useWebCall from "./useWebCall";
import useMockCall from "./useMockCall";
import { useDispatchDetection } from "./useDispatchDetection";
import { useCallStore, type CallStoreState } from "../stores/callStore";
import { STORAGE_KEYS } from "../constants/storageKeys";
import { isApp } from "../utils/bridge";
import { getCallOrderPrice } from "../http/api";

const TAG = "useCall";

export const useCallInit = () => {
  const isInitializedRef = useRef(false);
  const { userInfo, timUsersig } = useUser();
  const { setReleasePrice } = useCall();
  //首次登陆时store在中还没有 userInfo
  const storageAuth = JSON.parse(localStorage.getItem(STORAGE_KEYS.AUTH_STORAGE) || "{}");

  const userId = userInfo?.userId || storageAuth?.userInfo?.userId;
  const userSig = timUsersig || storageAuth?.timUsersig;
  // 使用 useWebCall 处理通话事件
  useWebCall();
  // 使用 useMockCall 处理 mock 通话事件
  useMockCall();
  // 启动/反屏时 Dispatch 检测
  useDispatchDetection();

  // 初始化获取通话单价并设置到 store
  const initReleasePrice = useCallback(async () => {
    try {
      const response = await getCallOrderPrice();

      const price = Number(response.commonVideoOrderPrice);
      if (response.commonVideoOrderPrice && price > 0) {
        setReleasePrice(price);
      } else {
        console.warn("initReleasePrice: 价格字段无效，使用默认值", response.commonVideoOrderPrice);
      }
    } catch (e) {
      console.error("initReleasePrice: 请求失败，使用默认值", e);
    }
  }, [setReleasePrice]);

  const initCall = useCallback(async () => {
    if (isInitializedRef.current) {
      console.warn(TAG, "CallManager already initialized");
      return;
    }
    try {
      console.log(TAG, "Initializing CallManager...", userSig, userId);

      if (!userId || !userSig) {
        console.error(TAG, "Invalid userId or userSig");
        return;
      }

      // 获取真实通话单价
      initReleasePrice();

      // 初始化 CallManager

      if (isApp()) {
        await NativeCallManager.initialize(userId.toString());
      } else {
        await WebCallManager.initialize(userId.toString(), userSig);
      }

      isInitializedRef.current = true;
      console.log(TAG, "CallManager initialized successfully");
    } catch (error) {
      console.error(TAG, "Failed to initialize CallManager:", error);
    }
  }, [userId, userSig, initReleasePrice]);

  return { initCall };
};

/**
 * 获取 CallStore 状态（在 React 组件外部使用）
 */
export const getCallStoreState = () => useCallStore.getState();

/**
 * 通话状态 Hook
 * 提供通话相关的状态读取和操作方法
 */
export const useCall = (): CallStoreState => {
  const store = useCallStore();
  return store;
};
```
