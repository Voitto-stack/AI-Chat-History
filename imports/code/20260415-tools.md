# tools

```js
/**
 * Build & Deploy Scripts Utilities
 * 构建和部署脚本的通用工具函数
 */

const { execSync } = require("child_process");

/**
 * 日志函数
 * @param {string} type - 日志类型: info, success, error, warn
 * @param {string} message - 日志消息
 */
function log(type, message) {
  const prefix =
    {
      info: "[INFO]",
      success: "[SUCCESS]",
      error: "[ERROR]",
      warn: "[WARN]",
    }[type] || "[LOG]";
  console.log(`${prefix} ${message}`);
}

/**
 * 执行命令
 * @param {string} command - 要执行的命令
 * @param {object} options - execSync 选项
 */
function exec(command, options = {}) {
  try {
    log("info", `执行: ${command}`);
    return execSync(command, {
      encoding: "utf-8",
      stdio: "inherit",
      ...options,
    });
  } catch (error) {
    throw new Error(`命令执行失败: ${command}\n${error.message}`);
  }
}

module.exports = {
  log,
  exec,
};
```
