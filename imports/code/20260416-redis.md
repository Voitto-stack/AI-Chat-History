# redis

```ts
import Redis from "ioredis";

let client: Redis | null = null;

/**
 * Initialize the shared Redis client.
 * Call once at application startup. Subsequent calls are no-ops.
 */
export function initRedis(url: string): void {
  if (client) return;
  client = new Redis(url, { lazyConnect: true, maxRetriesPerRequest: 3 });

  client.on("error", (err) => {
    console.error("[server-plugin-redis] connection error:", err.message);
  });

  // Fire-and-forget — the client auto-reconnects on failure,
  // so callers are not blocked if Redis is temporarily unavailable.
  client.connect().catch(() => {});
}

/**
 * Return the shared Redis client, or `null` when Redis is not initialised
 * or the connection is not ready. Callers should fall back to the database
 * when `null` is returned.
 */
export function getRedis(): Redis | null {
  if (!client) return null;
  if (client.status !== "ready") return null;
  return client;
}

/**
 * Gracefully close the Redis connection. Call during shutdown.
 */
export async function closeRedis(): Promise<void> {
  if (!client) return;
  await client.quit();
  client = null;
}
```
