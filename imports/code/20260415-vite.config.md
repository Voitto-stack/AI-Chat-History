# vite.config

```ts
import { defineConfig } from "vite";
import react from "@vitejs/plugin-react-swc";
import tailwindcss from "@tailwindcss/vite";
import { VitePWA } from "vite-plugin-pwa";
import { compression } from "vite-plugin-compression2";
import { sentryVitePlugin } from "@sentry/vite-plugin";
import path from "path";

export default defineConfig(({ mode }) => {
  const isProduction = mode === "production";
  const sentryUploadEnabled = process.env.SENTRY_UPLOAD === "true";
  const sentryAuthToken = process.env.SENTRY_AUTH_TOKEN;
  const sentryOrg = process.env.SENTRY_ORG || "presence-2h";
  const sentryProject = process.env.SENTRY_PROJECT || "javascript-react";
  const shouldUploadSourceMaps = isProduction && sentryUploadEnabled && Boolean(sentryAuthToken);

  const plugins = [
    react(),
    tailwindcss(),
    VitePWA({
      strategies: "injectManifest",
      srcDir: "src",
      filename: "service-worker.ts",
      injectRegister: "auto",
      devOptions: {
        enabled: true,
      },
      injectManifest: {
        maximumFileSizeToCacheInBytes: 50 * 1024 * 1024, // 50MB
        globIgnores: ["**/*.apk"],
      },
      manifest: {
        short_name: "GraceChat",
        name: "GraceChat",
        icons: [
          {
            src: "favicon.ico",
            sizes: "64x64 32x32 24x24 16x16",
            type: "image/x-icon",
          },
          {
            src: "logo192.png",
            type: "image/png",
            sizes: "192x192",
          },
          {
            src: "logo512.png",
            type: "image/png",
            sizes: "512x512",
          },
        ],
        start_url: ".",
        display: "fullscreen",
        theme_color: "#F5F7FB",
        background_color: "#EFF8FF",
        description: "Make money while you chat. Install required. 👆🏻",
        screenshots: [
          {
            src: "screenshots/screenshot1.png",
            type: "image/png",
            sizes: "375x812",
            form_factor: "narrow",
            platform: "narrow",
          },
          {
            src: "screenshots/screenshot2.png",
            type: "image/png",
            sizes: "375x812",
            form_factor: "narrow",
            platform: "narrow",
          },
        ],
      },
    }),
    // gzip + brotli 压缩（threshold 1KB 以上才压缩）
    compression({ algorithms: ["gzip", "brotliCompress"], threshold: 1024 }),
  ];

  // Enable Sentry sourcemap upload only for production builds with explicit opt-in and a valid token
  if (shouldUploadSourceMaps) {
    plugins.push(
      sentryVitePlugin({
        org: sentryOrg,
        project: sentryProject,
        authToken: sentryAuthToken,
        telemetry: false, // 禁用遥测数据收集
        sourcemaps: {
          assets: "./dist/assets/**",
        },
        release: {
          name: process.env.VITE_APP_VERSION || "1.0.0",
        },
      }),
    );
  }

  return {
    define: {
      "import.meta.env.VITE_APP_VERSION": JSON.stringify(process.env.VITE_APP_VERSION || "1.0.0"),
      "import.meta.env.VITE_BUILD_NUMBER": JSON.stringify(
        process.env.VITE_BUILD_NUMBER || process.env.BUILD_NUMBER || "local",
      ),
    },
    plugins,
    resolve: {
      alias: {
        "@": path.resolve(__dirname, "./src"),
      },
    },
    build: {
      outDir: "dist",
      // 面向现代浏览器，减小产物体积
      target: "es2020",
      // 生产环境移除 console/debugger
      minify: "terser",
      terserOptions: {
        compress: {
          drop_debugger: true,
        },
        format: {
          // 移除注释
          comments: false,
        },
      },
      // 分包策略
      rollupOptions: {
        input: {
          // 主应用（React SPA）
          main: path.resolve(__dirname, "index.html"),

          // 活动落地页（独立 HTML）
          installToClaim: path.resolve(__dirname, "activity/installToClaim.html"),
          getallmylinks: path.resolve(__dirname, "activity/getallmylinks.html"),
        },
        output: {
          manualChunks: {
            // React 核心
            "vendor-react": ["react", "react-dom", "react-router-dom"],
            // 状态管理
            "vendor-zustand": ["zustand"],
            // 腾讯云 SDK（体积大，单独拆）
            "vendor-trtc": ["trtc-sdk-v5"],
            "vendor-tuicall": ["tuicall-engine-webrtc"],
            "vendor-im": ["@tencentcloud/chat", "@tencentcloud/chat-uikit-engine"],
            // 图表
            "vendor-echarts": ["echarts"],
            // MediaPipe（人脸检测）
            "vendor-mediapipe": ["@mediapipe/tasks-vision"],
            // 工具库
            "vendor-utils": ["date-fns", "date-fns-tz", "lottie-react"],
            // 图片处理
            "vendor-image": ["browser-image-compression", "react-easy-crop"],
            // Proto（体积大，单独拆避免阻塞首屏）
            "vendor-proto": ["@heyhru/business-pwa-proto"],
          },
          // 资源文件按类型分目录，利于 CDN 缓存策略
          chunkFileNames: "assets/js/[name]-[hash].js",
          entryFileNames: "assets/js/[name]-[hash].js",
          assetFileNames: "assets/[ext]/[name]-[hash].[ext]",
        },
      },
      // chunk 大小警告阈值
      chunkSizeWarningLimit: 500,
      // 资源文件内联阈值（小于 4KB 内联为 base64）
      assetsInlineLimit: 4096,
      // 启用 CSS 代码分割
      cssCodeSplit: true,
      // 生产环境启用 hidden sourcemap（上传 Sentry 后不暴露给用户）
      sourcemap: shouldUploadSourceMaps ? "hidden" : false,
    },
    server: {
      port: 3000,
      host: true,
    },
  };
});
```
