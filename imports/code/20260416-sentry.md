# sentry

```ts
/**
 * Sentry 错误监控初始化
 * 与 react-haven 保持完全一致
 */
import * as Sentry from "@sentry/react";

const appVersion = import.meta.env.VITE_APP_VERSION || "1.0.0";

// 判断是否为本地开发环境
const isLocalDev =
  import.meta.env.DEV || window.location.hostname === "localhost" || window.location.hostname === "127.0.0.1";

// 初始化 Sentry（使用 enabled 控制是否启用）
Sentry.init({
  dsn: "https://ac53fd8f2998ef29593ba339aaaa4865@o4510434569158656.ingest.us.sentry.io/4510434640986112",

  // 控制是否启用 Sentry（本地开发环境禁用）
  enabled: !isLocalDev,

  // 性能监控 integrations
  integrations: [
    Sentry.browserTracingIntegration({
      // 启用自动追踪路由变化
      enableInp: true,
    }),
    // 启用浏览器性能分析
    Sentry.browserProfilingIntegration(),
  ],

  // 性能监控采样率（1.0 = 100%, 建议生产环境设置为 0.1-0.2）
  tracesSampleRate: 0.1,

  // Profiling 采样率（相对于 tracesSampleRate）
  profilesSampleRate: 0.1,

  // Session Replay 已关闭（采样率设为 0）
  replaysSessionSampleRate: 0,
  replaysOnErrorSampleRate: 0,

  // Environment（与 react-haven 的 REACT_APP_APP_ENV 等价）
  environment: import.meta.env.VITE_APP_ENV || import.meta.env.MODE || "development",

  // Release version
  release: appVersion,

  // 忽略特定错误
  ignoreErrors: [
    // 网络错误
    "Network request failed",
    "NetworkError",
    "Failed to fetch",
    // 其他常见的可忽略错误
    "ResizeObserver loop limit exceeded",
    "Non-Error promise rejection captured",
    // Safari 自动播放策略限制：RTC SDK 在非用户手势上下文中自动调用 setSinkId() 设置音频输出设备时触发
    // 属于 Safari 的预期行为限制，不影响实际音频播放
    "NotAllowedError: A user gesture is required",
    // RTC SDK 重连机制副作用：连接断开触发 reJoin() 时，挂起的订阅操作被强制中止
    // 属于 SDK 内部预期行为，用户侧表现短暂重连
    "RtcError",
  ],

  // 自定义错误过滤
  beforeSend(event) {
    return event;
  },
});

export default Sentry;
```
