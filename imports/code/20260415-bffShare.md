import { showToast } from "@/hooks/useToast";

  title?: string;
  text: string;
  url: string;

/** pwaBridge 接口声明在 utils/bridge.ts 中，这里仅导入类型 */

/**
 * 尝试调用原生 Bridge 分享（使用 nativeShare 方法）
 */
const tryNativeBridgeShare = (payload: { title?: string; text: string; url: string }): Promise<boolean> => {
  return new Promise((resolve) => {
    if (!window?.pwaBridge?.callHandler) {
      console.log("[shareBffLink] pwaBridge.callHandler 不存在");
      resolve(false);
      return;
    }

    try {
      // 使用 nativeShare 接口，type=text 分享纯文本和链接
      const shareParams = {
        type: "text",
        title: payload.title || "Share",
        text: `${payload.text} ${payload.url}`,
      };

      console.log("[shareBffLink] 调用 pwaBridge.nativeShare，参数:", shareParams);

      window.pwaBridge.callHandler("nativeShare", shareParams, (responseData) => {
        try {
          const response = JSON.parse(responseData);
          if (response.code === 0) {
            console.log("[shareBffLink] nativeShare 成功");
            resolve(true);
          } else {
            console.warn("[shareBffLink] nativeShare 失败:", response.reason);
            resolve(false);
          }
        } catch (error) {
          console.error("[shareBffLink] nativeShare 响应解析失败:", error);
          resolve(false);
        }
      });
    } catch (error) {
      console.error("[shareBffLink] nativeShare 调用异常:", error);
      resolve(false);
    }
  });
};

 * 复制文本到剪贴板
 */
export const copyToClipboard = async (text: string): Promise<boolean> => {
  try {
    if (navigator?.clipboard && window.isSecureContext) {
      await navigator.clipboard.writeText(text);
      showToast("success", "Copied!", 2000);
      return true;
    }
  } catch (error) {
    console.warn("[copyToClipboard] Clipboard API failed, trying fallback:", error);
  }

  // 兼容旧版安卓 WebView: execCommand("copy") 兜底
  try {
    const textarea = document.createElement("textarea");
    textarea.value = text;
    textarea.style.position = "fixed";
    textarea.style.opacity = "0";
    document.body.appendChild(textarea);
    textarea.select();

    const copied = document.execCommand("copy");
    if (document.body.contains(textarea)) {
      document.body.removeChild(textarea);
    }

    if (!copied) {
      showToast("error", "Copy failed", 2000);
      return false;
    }

    showToast("success", "Copied!", 2000);
    return true;
  } catch (error) {
    console.warn("[copyToClipboard] Fallback failed:", error);
    showToast("error", "Copy failed", 2000);
    return false;
  }
};
