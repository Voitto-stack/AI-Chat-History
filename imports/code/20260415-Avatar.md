# Avatar

```tsx
import { useState, useRef, useCallback, useEffect } from "react";
import { useNavigate, useLocation, useSearchParams } from "react-router-dom";
import { useUser } from "@/hooks/useUser";
import { compressImage, readFileAsDataUrl } from "@/utils/fileUtils";
import { toast } from "@/utils/toast";
import { useLoading } from "@/hooks/useLoading";
import {
  getFaceIdToken,
  getFaceIdResult,
  redirectToFaceIdPage,
  generateBizNo,
  generateUUID,
} from "@/utils/faceIdUtils";
import { faceScore } from "@/http/api";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";
import Layout from "@/components/Layout";
import NavigationBar from "@/components/NavigationBar";
import Button from "@/components/Button";
import CropPhotoModal from "@/components/CropPhotoModal";
import PhotoChangeModal from "@/components/PhotoChangeModal";
import iconEditPhotoEdit from "@/assets/images/profile/edit.webp";

export default function Avatar() {
  const navigate = useNavigate();
  const location = useLocation();
  const { userInfo, updateAvatar } = useUser();
  const { showLoading, hideLoading } = useLoading();
  const [searchParams] = useSearchParams();

  // 检测是否从 FaceID 返回
  const isReturningFromFaceId = searchParams.get("scanFaceFinish") === "true";
  const bizId = searchParams.get("biz_id");

  // 图片状态
  const [imageSrc, setImageSrc] = useState<string>("");
  const [imageFile, setImageFile] = useState<File | null>(null);
  const [imageSize, setImageSize] = useState({ width: 0, height: 0 });

  // 弹窗状态
  const [showPhotoModal, setShowPhotoModal] = useState(false);
  const [showCropDialog, setShowCropDialog] = useState(false);
  const [showCancelDialog, setShowCancelDialog] = useState(false);

  // 文件输入引用
  const fileInputRef = useRef<HTMLInputElement>(null);

  const pendingCropRef = useRef(false);

  // 是否有新图片可保存
  const hasNewImage = !!imageFile;

  // 页面加载埋点
  useEffect(() => {
    bpTrack(EventName.pwa_avatar_page_show);
    bpTrack(EventName.pwa_onboarding_avatar_uploaded_page_show);
  }, []);

  // imageSize 就绪后，打开裁剪弹窗（保证裁剪组件拿到正确的尺寸）
  useEffect(() => {
    if (pendingCropRef.current && imageSize.width > 0 && imageSize.height > 0) {
      setShowCropDialog(true);
      pendingCropRef.current = false;
    }
  }, [imageSize]);

  // 处理文件选择
  const handleFileChange = async (event: React.ChangeEvent<HTMLInputElement>) => {
    const file = event.target.files?.[0];
    if (!file) {
      event.target.value = "";
      return;
    }

    try {
      const url = await readFileAsDataUrl(file);

      setImageSrc(url);
      setImageFile(file);

      // 加载图片尺寸，完成后通过 useEffect 打开裁剪弹窗
      const img = new Image();
      img.onload = () => {
        setImageSize({ width: img.naturalWidth, height: img.naturalHeight });
      };
      pendingCropRef.current = true;
      img.src = url;
    } catch (error) {
      console.error("Failed to load image:", error);
      toast.error("Failed to load image, please try again");
    } finally {
      event.target.value = "";
    }
  };

  // 触发文件选择
  const triggerFileInput = (capture: boolean) => {
    const input = fileInputRef.current;
    if (!input) return;

    if (capture) {
      input.setAttribute("capture", "environment");
    } else {
      input.removeAttribute("capture");
    }

    input.click();
    setShowPhotoModal(false);
  };

  // 裁剪完成回调
  const onCropPhoto = useCallback(({ imageUrl, imageFile: croppedFile }: { imageUrl: string; imageFile: File }) => {
    setImageSrc(imageUrl);
    setImageFile(croppedFile);
    setShowCropDialog(false);
  }, []);

  // 点击头像区域
  const handleAvatarClick = () => {
    setShowPhotoModal(true);
  };

  // 处理取消弹窗 - Cancel 按钮（取消头像更新，返回头像上传页）
  const handleCancelVerification = () => {
    console.log("🚫 用户取消头像更新");
    // 清理 sessionStorage
    sessionStorage.removeItem("pendingAvatarFile");
    sessionStorage.removeItem("faceVerifyHistoryStackLength");
    setShowCancelDialog(false);
    // 清空当前选择的图片
    setImageSrc("");
    setImageFile(null);
  };

  // 处理取消弹窗 - Verify now 按钮（重新进入验证）
  const handleRetryVerification = async () => {
    setShowCancelDialog(false);

    const pendingAvatarFile = sessionStorage.getItem("pendingAvatarFile");
    if (!pendingAvatarFile) {
      toast.error("Image data lost, please select photo again");
      return;
    }

    showLoading();

    try {
      // 从 sessionStorage 读取图片
      const response = await fetch(pendingAvatarFile);
      const blob = await response.blob();
      const file = new File([blob], "avatar.jpg", { type: blob.type });

      // 1️⃣ 重新校验颜值分（确保数据一致性）
      console.log("🔍 重新校验颜值分...");
      const passed = await checkBeautyScore(file);
      if (!passed) {
        console.error("❌ 颜值分校验不通过");
        hideLoading();
        return;
      }
      console.log("✅ 颜值分校验通过");

      // 2️⃣ ��值分通过后，重新获取 FaceID Token
      const bizNo = generateBizNo();
      const uuid = generateUUID();
      const returnUrl = `${window.location.origin}${location.pathname}?scanFaceFinish=true`;
      const notifyUrl = window.location.hostname === "localhost" ? "https://app.gracechat.com" : window.location.origin;

      console.log("🔄 重新获取 FaceID Token...");

      const result = await getFaceIdToken({
        bizNo,
        comparisonType: "0",
        uuid,
        imageRef: file,
        returnUrl,
        notifyUrl,
      });

      hideLoading();

      if ("error_message" in result) {
        console.error("❌ FaceID Token 获取失败:", result.error_message);
        toast.error("Failed to start verification, please try again");
        return;
      }

      // 保存 Token 并跳转
      localStorage.setItem("faceIdToken", result.token);
      bpTrack(EventName.pwa_avatar_face_liveness_page_show);
      redirectToFaceIdPage(result.token);
    } catch (error) {
      console.error("❌ 重新验证失败:", error);
      toast.error("Failed to start verification, please try again");
      hideLoading();
    }
  };

  /**
   * 颜值分校验：appearance < 60 不允许通过
   * 与 useVerifyPhoto.ts 中的 checkBeautyScore 逻辑保持一致
   */
  const checkBeautyScore = async (file: File): Promise<boolean> => {
    if (!userInfo?.userId) return false;

    try {
      // 获取图片 base64，压缩后用于打分
      const scoreFile = await compressImage(file, 0.05, 300);
      const scoreBase64 = await readFileAsDataUrl(scoreFile);

      const response = await faceScore({
        userId: userInfo.userId,
        imageUrl: scoreBase64,
      });

      const appearance = response.beautyScore?.appearance ?? 0;
      if (appearance < 60) {
        toast.error("Photo doesn't meet requirements. Please upload a clear, well-lit photo of your face.");
        return false;
      }

      return true;
    } catch (error) {
      console.error("Beauty score check failed:", error);
      toast.error("Photo doesn't meet requirements. Please upload a clear, well-lit photo of your face.");
      return false;
    }
  };

  // 处理从 FaceID 返回后的认证结果
  useEffect(() => {
    const checkFaceIdResult = async () => {
      if (isReturningFromFaceId && bizId) {
        showLoading();
        try {
          const resultData = await getFaceIdResult(bizId);
          console.log("📋 FaceID 验证结果:", resultData);

          // 检查活体检测和人脸比对结果
          if (resultData?.liveness_result?.result === "PASS") {
            const resultRef1 = resultData?.verify_result?.result_ref1;
            const isConfidenceAboveThreshold =
              resultRef1?.confidence !== undefined &&
              resultRef1?.thresholds?.["1e-4"] !== undefined &&
              resultRef1.confidence > resultRef1.thresholds["1e-4"];

            if (isConfidenceAboveThreshold) {
              // 认证成功，上传头像
              console.log("✅ 验证通过，开始上传头像");
              bpTrack(EventName.pwa_avatar_liveness_result, {
                result: "success",
                confidence: resultRef1.confidence,
              });
              bpTrack(EventName.pwa_avatar_face_score_result, {
                confidence: resultRef1.confidence,
                threshold: resultRef1.thresholds?.["1e-4"],
                passed: true,
              });
              bpTrack(EventName.pwa_avatar_verification_success);

              const pendingAvatarFile = sessionStorage.getItem("pendingAvatarFile");
              if (pendingAvatarFile && userInfo?.userId) {
                // 将 base64 转回 File 对象
                const response = await fetch(pendingAvatarFile);
                const blob = await response.blob();
                const file = new File([blob], "avatar.jpg", { type: blob.type });

                const compressedFile = await compressImage(file, 1, 800);
                await updateAvatar({
                  userId: userInfo.userId,
                  avatarFile: compressedFile,
                });

                console.log("✅ 头像上传成功");
                bpTrack(EventName.pwa_avatar_upload_success);
                bpTrack(EventName.pwa_profile_photo_upload, { source: "avatar_page" });
                bpTrack(EventName.pwa_profile_update_success, { field: "avatar" });
                toast.success("Avatar updated successfully!");

                // 清理 sessionStorage
                sessionStorage.removeItem("pendingAvatarFile");
                sessionStorage.removeItem("faceVerifyHistoryStackLength");

                // 延迟返回以显示成功消息
                setTimeout(() => {
                  navigate(-1);
                }, 1500);
              }
            } else {
              // 认证失败：照片不是同一人
              console.error("❌ 人脸比对失败");
              bpTrack(EventName.pwa_avatar_liveness_result, {
                result: "verify_failure",
                confidence: resultRef1?.confidence,
                threshold: resultRef1?.thresholds?.["1e-4"],
              });
              bpTrack(EventName.pwa_avatar_face_score_result, {
                confidence: resultRef1?.confidence,
                threshold: resultRef1?.thresholds?.["1e-4"],
                passed: false,
              });
              bpTrack(EventName.pwa_avatar_verification_failed);
              toast.error("Please upload a real photo of yourself.");
            }
          } else {
            // 活体检测失败
            console.error("❌ 活体检测失败:", resultData?.liveness_result?.result);

            // 检查是否是用户取消
            if (resultData?.status === "CANCELLED") {
              console.log("🚫 用户取消了验证");
              hideLoading();
              navigate(location.pathname, { replace: true });
              setShowCancelDialog(true);
              return;
            }

            bpTrack(EventName.pwa_avatar_liveness_result, {
              result: "liveness_failure",
              liveness_result: resultData?.liveness_result?.result,
            });
            toast.error("Please upload a real photo of yourself.");
          }
        } catch (error) {
          console.error("❌ 获取验证结果失败:", error);
          bpTrack(EventName.pwa_avatar_upload_failed);
          toast.error("Failed to verify photo. Please try again.");
        } finally {
          hideLoading();
          // 清除 URL 参数
          navigate(location.pathname, { replace: true });
        }
      }
    };

    checkFaceIdResult();
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [isReturningFromFaceId, bizId]);

  // 保存头像（与 ProfileEditHead feat/upload-avatar 分支一致：先验证，通过后再上传）
  const handleSave = async () => {
    if (!imageFile || !imageSrc) {
      setShowPhotoModal(true);
      return;
    }

    console.log("🚀 开始头像保存流程（feat/upload-avatar 分支）");
    bpTrack(EventName.pwa_avatar_save_click);
    bpTrack(EventName.pwa_onboarding_avatar_upload_button_click);
    showLoading();

    try {
      // 1️⃣ 先校验颜值分（< 60 不允许通过）
      console.log("🔍 开始颜值分校验...");
      const passed = await checkBeautyScore(imageFile);
      if (!passed) {
        console.error("❌ 颜值分校验不��过");
        hideLoading();
        return;
      }
      console.log("✅ 颜值分校验通过");

      // 2️⃣ 颜值分通过后，保存图片到 sessionStorage 并进行人脸识别
      // 保存图片到 sessionStorage 以便认证后使用
      const reader = new FileReader();
      reader.onload = async () => {
        const base64 = reader.result as string;
        sessionStorage.setItem("pendingAvatarFile", base64);
        console.log("💾 图片已保存到 sessionStorage");

        // 获取 FaceID Token
        const bizNo = generateBizNo();
        const uuid = generateUUID();
        const returnUrl = `${window.location.origin}${location.pathname}?scanFaceFinish=true`;
        const notifyUrl =
          window.location.hostname === "localhost" ? "https://app.gracechat.com" : window.location.origin;

        console.log("🔑 获取 FaceID Token...", { bizNo, uuid, returnUrl });

        const result = await getFaceIdToken({
          bizNo,
          comparisonType: "0",
          uuid,
          imageRef: imageFile,
          returnUrl,
          notifyUrl,
        });

        hideLoading();

        // 处理错误响应
        if ("error_message" in result) {
          console.error("❌ FaceID Token 获取失败:", result.error_message);

          switch (result.error_message) {
            case "NO_FACE_FOUND":
              toast.error("No face detected. Please upload a clear photo showing your face.");
              break;
            case "MULTIPLE_FACES":
              toast.error("Multiple faces detected. Please upload a photo of just yourself.");
              break;
            default:
              toast.error("Failed to verify image, please try another photo");
          }
          return;
        }

        // 保存 Token 并跳转到 FaceID 页面
        console.log("✅ FaceID Token 获取成功，准备跳转");
        localStorage.setItem("faceIdToken", result.token);

        // 保存历史栈长度
        if (!sessionStorage.getItem("faceVerifyHistoryStackLength")) {
          sessionStorage.setItem("faceVerifyHistoryStackLength", window.history.length.toString());
        }

        bpTrack(EventName.pwa_avatar_get_face_id_token);
        bpTrack(EventName.pwa_avatar_face_liveness_page_show);
        bpTrack(EventName.pwa_onboarding_facial_recognition_page_show);

        // 跳转到 FaceID 页面
        redirectToFaceIdPage(result.token);
      };

      reader.readAsDataURL(imageFile);
    } catch (error) {
      console.error("❌ 准备验证失败:", error);
      toast.error("Failed to start verification. Please try again.");
      hideLoading();
    }
  };

  return (
    <Layout showTabBar={false}>
      <div className="flex flex-col h-full bg-surface">
        {/* 顶部导航 */}
        <NavigationBar />

        {/* 内容区域 */}
        <div className="flex-1 flex flex-col px-6 overflow-y-auto">
          {/* 标题 */}
          <h1 className="font-['TTFellows'] text-[24px] font-bold leading-[28px] mt-10 mb-4">Upload Profile Photo</h1>
          <p className="font-['Pangram'] font-normal text-base leading-[21px] tracking-[-0.23px] text-[#3c3c4399] mb-10">
            Select a clear photo where your face is visible
          </p>
          {/* 头像预览区 */}
          <button
            type="button"
            onClick={handleAvatarClick}
            className="relative w-[90%] mx-auto aspect-square rounded-[25px] overflow-hidden border border-dashed border-[#e5e5ea] bg-white/50 cursor-pointer"
          >
            {imageSrc ? (
              <>
                <img src={imageSrc} alt="Avatar" className="w-full h-full object-cover" />
                {/* 编辑 icon */}
                <div className="absolute right-3 bottom-3 w-8 h-8 flex items-center justify-center bg-white rounded-full shadow-sm">
                  <img src={iconEditPhotoEdit} width={18} height={18} alt="edit" />
                </div>
              </>
            ) : (
              <span className="flex items-center justify-center w-full h-full text-brand text-5xl font-normal">+</span>
            )}
          </button>
          {/* 隐藏的文件输入 */}
          <input ref={fileInputRef} type="file" accept="image/*" className="hidden" onChange={handleFileChange} />
        </div>

        {/* 底部按钮 */}
        <div className="shrink-0 px-8 pb-[65px]">
          <Button onClick={handleSave} disabled={!hasNewImage} variant="primary">
            Save
          </Button>
        </div>

        {/* 图片选择弹窗 */}
        <PhotoChangeModal
          open={showPhotoModal}
          onClose={() => setShowPhotoModal(false)}
          onChooseFromAlbum={() => triggerFileInput(false)}
          onTakePhoto={() => triggerFileInput(true)}
        />

        {/* 裁剪弹窗 */}
        {imageFile && (
          <CropPhotoModal
            open={showCropDialog}
            onClose={() => setShowCropDialog(false)}
            avatarUri={imageSrc}
            avatarFile={imageFile}
            onCropPhoto={onCropPhoto}
            initImgSize={imageSize}
          />
        )}

        {/* 取消验证确认弹窗 */}
        {showCancelDialog && (
          <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50">
            <div className="flex flex-col items-center w-[292px] p-4 bg-[#f8fcff] rounded-2xl mx-4">
              <span className="mt-2 text-brand-dark font-bold text-[15px] leading-[22px] text-center">
                Unable to change profile picture. Real-person verification is required.
              </span>
              <div className="flex flex-row mt-5 gap-2 w-full">
                <button
                  type="button"
                  className="flex-1 h-10 border-[0.5px] border-[#6bc4ff66] bg-transparent rounded-[20px] text-black font-medium text-[15px]"
                  onClick={handleCancelVerification}
                >
                  Cancel
                </button>
                <button
                  type="button"
                  className="flex-1 h-10 border-none rounded-[20px] bg-[#6bc4ff] text-black font-medium text-[15px]"
                  onClick={handleRetryVerification}
                >
                  Verify now
                </button>
              </div>
            </div>
          </div>
        )}
      </div>
    </Layout>
  );
}
```
