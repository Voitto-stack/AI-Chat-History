# Chats

```tsx
import { useCallback, useEffect, useMemo, useRef, useState } from "react";
import type { Conversation } from "@tencentcloud/chat";
import { useNavigate } from "react-router-dom";
import ChatItem from "@/components/ChatItem";
import EmptyView from "@/components/EmptyView";
import Skeleton from "@/components/Skeleton";
import Spinner from "@/components/Spinner";
import { useStaggeredList } from "@/hooks/useStaggeredList";
import ScrollToTop from "@/components/ScrollToTop";
import { MissCallBar } from "@/components/MissCallBar";
import { useMissCall } from "@/hooks/useMissCall";
import useChatList, { refreshChatList } from "@/hooks/chatList";
import IMManager from "@/services/IMManager";
import { useUser } from "@/hooks/useUser";
import icChatHaogan from "@/assets/images/ic_chat_haogan.svg";
import icChatHaoganSelected from "@/assets/images/ic_chat_haogan_selected.svg";
import icChatDate from "@/assets/images/ic_chat_date.svg";
import icChatDateSelected from "@/assets/images/ic_chat_date_selected.svg";

enum FilterType {
  Date = "date",
  Intimacy = "intimacy",
}

const FILTER_OPTIONS = [
  { type: FilterType.Intimacy, icon: icChatHaogan, iconSelected: icChatHaoganSelected },
  { type: FilterType.Date, icon: icChatDate, iconSelected: icChatDateSelected },
];

const PAGE_SIZE = 20;

export default function Chats() {
  const { userInfo } = useUser();
  const [isInitializing, setIsInitializing] = useState(() => !IMManager.ready);
  const [filter, setFilter] = useState(FilterType.Date);
  const [totalCount, setTotalCount] = useState(PAGE_SIZE);
  const sentinelRef = useRef<HTMLDivElement>(null);
  const scrollRef = useRef<HTMLDivElement>(null);

  const { chatList } = useChatList();
  const { showMissedCallsModal } = useMissCall();

  useEffect(() => {
    if (IMManager.ready) {
      setIsInitializing(false);
      return;
    }
    const handleReady = () => setIsInitializing(false);
    IMManager.onReady(handleReady);
    return () => IMManager.offReady(handleReady);
  }, []);

  useEffect(() => {
    refreshChatList();
  }, []);

  const sortedList = useMemo(() => {
    if (filter === FilterType.Intimacy) {
      return [...chatList].sort((a, b) => (b.intimacyNum || 0) - (a.intimacyNum || 0));
    }
    return chatList;
  }, [chatList, filter]);

  useEffect(() => {
    setTotalCount(PAGE_SIZE);
  }, [filter]);

  const splitList = useMemo(() => sortedList.slice(0, totalCount), [sortedList, totalCount]);
  const staggerStyles = useStaggeredList(Math.min(splitList.length, 10)); // 只对前10项做动画

  const hasMore = totalCount < sortedList.length;

  const loadMore = useCallback(() => {
    setTotalCount((prev) => prev + PAGE_SIZE);
  }, []);

  useEffect(() => {
    const sentinel = sentinelRef.current;
    if (!sentinel || !hasMore) return;

    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting) loadMore();
      },
      { threshold: 0.1 },
    );
    observer.observe(sentinel);
    return () => observer.disconnect();
  }, [hasMore, loadMore]);

  const navigate = useNavigate();

  const onClick = useCallback(
    (conversation: Conversation) => {
      IMManager.setMessageRead(conversation.conversationID);
      navigate("/chat-detail", { state: { conversationId: conversation.conversationID } });
    },
    [navigate],
  );

  const handleSwitchFilter = useCallback((type: FilterType) => () => setFilter(type), []);

  return (
    <div className="flex h-full flex-col overflow-hidden bg-white px-4">
      <div className="flex items-center justify-between pt-[calc(var(--sat,0px)+25px)] pb-[16px]">
        <span className="font-[Pangram] text-[34px] font-black leading-normal text-[#000610]">Chats</span>
        {/* 筛选按钮 */}
        <div className="flex h-[30px] w-[88px] items-center justify-center rounded-[18px] bg-[#f3f4f6]">
          {FILTER_OPTIONS.map(({ type, icon, iconSelected }) => (
            <button
              key={type}
              type="button"
              className={`flex h-[26px] w-1/2 items-center justify-center border-none p-0 ${
                filter === type ? "rounded-[999px] bg-white" : "bg-transparent"
              }`}
              onClick={handleSwitchFilter(type)}
            >
              <img className="block h-[18px] w-[18px]" src={filter === type ? iconSelected : icon} alt={type} />
            </button>
          ))}
        </div>
      </div>

      {/* 未接来电提示条 */}
      <MissCallBar onView={showMissedCallsModal} />

      {/* 会话列表区域 */}
      <div ref={scrollRef} className="flex flex-1 flex-col overflow-y-auto">
        {isInitializing ? (
          <Skeleton.ChatList />
        ) : sortedList.length > 0 && userInfo ? (
          <>
            {splitList.map((conversation, i) => (
              <div key={conversation.conversationID} className="animate-list-item-in" style={staggerStyles[i]}>
                <ChatItem item={conversation} onClick={onClick} />
              </div>
            ))}
            {hasMore && (
              <div ref={sentinelRef} className="flex justify-center py-3">
                <Spinner size={20} />
              </div>
            )}
          </>
        ) : (
          <div className="flex h-full items-center justify-center">
            <EmptyView />
          </div>
        )}
      </div>
      <ScrollToTop scrollRef={scrollRef} />
    </div>
  );
}
```
