# tiktok

```ts
// TikTok Pixel 追踪
// https://ads.tiktok.com/help/article/standard-events-parameters

/**
 * TikTok Pixel 上报
 * @param eventName 事件名称
 * @param params 事件参数
 */
export async function ttTrack(eventName: string, params: Record<string, unknown> = {}): Promise<void> {
  try {
    if (typeof window.ttq?.track === "function") {
      window.ttq.track(eventName, params);
      console.log(`[TikTok] Event: ${eventName}`, params);
    } else {
      console.warn("[TikTok] SDK not loaded, event dropped:", eventName);
    }
  } catch (e) {
    console.error("[TikTok] Track error:", e);
  }
}

/**
 * 检查 TikTok Pixel 是否已加载
 */
export function isTikTokPixelLoaded(): boolean {
  return typeof window.ttq?.track === "function";
}

```
