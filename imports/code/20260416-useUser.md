# useUser

```ts
/**
 * 用户信息相关的 Hooks
 */

import { useCallback } from "react";
import { useUserStore, UserState } from "@/stores/userStore";
import { queryUserPaypalInfo, bindPaypalId, fetchUserInfo } from "@/http/api";
import {
  editProfile as editProfileApi,
  saveUserInsId,
  updateAvatar as updateAvatarApi,
  deleteUser,
  adminDeletePaypalId,
} from "@/http/userApi";
import type { EditProfileRequest } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";
import { UserServiceCommonCode } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";
import { bpUpdateProfile } from "@/tracking";

// 导出 UserState 供业务使用
export { UserState };

export function useUser() {
  const {
    isReady,
    userState,
    token,
    userInfo,
    cash,
    paypalAccount,
    setToken,
    setUserInfo,
    setPaypalAccount,
    setCash,
    logout,
    setTimUsersig,
    timUsersig,
  } = useUserStore();

  // 刷新用户信息
  const refreshUserInfo = useCallback(async () => {
    try {
      const response = await fetchUserInfo();
      if (response?.uerInfo) {
        setUserInfo(response.uerInfo);

        const source = (response.uerInfo as Record<string, unknown>).source as string | undefined;
        if (source) {
          bpUpdateProfile({ source });
        }
      }
    } catch (error) {
      console.error("Failed to refresh user info:", error);
    }
    return null;
  }, [setUserInfo]);

  // 更新用户资料（自动刷新用户信息）
  const updateProfile = useCallback(
    async (params: Partial<EditProfileRequest>) => {
      await editProfileApi(params);
      await refreshUserInfo();
    },
    [refreshUserInfo],
  );

  // 更新用户头像（自动刷新用户信息）
  const updateAvatar = useCallback(
    async (params: { userId: number; avatarFile: File }) => {
      const response = await updateAvatarApi(params);
      if (!response.isSuccessful) {
        throw new Error(response.message || "Upload photo failed");
      }
      await refreshUserInfo();
    },
    [refreshUserInfo],
  );

  // 绑定 Instagram 账号（自动刷新用户信息）
  const bindInstagram = useCallback(
    async (params: { insId: string; insAvatar: string }) => {
      await saveUserInsId(params);
      await refreshUserInfo();
    },
    [refreshUserInfo],
  );

  // 获取 PayPal 账号信息
  const fetchPaypalAccount = useCallback(async () => {
    try {
      const response = await queryUserPaypalInfo();
      setPaypalAccount(response.paypalId ?? "");
    } catch (error) {
      console.error("Failed to fetch PayPal account:", error);
    }
  }, [setPaypalAccount]);

  // 绑定 PayPal 账号
  const bindPaypal = useCallback(
    async (paypalId: string, rawData?: string) => {
      try {
        const response = await bindPaypalId({ paypalId, rawData });
        if (response.code === UserServiceCommonCode.Success) {
          setPaypalAccount(paypalId);
          return true;
        }
        return false;
      } catch (error) {
        console.error("Failed to bind PayPal account:", error);
        return false;
      }
    },
    [setPaypalAccount],
  );

  /**
   * 删除账户
   * 1. 删除 PayPal 信息
   * 2. 删除用户账户
   */
  // rerender-use-callback: 使用 useCallback 稳定回调引用
  const deleteAccount = useCallback(async () => {
    if (!userInfo?.userId) {
      console.error("No user ID found");
      return false;
    }

    try {
      // 1. 删除 PayPal 信息
      await adminDeletePaypalId(userInfo.userId);

      // 2. 删除用户账户
      const response = await deleteUser(userInfo.userId);

      if (response.isSuccessful) {
        console.log("Account deleted successfully");
        return true;
      }

      console.error("Failed to delete account");
      return false;
    } catch (error) {
      console.error("Failed to delete account:", error);
      return false;
    }
  }, [userInfo]);

  return {
    // 基础状态
    isReady,
    userState,
    token,

    // 基础数据
    userInfo,
    cash,
    paypalAccount,
    timUsersig,

    // Store 方法（供基础设施使用）
    setToken,
    setUserInfo,
    setCash,
    logout,
    setTimUsersig,

    // 业务方法
    refreshUserInfo,
    updateProfile,
    updateAvatar,
    bindInstagram,
    fetchPaypalAccount,
    bindPaypal,
    deleteAccount,
  };
}

export const getUserStoreState = () => useUserStore.getState();
```
