# auth.permissions

```ts
import { type FastifyRequest, type FastifyReply } from "fastify";
import { verifyToken } from "./auth.middleware.js";

interface RoutePermission {
  auth: boolean;
  roles?: string[];
}

const PERMISSIONS: Record<string, RoutePermission> = {
  // Public
  "POST /auth/login": { auth: false },
  "POST /auth/logout": { auth: false },
  "POST /auth/me": { auth: true },
  "POST /auth/change-password": { auth: true },

  // Users — admin only
  "POST /users/list": { auth: true, roles: ["admin"] },
  "POST /users/get": { auth: true, roles: ["admin"] },
  "POST /users/create": { auth: true, roles: ["admin"] },
  "POST /users/update": { auth: true, roles: ["admin"] },
  "POST /users/delete": { auth: true, roles: ["admin"] },

  // Datasources
  "POST /datasources/list": { auth: true },
  "POST /datasources/get": { auth: true },
  "POST /datasources/create": { auth: true, roles: ["admin"] },
  "POST /datasources/update": { auth: true, roles: ["admin"] },
  "POST /datasources/delete": { auth: true, roles: ["admin"] },

  // SQL
  "POST /sql/parse": { auth: true },
  "POST /sql/execute": { auth: true, roles: ["admin", "maintainer", "developer", "viewer"] },
  "POST /sql/databases": { auth: true },
  "POST /sql/tables": { auth: true },

  // Approvals
  "POST /approvals/list": { auth: true },
  "POST /approvals/get": { auth: true },
  "POST /approvals/create": { auth: true, roles: ["admin", "maintainer", "developer"] },
  "POST /approvals/approve": { auth: true, roles: ["admin", "maintainer"] },
  "POST /approvals/reject": { auth: true, roles: ["admin", "maintainer"] },
  "POST /approvals/execute": { auth: true, roles: ["admin", "maintainer"] },

  // Audit
  "POST /logs/list": { auth: true, roles: ["admin", "maintainer"] },

  // Saved SQLs
  "POST /saved-sqls/list": { auth: true },
  "POST /saved-sqls/create": { auth: true },
  "POST /saved-sqls/update": { auth: true },
  "POST /saved-sqls/delete": { auth: true },
};

export async function authHook(req: FastifyRequest, reply: FastifyReply) {
  const key = `${req.method} ${req.routeOptions.url}`;
  const perm = PERMISSIONS[key];
  if (!perm?.auth) return;

  const auth = req.headers.authorization;
  const token = auth?.startsWith("Bearer ") ? auth.slice(7) : undefined;
  if (!token) return reply.code(401).send({ error: "Unauthorized" });

  try {
    req.user = verifyToken(token);
  } catch {
    return reply.code(401).send({ error: "Invalid token" });
  }

  if (perm.roles && !perm.roles.includes(req.user.role)) {
    return reply.code(403).send({ error: "Forbidden" });
  }
}
```
