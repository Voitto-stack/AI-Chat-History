# voiceApi

```ts
/**
 * 语音相关 API
 */

import { UploadVoiceGreetingRequest, UploadVoiceGreetingResponse } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";
import httpClient from "./httpClient";

/**
 * 上传语音问候音频
 *
 * 该接口会完成以下操作：
 * 1. 接收音频文件
 * 2. 语音转文本（后端使用语音识别服务）
 * 3. 内容审核（后端使用 AI 模型审核）
 * 4. 存储音频到云存储
 * 5. 返回音频 URL 和转换后的文本
 *
 * 如果审核不通过，后端应返回错误码和具体原因
 */
export async function uploadVoiceGreeting(params: {
  audioBlob: Blob;
  duration: number;
}): Promise<UploadVoiceGreetingResponse> {
  const arrayBuffer = await params.audioBlob.arrayBuffer();
  const fileUnit8Array = new Uint8Array(arrayBuffer);

  const request: UploadVoiceGreetingRequest = {
    file: fileUnit8Array,
    fileType: "wav",
    second: Math.ceil(params.duration),
    size: params.audioBlob.size,
  };

  return await httpClient.requestPost2(UploadVoiceGreetingRequest, request, UploadVoiceGreetingResponse);
}
```
