# timezone

```ts
/**
 * 时区工具 — 基于 Intl.DateTimeFormat 实现，不依赖浏览器系统时区
 *
 * 使用用户浏览器的实际时区（Intl.DateTimeFormat().resolvedOptions().timeZone），
 * 确保无论开发环境还是线上环境，日期归属都与用户本地认知一致。
 */

/** 用户浏览器时区，如 "America/New_York"、"Asia/Shanghai" 等 */
export const USER_TZ = Intl.DateTimeFormat().resolvedOptions().timeZone;

// 复用 formatter 避免重复创建
const partsFormatter = new Intl.DateTimeFormat("en-US", {
  timeZone: USER_TZ,
  year: "numeric",
  month: "2-digit",
  day: "2-digit",
  hour: "2-digit",
  minute: "2-digit",
  second: "2-digit",
  hourCycle: "h23",
  weekday: "short",
});

interface TzParts {
  year: number;
  month: number; // 1-12
  day: number;
  hour: number;
  minute: number;
  second: number;
  weekday: number; // 0(Sun)..6(Sat)
}

const WEEKDAY_MAP: Record<string, number> = {
  Sun: 0,
  Mon: 1,
  Tue: 2,
  Wed: 3,
  Thu: 4,
  Fri: 5,
  Sat: 6,
};

/** 在用户时区下解析一个时间戳的各部分 */
export function tzParts(ms: number): TzParts {
  const parts = partsFormatter.formatToParts(new Date(ms));
  const get = (type: Intl.DateTimeFormatPartTypes) => parts.find((p) => p.type === type)!.value;

  return {
    year: Number(get("year")),
    month: Number(get("month")),
    day: Number(get("day")),
    hour: Number(get("hour")),
    minute: Number(get("minute")),
    second: Number(get("second")),
    weekday: WEEKDAY_MAP[get("weekday")] ?? 0,
  };
}

/**
 * 在用户时区下，获取给定时间戳所在日期的 00:00:00 对应的 unix 毫秒。
 *
 * 原理：先减去时区下的 h/m/s 得到候选值，再用 tzParts 验证日期是否匹配。
 * DST 切换日（春 +1h / 秋 -1h）可能导致候选值偏移，因此需要校验并修正。
 */
export function tzStartOfDay(ms: number): number {
  const p = tzParts(ms);
  const elapsed = (p.hour * 3600 + p.minute * 60 + p.second) * 1000;
  const candidate = ms - elapsed;

  const check = tzParts(candidate);
  if (check.day === p.day && check.hour === 0) return candidate;

  // DST 边界：spring-forward 时 00:00 可能不存在（跳到 01:00），
  // fall-back 时 00:00 可能出现两次。尝试 ±1h 修正。
  for (const delta of [-3600_000, 3600_000]) {
    const alt = candidate + delta;
    const c = tzParts(alt);
    if (c.day === p.day && c.hour === 0) return alt;
  }
  // 兜底：spring-forward 场景下该天没有 00:00，返回最接近的开始时刻
  return candidate;
}

/** 在用户时区下格式化为 YYYY.MM.DD */
export function tzDateKey(ms: number): string {
  const p = tzParts(ms);
  return `${p.year}.${String(p.month).padStart(2, "0")}.${String(p.day).padStart(2, "0")}`;
}

/** 在用户时区下获取 weekday (0=Sun..6=Sat) */
export function tzWeekday(ms: number): number {
  return tzParts(ms).weekday;
}

/** 在用户时区下格式化为 HH:MM */
export function tzTimeHHMM(ms: number): string {
  const p = tzParts(ms);
  return `${p.hour}:${String(p.minute).padStart(2, "0")}`;
}

/** 在用户时区下格式化为 "Mar 5" 这样的短日期 */
const shortDateFormatter = new Intl.DateTimeFormat("en-US", {
  timeZone: USER_TZ,
  month: "short",
  day: "numeric",
});

export function tzShortDate(ms: number): string {
  return shortDateFormatter.format(new Date(ms));
}

/** 将 YYYY.MM.DD 字符串解析为用户时区下该天的起止 unix 秒 */
export function tzDayRange(dateKey: string): { startSec: number; endSec: number } {
  const [y, m, d] = dateKey.split(".").map(Number);
  // 用正午 UTC 作为近似值，避免 DST 边界导致日期偏移
  const noonUtcMs = Date.UTC(y, m - 1, d, 12, 0, 0);
  const startMs = tzStartOfDay(noonUtcMs);
  const endMs = startMs + 86_400_000 - 1;
  return {
    startSec: Math.floor(startMs / 1000),
    endSec: Math.floor(endMs / 1000),
  };
}
```
