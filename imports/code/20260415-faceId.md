# faceId

```ts
/**
 * FaceID 工具函数 — URL 解析、页面跳转、错误解析
 */

// ==================== URL 工具 ====================

/** 从当前 URL 解析 biz_id 参数 */
export function getBizIdFromUrl(): string | null {
  return new URLSearchParams(window.location.search).get("biz_id");
}

/** 构建旷视回跳 URL */
export function buildReturnUrl(): string {
  return `${window.location.protocol}//${window.location.host}/verifyPhoto?scanFaceFinish=true`;
}

// ==================== 跳转与导航 ====================

/**
 * 跳转到旷视 H5 验证页面（Web 端）
 * @param faceIdUrl 后端返回的完整跳转 URL
 */
export function redirectToFaceIdWeb(faceIdUrl: string): void {
  window.location.href = faceIdUrl;
}

/** 解析后端返回的 message 为已知错误类型 */
export function parseTokenErrorType(message?: string): "NO_FACE_FOUND" | "MULTIPLE_FACES" | "UNKNOWN" {
  if (message === "NO_FACE_FOUND") return "NO_FACE_FOUND";
  if (message === "MULTIPLE_FACES") return "MULTIPLE_FACES";
  return "UNKNOWN";
}
```
