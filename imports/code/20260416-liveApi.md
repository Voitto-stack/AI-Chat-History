# liveApi

```ts
import {
  PwaChangeDispatchStatusRequest,
  PwaChangeDispatchStatusResponse,
} from "@heyhru/business-pwa-proto/gen/archat_api/user_api";
import httpClient from "./httpClient";

// 切换分发状态（开播/停播）
export async function changeDispatchStatus(dispatch: boolean): Promise<void> {
  const request: PwaChangeDispatchStatusRequest = { dispatch };
  await httpClient.requestPost2(PwaChangeDispatchStatusRequest, request, PwaChangeDispatchStatusResponse);
}
```
