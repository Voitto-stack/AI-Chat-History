# Age

```tsx
import { useEffect, useRef, useState } from "react";
import { useOnboarding } from "@/hooks/useOnboarding";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";
import Input from "./Input";
import Button from "@/components/Button";
import { getInstallState, InstallState } from "@/services/pwaInstall";

interface OnboardingAgeProps {
  onContinue: () => void;
}

const DEFAULT_HINT = "We'll match you with people near your age";
const MIN_AGE = 18;
const MAX_AGE = 99;

export default function OnboardingAge({ onContinue }: OnboardingAgeProps) {
  const { inputAge, setInputAge } = useOnboarding();
  const [value, setValue] = useState(inputAge);
  const [error, setError] = useState<string | null>(null);
  const inputRef = useRef<HTMLInputElement>(null);

  useEffect(() => {
    setValue(inputAge);
  }, [inputAge]);

  useEffect(() => {
    requestAnimationFrame(() => {
      inputRef.current?.focus();
      bpTrack(EventName.pwa_onboarding_age_page_show);
      bpTrack(EventName.pwa_conv_set_age_show); // 转化埋点
    });
  }, []);

  const validateValue = (value: string) => {
    if (!/^\d*$/.test(value)) {
      setError("Please enter your age as a number");
      return false;
    }

    const numValue = Number(value);
    if (numValue < MIN_AGE) {
      setError(`Sorry, users under ${MIN_AGE} are not allowed to register`);
      return false;
    }

    if (numValue > MAX_AGE) {
      setError(`Sorry, users over ${MAX_AGE} are not allowed to register`);
      return false;
    }

    setError(null);
    return true;
  };

  const onChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    setValue(Number(event.target.value));
    setError(null);
  };

  const onClick = () => {
    if (!validateValue(value.toString())) return;
    setInputAge(value);
    bpTrack(EventName.pwa_onboarding_age_page_confirm, {
      user_age: value,
    });
    bpTrack(EventName.pwa_conv_set_age_clickButton, {
      user_age: value,
    }); // 转化埋点
    // 埋点：用户填写年龄时 PWA 未安装
    if (getInstallState() !== InstallState.Installed) {
      bpTrack(EventName.pwa_not_installed_when_input_age);
    }
    onContinue();
  };

  const isButtonDisabled = !!error || value === 0;

  return (
    <>
      <Input
        ref={inputRef}
        title="How old are you?"
        error={error}
        hint={!error && <p className="text-[#159cd7] font-medium text-xs leading-normal">{DEFAULT_HINT}</p>}
        inputProps={{
          type: "number",
          value: value === 0 ? "" : value,
          onChange,
          autoComplete: "off",
        }}
      />

      <Button onClick={onClick} disabled={isButtonDisabled}>
        Claim 0.1 $
      </Button>
    </>
  );
}
```
