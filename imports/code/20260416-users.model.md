# users.model

```ts
import { getPgDb } from "@heyhru/server-plugin-pg";
import * as Q from "./users.sql.js";

export function listUsers() {
  return getPgDb().query(Q.LIST);
}

export function getUserById(id: string) {
  return getPgDb().queryOne(Q.FIND_BY_ID, [id]);
}

export function getUserByUsername(username: string) {
  return getPgDb().queryOne(Q.FIND_BY_USERNAME, [username]);
}

export function createUserRow(username: string, email: string, hash: string, role: string) {
  return getPgDb().queryOne(Q.CREATE, [username, email, hash, role]);
}

export async function updateUserRow(id: string, data: { email?: string; role?: string }) {
  const fields: string[] = [];
  const values: unknown[] = [];
  if (data.email) {
    fields.push("email = ?");
    values.push(data.email);
  }
  if (data.role) {
    fields.push("role = ?");
    values.push(data.role);
  }
  if (!fields.length) return getUserById(id);
  fields.push("updated_at = NOW()");
  values.push(id);
  return getPgDb().queryOne(
    `UPDATE users SET ${fields.join(", ")} WHERE id = ? RETURNING id, username, email, role, created_at`,
    values
  );
}

export function deleteUser(id: string) {
  return getPgDb().run(Q.DELETE, [id]);
}

export function updatePasswordHash(id: string, hash: string) {
  return getPgDb().run(Q.UPDATE_PASSWORD(), [hash, id]);
}
```
