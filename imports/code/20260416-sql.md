# sql

```ts
import { client } from "./client";

export interface ExecutePayload {
  dataSourceId: string;
  database?: string;
  sql: string;
  page?: number;
  pageSize?: number;
}

export interface ExecuteResult {
  rows: Record<string, unknown>[];
  total: number;
  page: number;
  pageSize: number;
}

export interface ParseResult {
  category: "select" | "dml";
}

export const parseSQL = (sql: string) => client.post<ParseResult>("/sql/parse", { sql });

export const executeSQL = (payload: ExecutePayload) =>
  client.post<ExecuteResult>("/sql/execute", payload);

export const listDatabases = (dataSourceId: string) =>
  client.post<string[]>("/sql/databases", { dataSourceId });

export const listTables = (dataSourceId: string, database: string) =>
  client.post<string[]>("/sql/tables", { dataSourceId, database });

```
