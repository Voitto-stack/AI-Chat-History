# tagStore

```ts
import { create } from "zustand";
import type { InterestTab as TagTab } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";

interface TagState {
  tagTabs: TagTab[];
  setTagTabs: (tabs: TagTab[]) => void;
}

export const useTagStore = create<TagState>((set) => ({
  tagTabs: [],
  setTagTabs: (tabs) => set({ tagTabs: tabs }),
}));
```
