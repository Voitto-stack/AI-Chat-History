# profile

```ts
import { client } from "./client";

export interface UserProfile {
  id: string;
  name: string;
  age: number;
  gender: "male" | "female" | "other";
  bio: string;
  avatar: string | null;
  photos: string[];
}

const USE_MOCK = true;

const MOCK_PROFILE: UserProfile = {
  id: "mock-user-001",
  name: "Demo User",
  age: 25,
  gender: "male",
  bio: "Just exploring the world",
  avatar: "https://picsum.photos/seed/me/200/200",
  photos: [
    "https://picsum.photos/seed/p1/400/400",
    "https://picsum.photos/seed/p2/400/400",
    "https://picsum.photos/seed/p3/400/400",
  ],
};

function delay(ms: number): Promise<void> {
  return new Promise((r) => setTimeout(r, ms));
}

export async function getProfile(): Promise<UserProfile> {
  if (USE_MOCK) {
    await delay(300);
    return { ...MOCK_PROFILE };
  }
  return client.post<UserProfile>("/user/profile");
}

export async function updateProfile(data: Partial<UserProfile>): Promise<UserProfile> {
  if (USE_MOCK) {
    await delay(400);
    return { ...MOCK_PROFILE, ...data };
  }
  return client.post<UserProfile>("/user/update-profile", data);
}
```
