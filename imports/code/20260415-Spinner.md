# Spinner

```tsx
interface SpinnerProps {
  size?: number;
  color?: string;
  className?: string;
}

export default function Spinner({ size = 32, color = "#69c0ff", className = "" }: SpinnerProps) {
  return (
    <div
      className={`rounded-full animate-spin shrink-0 ${className}`}
      style={{
        width: size,
        height: size,
        border: "2px solid #e8f0fe",
        borderTopColor: color,
      }}
    />
  );
}

```
