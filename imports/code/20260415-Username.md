# Username

```tsx
import { useEffect, useRef, useState } from "react";
import { useOnboarding } from "@/hooks/useOnboarding";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";
import Input from "./Input";
import Button from "@/components/Button";

interface OnboardingUsernameProps {
  onContinue: () => void;
}

const USERNAME_REGEX = /^[a-zA-Z0-9._]+$/;
const MIN_LENGTH = 2;
const MAX_LENGTH = 32;

export default function OnboardingUsername({ onContinue }: OnboardingUsernameProps) {
  const { inputUserName, setInputUserName } = useOnboarding();
  const [value, setValue] = useState(inputUserName);
  const [error, setError] = useState<string | null>(null);
  const inputRef = useRef<HTMLInputElement>(null);

  useEffect(() => {
    inputRef.current?.focus();
    bpTrack(EventName.pwa_onboarding_username_page_show);
    bpTrack(EventName.pwa_conv_set_name_show); // 转化埋点
    // 广告埋点：注册页面显示
    bpTrack(EventName.ad_RegisterPageView);
  }, []);

  const validateValue = (value: string) => {
    if (!value) {
      setError("Please enter your name");
      return false;
    }

    if (!USERNAME_REGEX.test(value)) {
      setError("Name can only contain letters, numbers, underscores (_) and periods (.).");
      return false;
    }

    if (value.length < MIN_LENGTH) {
      setError(`Your username must be longer than ${MIN_LENGTH} characters`);
      return false;
    }

    if (value.length >= MAX_LENGTH) {
      setError(`Your username must be shorter than ${MAX_LENGTH} characters`);
      return false;
    }

    setError(null);
    return true;
  };

  const onChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    const v = event.target.value;
    setValue(v ? v.charAt(0).toUpperCase() + v.slice(1) : v);
    setError(null);
  };

  const onClick = () => {
    if (validateValue(value)) {
      setInputUserName(value);
      bpTrack(EventName.pwa_onboarding_username_page_confirm, {
        user_name: value,
      });
      bpTrack(EventName.pwa_conv_set_name_clickButton, {
        user_name: value,
      }); // 转化埋点
      onContinue();
    }
  };

  const isButtonDisabled = !!error || !value;

  return (
    <>
      <Input
        ref={inputRef}
        title="What's your first name?"
        error={error}
        inputProps={{
          type: "text",
          placeholder: "Enter your name",
          value,
          onChange,
          autoComplete: "off",
          autoCapitalize: "words",
        }}
      />

      <Button onClick={onClick} disabled={isButtonDisabled}>
        Claim 0.1 $
      </Button>
    </>
  );
}
```
