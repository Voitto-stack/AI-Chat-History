# Username

```tsx
import { useEffect, useRef, useState } from "react";
import { useOnboarding } from "@/hooks/useOnboarding";
import { useUser } from "@/hooks/useUser";
import { useLoading } from "@/hooks/useLoading";
import { useGoogleAuth } from "@/hooks/useGoogleAuth";
import { useAuthNavigation } from "@/hooks/useAuthNavigation";
import { startGoogleOAuth } from "@/utils/googleAuth";
import { UserThirdPartyLoginState } from "@sitin/api-proto/gen/archat_api/user_api";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";
import Input from "./Input";
import Button from "@/components/Button";

interface OnboardingUsernameProps {
  onContinue: () => void;
}

const USERNAME_REGEX = /^[a-zA-Z0-9._]+$/;
const MIN_LENGTH = 2;
const MAX_LENGTH = 32;

export default function OnboardingUsername({ onContinue }: OnboardingUsernameProps) {
  const { inputUserName, setInputUserName } = useOnboarding();
  const { userInfo } = useUser();
  const { showLoading } = useLoading();
  const [value, setValue] = useState(inputUserName);
  const [error, setError] = useState<string | null>(null);
  const inputRef = useRef<HTMLInputElement>(null);

  // 处理 Google 账号 OAuth 回调
  useGoogleAuth({ mode: "bind" });

  // 统一的认证导航处理
  useAuthNavigation();

  // 判断是否显示 "Log in" 按钮 只有未绑定第三方账号时才显示
  const shouldShowLogin = userInfo?.userThirdPartyLoginState === UserThirdPartyLoginState.USER_UNAUTHORIZED;

  useEffect(() => {
    inputRef.current?.focus();
    bpTrack(EventName.pwa_onboarding_username_page_show);
  }, []);

  // 处理绑定账号
  const handleBind = () => {
    bpTrack(EventName.pwa_google_login_button_click);
    showLoading();
    startGoogleOAuth();
  };

  const validateValue = (value: string) => {
    if (!value) {
      setError("Your username cannot be empty");
      return false;
    }

    if (!USERNAME_REGEX.test(value)) {
      setError("Your username can only contain letters, numbers, '_' and '.'");
      return false;
    }

    if (value.length < MIN_LENGTH) {
      setError(`Your username must be longer than ${MIN_LENGTH} characters`);
      return false;
    }

    if (value.length >= MAX_LENGTH) {
      setError(`Your username must be shorter than ${MAX_LENGTH} characters`);
      return false;
    }

    setError(null);
    return true;
  };

  const onChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    const v = event.target.value;
    setValue(v ? v.charAt(0).toUpperCase() + v.slice(1) : v);
    setError(null);
  };

  const onClick = () => {
    if (validateValue(value)) {
      setInputUserName(value);
      bpTrack(EventName.pwa_onboarding_username_page_confirm, {
        user_name: value,
      });
      onContinue();
    }
  };

  const isButtonDisabled = !!error || !value;

  return (
    <>
      <div className="flex-1">
        <Input
          ref={inputRef}
          title="What's your first name?"
          error={error}
          inputProps={{
            type: "text",
            placeholder: "Enter your name",
            value,
            onChange,
            autoComplete: "off",
            autoCapitalize: "words",
          }}
        />
      </div>

      <div className="flex flex-col items-center gap-2">
        <Button onClick={onClick} disabled={isButtonDisabled}>
          Claim 0.1 $
        </Button>

        {shouldShowLogin && (
          <p className="m-0 text-[rgba(60,60,67,0.6)] font-normal text-xs leading-normal text-center">
            Already have an account?
            <span className="text-[#6bc4ff] font-medium text-xs underline cursor-pointer ml-0.5" onClick={handleBind}>
              Log in
            </span>
          </p>
        )}
      </div>
    </>
  );
}
```
