import TencentCloudChat, { Conversation } from "@tencentcloud/chat";
import { useState, useEffect } from "react";
import { UserType, UserInfo } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";
import { getUserBasicInfos, getIntimacyBatch } from "@/http/api";
import IMManager, { OnReceiveMsg } from "@/services/IMManager";
import { useChatListStore } from "@/stores/chatListStore";
import { debounce } from "@/utils/debounce";
import { convToUid } from "@/utils/chatUtils";
import { STORAGE_KEYS } from "@/constants/storageKeys";

export type ChatListItem = Conversation & { intimacyNum: number };

/**
 * 拉取全量会话列表
 */
const fetchList = debounce(() => {
  if (!IMManager.ready) return;
  IMManager.getConversationList()
    .then((fullList) => buildList(fullList))
    .then((filtered) => {
      useChatListStore.getState().setChatList(filtered);
      updateIntimacy(); // 列表已显示，后台刷新亲密度
    })
    .catch(() => {});
}, 300);

/**
 * 刷新所有会话的亲密度
 */
const updateIntimacy = debounce(async () => {
  const store = useChatListStore.getState();
  const uids = store.chatList.map((c) => convToUid(c.conversationID)).filter((uid) => !isNaN(uid) && uid > 0);
  if (!uids.length) return;

  try {
    const resp = await getIntimacyBatch(uids);
    const infos = resp?.intimacyInfos ?? [];
    store.setChatList(
      store.chatList.map((conv) => ({
        ...conv,
        intimacyNum:
          infos.find((info) => `C2C${info.userId}` === conv.conversationID)?.levelInfo?.value ?? conv.intimacyNum,
      })),
    );
  } catch {
    /* empty */
  }
}, 1000);

/**
 * 更新聊天列表
 */
function updateList(delta: Conversation[]) {
  const store = useChatListStore.getState();
  const merged = store.chatList
    .map((conv) => {
      const updated = delta.find((d) => d.conversationID === conv.conversationID);
      return updated ? { ...updated, intimacyNum: conv.intimacyNum } : conv;
    })
    .sort((a, b) => (b.lastMessage?.lastTime ?? 0) - (a.lastMessage?.lastTime ?? 0));
  store.setChatList(merged);
}

/**
 * 过滤会话列表，用于显示
 */
async function buildList(list: Conversation[]): Promise<ChatListItem[]> {
  if (!Array.isArray(list)) return [];

  const convs = list.filter((item) => item.type !== TencentCloudChat.TYPES.CONV_SYSTEM);

  //  isAllChatList=1：跳过用户过滤
  if (localStorage.getItem(STORAGE_KEYS.DEV_IS_ALL_CHAT_LIST) === "1") {
    const cachedMap = new Map(useChatListStore.getState().chatList.map((c) => [c.conversationID, c.intimacyNum]));
    return convs.map((c) => ({ ...c, intimacyNum: cachedMap.get(c.conversationID) ?? 0 }));
  }

  // 预计算 uid，避免在后续 filter 里重复解析
  const convWithUid = convs
    .map((conv) => ({ conv, uid: convToUid(conv.conversationID) }))
    .filter(({ uid }) => !isNaN(uid) && uid > 0);

  const userMap = new Map<number, { isPaid?: boolean; isOfficial?: boolean; userType?: UserType }>();

  if (convWithUid.length > 0) {
    try {
      const resp = await getUserBasicInfos(convWithUid.map(({ uid }) => uid));
      const basicInfos = resp?.userBasicInfos ?? [];
      useChatListStore.getState().setUserInfoBatch(basicInfos);
      basicInfos.forEach((info) => {
        if (info.userId !== undefined) {
          userMap.set(info.userId, { isPaid: info.isPaid, isOfficial: info.isOfficial, userType: info.userType });
        }
      });
    } catch {
      // 接口失败时用 store 缓存兜底，避免清空列表
      const cachedUserInfoMap = useChatListStore.getState().userInfoMap;
      convWithUid.forEach(({ uid }) => {
        const info = cachedUserInfoMap[uid];
        if (info) {
          userMap.set(uid, { isPaid: info.isPaid, isOfficial: info.isOfficial, userType: info.userType });
        }
      });
    }
  }

  const cachedMap = new Map(useChatListStore.getState().chatList.map((c) => [c.conversationID, c.intimacyNum]));

  return convWithUid
    .filter(({ conv, uid }) => {
      const info = userMap.get(uid);
      if (!info) return false;
      if (info.userType !== UserType.User) return true; // 非真人（DH/CaiBot）始终展示
      if (info.isOfficial) return true; // 官方号始终展示
      // 需已付费且对方发过消息
      const { peerSendCount } = IMManager.parseConversationCustomData(conv.customData);
      return Boolean(info.isPaid && peerSendCount);
    })
    .map(({ conv }) => ({ ...conv, intimacyNum: cachedMap.get(conv.conversationID) ?? 0 }));
}

// 防止重复注册
let onConvUpdated: ((event: { data: Conversation[] }) => void) | null = null;
let onMsgReceived: OnReceiveMsg | null = null;

/**
 * 初始化聊天列表监听器（在 App 层调用一次）
 */
export function initChatList() {
  if (onConvUpdated) return;
  onConvUpdated = (event: { data: Conversation[] }) => {
    const delta = event.data || [];
    if (!delta.length) return;

    const currentList = useChatListStore.getState().chatList;
    const hasNew = delta.some((d) => !currentList.find((c) => c.conversationID === d.conversationID));
    hasNew ? fetchList() : updateList(delta);
  };
  onMsgReceived = {
    onMessage: async (msg) => {
      const conv = await IMManager.getConversation(msg.conversationID);
      if (!conv) return;

      const currentList = useChatListStore.getState().chatList;
      const exists = currentList.some((c) => c.conversationID === conv.conversationID);
      if (exists) {
        updateList([conv]); // 立即更新 UI（排序/lastMessage）
        updateIntimacy(); // 刷新该会话亲密度
      }
    },
  };

  IMManager.registerConversationUpdateListener(onConvUpdated); // SDK 列表变化
  IMManager.registerReceiveMsgListener(onMsgReceived); // SDK 消息变化

  // 已就绪立即触发，未就绪则等待 SDK_READY 后触发（覆盖首次登录和用户切换两种场景）
  IMManager.onReady(fetchList);
}

/**
 * 清理聊天列表监听器（用户登出时调用）
 */
export function cleanupChatList() {
  fetchList.cancel();
  updateIntimacy.cancel();
  if (onConvUpdated) {
    IMManager.unregisterConversationUpdateListener(onConvUpdated);
    onConvUpdated = null;
  }
  if (onMsgReceived) {
    IMManager.unregisterReceiveMsgListener(onMsgReceived);
    onMsgReceived = null;
  }
  useChatListStore.getState().clearAll();
}

/**
 * 主动刷新聊天列表（进入聊天列表页时调用，有缓存即时显示，后台更新）
 */
export async function refreshChatList(): Promise<void> {
  if (!IMManager.ready) return;
  try {
    const filtered = await buildList(await IMManager.getConversationList());
    useChatListStore.getState().setChatList(filtered);
    updateIntimacy();
  } catch {
    /* empty */
  }
}

/**
 * 读取对应的用户信息
 */
export function useChatUserInfo(conversationID: string): { userInfo: UserInfo | null } {
  const userId = convToUid(conversationID);
  const cachedUserInfo = useChatListStore((s) => s.userInfoMap[userId] ?? null);
  const [fetchedUserInfo, setFetchedUserInfo] = useState<UserInfo | null>(null);
  const userInfo = cachedUserInfo ?? fetchedUserInfo;

  useEffect(() => {
    if (!userId || cachedUserInfo) return;

    getUserBasicInfos([userId])
      .then((resp) => {
        const info = resp?.userBasicInfos?.[0];
        if (info) {
          useChatListStore.getState().setUserInfo(info);
          setFetchedUserInfo(info);
        }
      })
      .catch(() => {});
  }, [userId, cachedUserInfo]);

  return { userInfo };
}

/**
 *  读取聊天列表
 */
function useChatList() {
  const chatList = useChatListStore((s) => s.chatList);
  return { chatList };
}

/**
 * 读取用户匹配卡片信息缓存
 */
export function useMatchCardCache(userId: number | undefined) {
  const cachedUserInfo = useChatListStore((s) => (userId ? s.userInfoMap[userId] : null));
  const setMatchCardInfo = useChatListStore((s) => s.setMatchCardInfo);
  return { cachedUserInfo, setMatchCardInfo };
}

export default useChatList;
