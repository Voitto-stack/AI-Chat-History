# PostList

```tsx
import { useEffect, useRef } from "react";
import { Post } from "@heyhru/business-pwa-proto/gen/archat_api/post_api";
import EmptyView from "@/components/EmptyView";
import PostCard from "@/components/PostCard";
import Spinner from "@/components/Spinner";

interface PostListProps {
  posts: Post[];
  isLoading: boolean;
  hasMore: boolean;
  onLoadMore: () => void;
}

export default function PostList({ posts, isLoading, hasMore, onLoadMore }: PostListProps) {
  const sentinelRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!sentinelRef.current) return;
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting && hasMore && !isLoading) {
          onLoadMore();
        }
      },
      { rootMargin: "200px" },
    );
    observer.observe(sentinelRef.current);
    return () => observer.disconnect();
  }, [hasMore, isLoading, onLoadMore]);

  if (!isLoading && posts.length === 0) {
    return <EmptyView className="py-12" />;
  }

  return (
    <div className="flex flex-col gap-3 p-4">
      {posts.map((post) => (
        <PostCard key={post.id} post={post} />
      ))}
      <div ref={sentinelRef} className="h-px" />
      {isLoading && (
        <div className="flex justify-center py-4">
          <Spinner size={20} />
        </div>
      )}
      {!hasMore && posts.length > 0 && <p className="text-center text-xs text-gray-300 py-4">No more posts</p>}
    </div>
  );
}
```
