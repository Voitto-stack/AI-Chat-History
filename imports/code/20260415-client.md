# client

```ts
/**
 * Minerva Server 请求封装
 * 负责数据解包和 Schema 验证
 *
 * 设计原则：
 * - 后端成功返回 HTTP 2xx + { success: true, data: T }
 * - 后端失败返回 HTTP 非 2xx + { success: false, error: {...} }
 * - 全局 errorHandler 已处理所有错误，此处只需解包 data 并可选验证
 */
import type { ApiSuccessResponse } from '@sitin/minerva-schemas';
import { request as umiRequest } from '@umijs/max';

/**
 * 请求选项
 */
export interface RequestOptions {
  method?: 'GET' | 'POST' | 'PUT' | 'PATCH' | 'DELETE';
  body?: unknown;
  headers?: Record<string, string>;
  skipAuth?: boolean;
}

export interface SchemaParser<T> {
  parse: (value: unknown) => T;
}

/**
 * 发送请求到 Minerva Server
 *
 * @param endpoint - API 路径
 * @param options - 请求选项
 * @param schema - 可选的 Zod schema，用于验证响应数据
 * @returns 解包后的 data（已通过 schema 验证，如果提供了 schema）
 */
export async function request<T>(
  endpoint: string,
  options: RequestOptions = {},
  schema?: SchemaParser<T>,
): Promise<T> {
  const { method = 'GET', body, headers = {}, skipAuth = false } = options;

  const response = await umiRequest<ApiSuccessResponse<T>>(endpoint, {
    method,
    data: body,
    headers: {
      'Content-Type': 'application/json',
      ...headers,
    },
    skipAuth,
  });

  // 可选：使用 Zod schema 验证响应数据
  if (schema) {
    return schema.parse(response.data);
  }

  return response.data;
}

```
