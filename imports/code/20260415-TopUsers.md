# TopUsers

```tsx
import { getAdminUserProfileUrl } from '@/utils/getUrl';
import { EyeOutlined } from '@ant-design/icons';
import type { ProColumns } from '@ant-design/pro-components';
import { ProTable } from '@ant-design/pro-components';
import { useRequest } from 'ahooks';
import { Button, message, Radio, Space } from 'antd';
import { useRef, useState } from 'react';
import { PwaDailyDataItem } from '../../config';
import { queryTopUser, TopUserInfo } from './api';

interface TopUsersProps {
  data: PwaDailyDataItem;
}

const TopUsers = ({ data }: TopUsersProps) => {
  const { userId } = data;
  const [messageApi, contextHolder] = message.useMessage();
  const actionRef = useRef<any>(null);

  // 时间范围筛选：'all' 为全部，7 为近7日
  const [limitDay, setLimitDay] = useState<'all' | number>('all');

  // 调用 Top 用户接口
  const { runAsync: loadTopUsers } = useRequest(queryTopUser, {
    manual: true,
  });

  // 表格请求函数
  const requestTable = async () => {
    try {
      const res = await loadTopUsers({
        user_id: userId,
        limit_day: limitDay === 'all' ? undefined : limitDay,
        top_n: 10,
      });

      if (res.code === 'Success') {
        return {
          data: res.topUserInfo || [],
          success: true,
          total: res.topUserInfo?.length || 0,
        };
      } else {
        messageApi.error(res.message || '获取 Top 用户失败');
        return {
          data: [],
          success: false,
          total: 0,
        };
      }
    } catch (error) {
      messageApi.error('获取 Top 用户失败，请稍后重试');
      return {
        data: [],
        success: false,
        total: 0,
      };
    }
  };

  // 查看用户 Profile
  const handleViewProfile = (id) => {
    const url = getAdminUserProfileUrl(id);
    window.open(url, '_blank', 'noopener,noreferrer');
  };

  // 格式化时长（秒 -> 小时）
  const formatDuration = (seconds?: number): string => {
    const hours = ((seconds || 0) / 3600).toFixed(2);
    return `${hours}h`;
  };

  // 格式化金额
  const formatAmount = (amount?: number): string => {
    return `$${(amount || 0).toFixed(2)}`;
  };

  // 获取排名图标
  const getRankIcon = (rank: number): React.ReactNode => {
    const iconStyle = {
      fontSize: '20px',
      fontWeight: 'bold',
    };

    switch (rank) {
      case 1:
        return <span style={{ ...iconStyle, color: '#FFD700' }}>🥇</span>;
      case 2:
        return <span style={{ ...iconStyle, color: '#C0C0C0' }}>🥈</span>;
      case 3:
        return <span style={{ ...iconStyle, color: '#CD7F32' }}>🥉</span>;
      default:
        return <span style={{ color: '#999', fontWeight: 500 }}>{rank}</span>;
    }
  };

  // 列配置
  const columns: ProColumns<TopUserInfo>[] = [
    {
      title: '排名',
      dataIndex: 'index',
      key: 'rank',
      width: 80,
      align: 'center',
      hideInSearch: true,
      render: (_, __, index) => getRankIcon(index + 1),
    },
    {
      title: '用户名',
      dataIndex: 'userName',
      key: 'userName',
      width: 150,
      hideInSearch: true,
    },
    {
      title: '累计充值金额',
      dataIndex: 'totalChargeAmount',
      key: 'totalChargeAmount',
      width: 150,
      align: 'right',
      hideInSearch: true,
      render: (_, record) => formatAmount(record.totalChargeAmount),
    },
    {
      title: '累计视频时长',
      dataIndex: 'totalVideoDuration',
      key: 'totalVideoDuration',
      width: 140,
      align: 'right',
      hideInSearch: true,
      render: (_, record) => formatDuration(record.totalVideoDuration),
    },
    {
      title: '与该小利女时长',
      dataIndex: 'pwaVideoDuration',
      key: 'pwaVideoDuration',
      width: 150,
      align: 'right',
      hideInSearch: true,
      render: (_, record) => formatDuration(record.pwaVideoDuration),
    },
    {
      title: '操作',
      key: 'action',
      width: 100,
      align: 'center',
      hideInSearch: true,
      render: (_, record) => (
        <Button
          type="link"
          icon={<EyeOutlined />}
          onClick={() => handleViewProfile(record.userId)}
        >
          查看
        </Button>
      ),
    },
  ];

  return (
    <div>
      {contextHolder}
      <ProTable<TopUserInfo>
        actionRef={actionRef}
        columns={columns}
        request={requestTable}
        rowKey="userId"
        search={false}
        options={false}
        pagination={false}
        cardBordered
        toolbar={{
          title: (
            <div style={{ display: 'flex', flexDirection: 'column', gap: '8px' }}>
              <div style={{ color: '#999', fontSize: '12px' }}>
                展示与该小利女视频时长最长的 Top 10 男用户
              </div>
              <Space>
                <span style={{ fontWeight: 500 }}>时间范围：</span>
                <Radio.Group
                  value={limitDay}
                  onChange={(e) => {
                    setLimitDay(e.target.value);
                    // 触发表格刷新
                    setTimeout(() => {
                      actionRef.current?.reload();
                    }, 0);
                  }}
                >
                  <Radio.Button value="all">历史累计</Radio.Button>
                  <Radio.Button value={7}>近7日</Radio.Button>
                </Radio.Group>
              </Space>
            </div>
          ),
        }}
      />
    </div>
  );
};

export default TopUsers;
```
