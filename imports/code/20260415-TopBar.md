import { useCash } from "@/hooks/useCash";
import { useEffect, useRef, useState } from "react";
import { useNavigate } from "react-router-dom";
import { goBack } from "@/utils/navigation";
import { useCashout } from "@/hooks/useCashout";
import { useTask } from "@/hooks/useTask";
import { stageTaskIdsMap } from "@/types/cashout";
import { TaskStatus } from "@/types/task";
import icBundleColorCash from "@/assets/images/ic_bundle_color_cash.webp";
import icProgress from "@/assets/images/ic_progress.svg";
import icGameBack from "@/assets/images/ic_game_back.webp";
import ProgressBar from "@/components/ProgressBar";

interface TopBarProps {
  showProgress?: boolean; // 是否显示进度条
  showBackButton?: boolean; // 是否显示返回按钮
}

/** 使用 requestAnimationFrame 实现平滑数字动画 */
function useAnimatedNumber(target: number, duration = 450) {
  const [current, setCurrent] = useState(target);
  const rafRef = useRef<number | undefined>(undefined);
  const startRef = useRef<{ value: number; time: number } | undefined>(undefined);

  useEffect(() => {
    if (rafRef.current) cancelAnimationFrame(rafRef.current);

    startRef.current = { value: current, time: performance.now() };

    const animate = (now: number) => {
      if (!startRef.current) return;

      const elapsed = now - startRef.current.time;
      const progress = Math.min(elapsed / duration, 1);
      const eased = 1 - Math.pow(1 - progress, 3); // ease-out cubic
      const value = startRef.current.value + (target - startRef.current.value) * eased;

      setCurrent(value);

      if (progress < 1) {
        rafRef.current = requestAnimationFrame(animate);
      } else {
        setCurrent(target);
      }
    };

    rafRef.current = requestAnimationFrame(animate);

    return () => {
      if (rafRef.current) cancelAnimationFrame(rafRef.current);
    };
  }, [target, duration, current]);

  return current;
}

export default function TopBar({ showProgress = true, showBackButton = false }: TopBarProps) {
  const navigate = useNavigate();
  const { cash } = useCash();
  const displayedCash = useAnimatedNumber(cash ?? 0);

  // 计算当前阶段任务完成进度
  const { currentTaskStage } = useCashout();
  const { tasks } = useTask();
  const stageTaskIds = stageTaskIdsMap[currentTaskStage] || [];
  const currentProgress = stageTaskIds.filter((taskId) => {
    const task = tasks.get(taskId);
    return task && task.status === TaskStatus.finish;
  }).length;
  const totalProgress = stageTaskIds.length;
  const percent = totalProgress > 0 ? (currentProgress / totalProgress) * 100 : 0;

  const handleClick = () => {
    navigate("/cashout"); // 跳转到提现页面
  };

  const handleBackClick = (e: React.MouseEvent) => {
    e.stopPropagation(); // 阻止事件冒泡，避免触发 handleClick
    goBack(navigate);
  };

  return (
    <div
      className="sticky top-0 z-50 flex items-center justify-between h-auto px-4 pt-[calc(var(--sat,0px)+14px)] pb-3.5 border-b border-gray-200 bg-white cursor-pointer"
      onClick={handleClick}
    >
      <div className="flex items-center gap-2">
        {showBackButton && (
          <div className="flex items-center justify-center w-8 h-8 cursor-pointer" onClick={handleBackClick}>
            <img src={icGameBack} alt="back" className="w-7 h-7" />
          </div>
        )}

        <div className="flex items-center h-[30px] px-2 py-0.5 gap-1.5 rounded-[20px] bg-gray-100 text-black font-bold text-[15px]">
          <img src={icBundleColorCash} alt="cash" className="w-[22px] h-[22px] object-contain" />
          <span>${displayedCash.toFixed(2)}</span>
        </div>
      </div>

      {showProgress && (
        <div className="flex items-center gap-2">
          <div className="relative w-[88px]">
            <ProgressBar progress={percent} fillColor="#2ab6ff" />
            <img
              className="absolute -top-[9px] left-0 z-[2] pointer-events-none transition-all duration-300"
              style={{ left: `calc(${percent}% - 6px)` }}
              src={icProgress}
              alt="progress"
            />
          </div>
          <span className="text-xs font-medium text-gray-500 whitespace-nowrap shrink-0">
            {currentProgress}/{totalProgress}
          </span>
        </div>
      )}
    </div>
  );
}
