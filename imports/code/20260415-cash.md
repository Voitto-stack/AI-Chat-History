# cash

```ts
import WebsocketManager from "./WebsocketManager";
import { useUserStore } from "@/stores/userStore";
import { S2CCashUpdate } from "@heyhru/business-pwa-proto/gen/ai_tcp/chat_match";
import { OnWsMessageListener } from "./WebSocketConfig";

const TAG = "CashWsHandler";

/**
 * 余额 WebSocket 处理器
 * 负责余额相关的 WebSocket 监听和推送更新逻辑
 */
class CashWsHandler {
  private wsManager: WebsocketManager | null = null;

  // 获取 WebsocketManager 实例（懒加载）
  private getWsManager(): WebsocketManager {
    if (!this.wsManager) {
      this.wsManager = WebsocketManager.getInstance();
    }
    return this.wsManager;
  }

  // 初始化余额处理（在 WebSocket 连接成功时调用）
  public init(): void {
    this.registerCashListener();
  }

  // 注册余额更新监听器
  private registerCashListener(): void {
    const onMessage: OnWsMessageListener<S2CCashUpdate> = {
      onMessage: (update: S2CCashUpdate) => {
        console.log(TAG, "收到余额更新推送:", update);
        const cash = parseFloat(update.cash || "0");
        useUserStore.getState().setCash(cash);

        // 记录更新原因（如果有）
        if (update.reason) {
          console.log(TAG, "余额变更原因:", update.reason);
        }
      },
      resType: () => S2CCashUpdate,
    };

    // 注册余额更新监听器
    this.getWsManager().addMessageListener("S2C_CashUpdate", onMessage);
  }
}

// 导出单例
export const cashHandler = new CashWsHandler();
```
