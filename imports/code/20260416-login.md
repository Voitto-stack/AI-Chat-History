import { useState, useRef, useEffect, useCallback } from "react";
import {
  View,
  Text,
  TextInput,
  TouchableOpacity,
  StyleSheet,
  KeyboardAvoidingView,
  Platform,
  ActivityIndicator,
  Alert,
} from "react-native";
import { useRouter } from "expo-router";
import { sendCode, verifyCode } from "../api/auth";
import { useAuthStore } from "../store/auth";

type LoginMethod = "phone" | "email";

function useLoginFlow() {
  const router = useRouter();
  const setUser = useAuthStore((s) => s.setUser);
  const [method, setMethod] = useState<LoginMethod>("phone");
  const [target, setTarget] = useState("");
  const [code, setCode] = useState("");
  const [codeSent, setCodeSent] = useState(false);
  const [countdown, setCountdown] = useState(0);
  const [sending, setSending] = useState(false);
  const [verifying, setVerifying] = useState(false);
  const codeInputRef = useRef<TextInput>(null);

  useEffect(() => {
    if (countdown <= 0) return;
    const timer = setTimeout(() => setCountdown((c) => c - 1), 1000);
    return () => clearTimeout(timer);
  }, [countdown]);

  const handleSendCode = useCallback(async () => {
    if (!target.trim()) {
      Alert.alert("Error", `Please enter your ${method === "phone" ? "phone number" : "email"}`);
      return;
    }
    setSending(true);
    try {
      await sendCode(target.trim(), method);
      setCodeSent(true);
      setCountdown(60);
      codeInputRef.current?.focus();
    } catch {
      Alert.alert("Error", "Failed to send verification code");
    } finally {
      setSending(false);
    }
  }, [target, method]);

  const handleVerify = useCallback(async () => {
    if (!code.trim()) {
      Alert.alert("Error", "Please enter the verification code");
      return;
    }
    setVerifying(true);
    try {
      const user = await verifyCode(target.trim(), code.trim(), method);
      setUser(user);
      router.replace("/(tabs)");
    } catch {
      Alert.alert("Error", "Invalid verification code");
    } finally {
      setVerifying(false);
    }
  }, [target, code, method, setUser, router]);

  const switchMethod = useCallback(() => {
    setMethod((m) => (m === "phone" ? "email" : "phone"));
    setTarget("");
    setCode("");
    setCodeSent(false);
    setCountdown(0);
  }, []);

  return { method, target, setTarget, code, setCode, codeSent, countdown, sending, verifying, codeInputRef, handleSendCode, handleVerify, switchMethod };
}

export default function LoginScreen() {
  const f = useLoginFlow();

  return (
    <KeyboardAvoidingView style={s.container} behavior={Platform.OS === "ios" ? "padding" : undefined}>
      <View style={s.inner}>
        <LoginHeader />
        <MethodToggle method={f.method} onSwitch={f.switchMethod} />
        <TargetInput method={f.method} value={f.target} onChange={f.setTarget} />
        <LoginActions
          codeSent={f.codeSent}
          sending={f.sending}
          verifying={f.verifying}
          code={f.code}
          countdown={f.countdown}
          codeInputRef={f.codeInputRef}
          onCodeChange={f.setCode}
          onSendCode={f.handleSendCode}
          onVerify={f.handleVerify}
        />
        <LoginFooter method={f.method} onSwitch={f.switchMethod} />
      </View>
    </KeyboardAvoidingView>
  );
}

function LoginHeader() {
  return (
    <View style={s.header}>
      <Text style={s.brand}>KAIRO</Text>
      <Text style={s.subtitle}>Discover meaningful connections</Text>
    </View>
  );
}

function LoginFooter({ method, onSwitch }: { method: LoginMethod; onSwitch: () => void }) {
  return (
    <>
      <TouchableOpacity style={s.switchBtn} onPress={onSwitch}>
        <Text style={s.switchText}>{method === "phone" ? "Use email instead" : "Use phone instead"}</Text>
      </TouchableOpacity>
      <Text style={s.hint}>Mock mode: use code 123456</Text>
    </>
  );
}

function MethodToggle({ method, onSwitch }: { method: LoginMethod; onSwitch: () => void }) {
  return (
    <View style={s.toggleRow}>
      <TouchableOpacity
        style={[s.toggleBtn, method === "phone" && s.toggleActive]}
        onPress={() => method !== "phone" && onSwitch()}
      >
        <Text style={[s.toggleText, method === "phone" && s.toggleTextActive]}>Phone</Text>
      </TouchableOpacity>
      <TouchableOpacity
        style={[s.toggleBtn, method === "email" && s.toggleActive]}
        onPress={() => method !== "email" && onSwitch()}
      >
        <Text style={[s.toggleText, method === "email" && s.toggleTextActive]}>Email</Text>
      </TouchableOpacity>
    </View>
  );
}

function TargetInput({ method, value, onChange }: { method: LoginMethod; value: string; onChange: (v: string) => void }) {
  return (
    <TextInput
      style={s.input}
      placeholder={method === "phone" ? "Phone number" : "Email address"}
      placeholderTextColor="#555"
      value={value}
      onChangeText={onChange}
      keyboardType={method === "phone" ? "phone-pad" : "email-address"}
      autoCapitalize="none"
      autoCorrect={false}
    />
  );
}

interface LoginActionsProps {
  codeSent: boolean;
  sending: boolean;
  verifying: boolean;
  code: string;
  countdown: number;
  codeInputRef: React.RefObject<TextInput | null>;
  onCodeChange: (v: string) => void;
  onSendCode: () => void;
  onVerify: () => void;
}

function LoginActions({ codeSent, sending, verifying, code, countdown, codeInputRef, onCodeChange, onSendCode, onVerify }: LoginActionsProps) {
  if (!codeSent) {
    return (
      <TouchableOpacity style={s.primaryBtn} onPress={onSendCode} disabled={sending}>
        {sending ? <ActivityIndicator color="#fff" /> : <Text style={s.primaryBtnText}>Send Code</Text>}
      </TouchableOpacity>
    );
  }

  return (
    <>
      <TextInput
        ref={codeInputRef}
        style={s.input}
        placeholder="Verification code"
        placeholderTextColor="#555"
        value={code}
        onChangeText={onCodeChange}
        keyboardType="number-pad"
        maxLength={6}
      />
      <TouchableOpacity style={s.primaryBtn} onPress={onVerify} disabled={verifying}>
        {verifying ? <ActivityIndicator color="#fff" /> : <Text style={s.primaryBtnText}>Log In</Text>}
      </TouchableOpacity>
      <TouchableOpacity style={s.resendBtn} onPress={onSendCode} disabled={countdown > 0 || sending}>
        <Text style={[s.resendText, countdown > 0 && s.resendDisabled]}>
          {countdown > 0 ? `Resend in ${countdown}s` : "Resend Code"}
        </Text>
      </TouchableOpacity>
    </>
  );
}
