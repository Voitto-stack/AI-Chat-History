# useTag

```ts
/**
 * 标签管理 Hook
 */

import { useCallback } from "react";
import { useTagStore } from "@/stores/tagStore";
import { getTagTabs } from "@/http/userApi";

export function useTag() {
  const { tagTabs, setTagTabs } = useTagStore();

  // 刷新标签列表
  const refreshTags = useCallback(async () => {
    try {
      const response = await getTagTabs();
      const tabs = response?.tabs ?? [];
      setTagTabs(tabs);
      return tabs;
    } catch {
      return null;
    }
  }, [setTagTabs]);

  return {
    tagTabs,
    refreshTags,
  };
}

```
