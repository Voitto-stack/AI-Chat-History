# bffStore

```ts
/**
 * BFF 裂变功能 Zustand Store
 * 分为两层：overview（轻量）和 dashboard（完整）
 */

import { create } from "zustand";
import { immer } from "zustand/middleware/immer";
import type { BffNotification, BffOverview, BffDashboard } from "@/types/bff";

// ==================== Overview 状态 ====================

interface BffOverviewState {
  overview: BffOverview | null;
  loading: boolean;
  error: string | null;

  // Actions
  setOverview: (data: BffOverview) => void;
  setLoading: (loading: boolean) => void;
  setError: (error: string | null) => void;
  reset: () => void;
}

export const useBffOverviewStore = create<BffOverviewState>()(
  immer((set) => ({
    overview: null,
    loading: false,
    error: null,

    setOverview: (data) =>
      set((state) => {
        state.overview = data;
        state.error = null;
      }),

    setLoading: (loading) =>
      set((state) => {
        state.loading = loading;
      }),

    setError: (error) =>
      set((state) => {
        state.error = error;
      }),

    reset: () =>
      set(() => ({
        overview: null,
        loading: false,
        error: null,
      })),
  })),
);

// ==================== Dashboard 状态 ====================

interface BffDashboardState {
  dashboard: BffDashboard | null;
  loading: boolean;
  error: string | null;
  lastUpdated: number;

  // Actions
  setDashboard: (data: BffDashboard) => void;
  setLoading: (loading: boolean) => void;
  setError: (error: string | null) => void;
  updateLastUpdated: () => void;
  reset: () => void;
}

export const useBffDashboardStore = create<BffDashboardState>()(
  immer((set) => ({
    dashboard: null,
    loading: false,
    error: null,
    lastUpdated: 0,

    setDashboard: (data) =>
      set((state) => {
        state.dashboard = data;
        state.error = null;
        state.lastUpdated = Date.now();
      }),

    setLoading: (loading) =>
      set((state) => {
        state.loading = loading;
      }),

    setError: (error) =>
      set((state) => {
        state.error = error;
      }),

    updateLastUpdated: () =>
      set((state) => {
        state.lastUpdated = Date.now();
      }),

    reset: () =>
      set(() => ({
        dashboard: null,
        loading: false,
        error: null,
        lastUpdated: 0,
      })),
  })),
);

// ==================== 全局通知状态 ====================

interface BffNotificationState {
  notification: BffNotification | null;
  hasUnread: boolean;

  // Actions
  setNotification: (data: BffNotification) => void;
  markAsRead: () => void;
  clear: () => void;
}

export const useBffNotificationStore = create<BffNotificationState>()(
  immer((set) => ({
    notification: null,
    hasUnread: false,

    setNotification: (data) =>
      set((state) => {
        state.notification = data;
        state.hasUnread = data.hasNewEvent;
      }),

    markAsRead: () =>
      set((state) => {
        state.hasUnread = false;
      }),

    clear: () =>
      set(() => ({
        notification: null,
        hasUnread: false,
      })),
  })),
);
```
