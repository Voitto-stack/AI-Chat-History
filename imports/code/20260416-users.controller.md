# users.controller

```ts
import { type FastifyInstance } from "fastify";
import { userList, userGet, userCreate, userUpdate, userDelete } from "@heyhru/business-dms-user";

export function userController(app: FastifyInstance) {
  app.post("/users/list", userList);
  app.post("/users/get", userGet);
  app.post("/users/create", userCreate);
  app.post("/users/update", userUpdate);
  app.post("/users/delete", userDelete);
}

```
