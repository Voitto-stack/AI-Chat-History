/**
 * 使用示例：如何在 PWA 项目中集成 @sitin/http-client
 */

import * as Sentry from "@sentry/react";
import { HttpClient, MonitorAdapter, LoggerAdapter } from "./src";
import { protoMap } from "@sitin/api-proto";
import DeviceUtil from "@sitin/ua";
// 1. 创建 Sentry 监控适配器
const sentryMonitor: MonitorAdapter = {
  getUserInfo() {
    return {} as any;
    // 从你的用户管理系统获取用户信息
    // const userInfo = UserInfoManager.getInstance().curUserInfo;
    // return {
    //   userId: userInfo?.userId || "not_logged_in",
    //   username: userInfo?.username,
    // };
  },

  reportSlowApi(params) {
    Sentry.captureMessage("Slow API Request Detected", {
      level: "warning",
      tags: {
        error_type: "slow_api",
        request_name: params.requestName,
        proto_id: String(params.protoId),
        user_id: String(params.userId),
        duration_ms: params.duration,
      },
      contexts: {
        slow_api: {
          request_name: params.requestName,
          proto_id: params.protoId,
          request_time: params.requestTime,
          request_timestamp: params.requestTimestamp,
          duration_ms: params.duration,
          duration_seconds: (params.duration / 1000).toFixed(2),
          user_id: params.userId,
          username: params.username,
          request_params: params.requestParams,
        },
      },
    });
  },

  reportApiError(params) {
    Sentry.captureException(params.error, {
      tags: {
        error_type: "api_error",
        api_error: true,
        request_url: params.requestUrl,
        request_method: params.requestMethod,
        status_code: params.statusCode,
        proto_id: String(params.protoId),
        request_name: params.requestName,
        user_id: String(params.userId),
      },
      contexts: {
        api_request: {
          url: params.requestUrl,
          method: params.requestMethod,
          status_code: params.statusCode,
          error_message: params.error.message,
          proto_id: params.protoId,
          request_name: params.requestName,
          request_time: params.requestTime,
          request_timestamp: params.requestTimestamp,
          duration_ms: params.duration,
          user_id: params.userId,
          username: params.username,
          request_params: params.requestParams,
          response_data: params.responseData,
        },
      },
      fingerprint: [
        "api-error",
        params.requestName || params.requestUrl,
        String(params.statusCode),
      ],
    });
  },

  reportBusinessError(params) {
    Sentry.captureMessage("Business Code Failure", {
      level: "warning",
      tags: {
        error_type: "business_error",
        business_code: params.businessCode,
        request_name: params.requestName,
        proto_id: String(params.protoId),
        user_id: String(params.userId),
      },
      contexts: {
        business_error: {
          business_code: params.businessCode,
          request_name: params.requestName,
          proto_id: params.protoId,
          user_id: params.userId,
          username: params.username,
          request_params: params.requestParams,
          error_message: params.errorMessage,
          response_data: params.responseData,
        },
      },
    });
  },
};

// 2. 创建自定义日志适配器（可选）
const customLogger: LoggerAdapter = {
  log(...args) {
    console.log("[NetworkUtils]", ...args);
  },
  error(...args) {
    console.error("[NetworkUtils]", ...args);
  },
  warn(...args) {
    console.warn("[NetworkUtils]", ...args);
  },
  info(...args) {
    console.info("[NetworkUtils]", ...args);
  },
};

// 3. 创建 HTTP 客户端实例
export const httpClient = new HttpClient({
  apiHost: "https://api.yourdomain.com",
  timeout: 45000,
  slowApiThreshold: 5000,
  businessErrorCodesToReport: [2, 4, 99, 120],
  protoMap: protoMap,

  // 获取配置信息
  getToken: () => {
    return localStorage.getItem("haven_token") || "";
  },
  getOs: () => {
    // 使用你的 DeviceUtil 获取操作系统
    return DeviceUtil.getOs();
  },
  getAppName: () => "haven_pwa",
  getLocale: () => {
    return DeviceUtil.getLocale();
  },
  getAppVersion: () => {
    // @ts-ignore
    return APP_VERSION || "1.0.0";
  },

  // 环境检查
  isAthenaEnv: () => {
    return false;
    // return import.meta.env.REACT_APP_ENV === "athena";
  },

  // 雅典娜环境允许的请求
  athenaAllowedRequests: [
    "ListTextEarningRequest",
    "ListCallOrderRequest",
    "ListMissCallOrderRequest",
    "GetPwaUserBalanceChangeHistoryRequest",
    "FkUserRequest",
    "FastLoginRequest",
    "QueryUserBalanceRequest",
    "QueryUserWithdrawTasksRequest",
    "GetUserOnlineStatusRequest",
    "QueryUserPaypalInfoRequest",
  ],

  // 401 错误处理
  on401Error: async () => {
    // 可以根据 一些 id 来判断 是哪个app
    // if (appId === "your_app_id") {
    //   // 处理特定应用的401错误
    // }
  },

  // 不需要登录的路径
  noLoginPaths: ["/onboarding", "/getallmylinks.com"],

  // 监控和日志适配器
  monitorAdapter: sentryMonitor,
  loggerAdapter: customLogger,
});

// 4. 导出常用方法（保持与原 NetworkUtils 兼容）
export const requestPost = httpClient.requestPost.bind(httpClient);
export const requestPost2 = httpClient.requestPost2.bind(httpClient);
export const requestPostCache = httpClient.requestPostCache.bind(httpClient);
export const setCache = httpClient.setCache.bind(httpClient);
export const clearNetworkToken = () => {
  localStorage.removeItem("haven_token");
};

// 5. 使用示例
