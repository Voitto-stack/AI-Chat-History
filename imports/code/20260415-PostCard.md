# PostCard

```tsx
import { Post } from "@heyhru/business-pwa-proto/gen/archat_api/post_api";
import ImageGrid from "@/components/ImageGrid";

interface PostCardProps {
  post: Post;
}

export default function PostCard({ post }: PostCardProps) {
  return (
    <article className="flex gap-3">
      <img src={post.authorAvatar || ""} alt="" className="size-12 shrink-0 rounded-full object-cover bg-gray-200" />
      <div className="flex-1 min-w-0">
        <span className="text-xs text-black/40">{post.authorName}</span>
        {post.content && (
          <p className="mt-0.5 mb-3 text-sm text-black/95 leading-relaxed break-words whitespace-pre-wrap select-text">
            {post.content}
          </p>
        )}
        {post.pictures?.length ? <ImageGrid pictures={post.pictures} /> : null}
      </div>
    </article>
  );
}
```
