import { AxiosError } from "axios";
import { BinaryReader, BinaryWriter } from "@bufbuild/protobuf/wire";

/**
 * MessageFns 接口定义（与 protobuf 生成的类型兼容）
 */
export interface MessageFns<T, Type extends string = string> {
  readonly $type: Type;
  encode(message: T, writer?: BinaryWriter): BinaryWriter;
  decode(input: BinaryReader | Uint8Array, length?: number): T;
  fromJSON(object: unknown): T;
  toJSON(message: T): unknown;
  create(base?: unknown): T;
  fromPartial(object: unknown): T;
}

/**
 * HTTP 客户端配置
 */
export interface HttpClientConfig {
  /** API 主机地址 */
  apiHost: string;
  /** 请求超时时间（毫秒），默认 45000 */
  timeout?: number;
  /** 获取授权 token */
  getToken: () => string;
  /** 获取操作系统信息 */
  getOs: () => string;
  /** 获取应用名称 */
  getAppName: () => string;
  /** 获取语言设置 */
  getLocale: () => string;
  /** 获取应用版本 */
  getAppVersion: () => string;
  /** Proto ID 映射表 */
  protoMap: Record<string, number>;
  /** 可选的监控适配器 */
  monitorAdapter?: MonitorAdapter;
  /** 可选的日志适配器 */
  loggerAdapter?: LoggerAdapter;
  /** 可选的环境检查函数 */
  isAthenaEnv?: () => boolean;
  /** 雅典娜环境允许的请求列表 */
  athenaAllowedRequests?: string[];
  /** 需要上报的业务错误码白名单 */
  businessErrorCodesToReport?: number[];
  /** 401 错误时的处理回调 */
  on401Error?: () => void | Promise<void>;
  /** 不需要登录的路径列表 */
  noLoginPaths?: string[];
  /** 慢接口阈值（毫秒），默认 5000 */
  slowApiThreshold?: number;
}

/**
 * 用户信息接口
 */
export interface UserInfo {
  userId: string | number;
  username?: string;
}

/**
 * 监控适配器接口（用于错误上报，如 Sentry）
 */
export interface MonitorAdapter {
  /**
   * 获取用户信息
   */
  getUserInfo(): UserInfo;

  /**
   * 上报慢接口
   */
  reportSlowApi(params: {
    requestName: string;
    protoId: number | string;
    requestTime: string;
    requestTimestamp: number;
    duration: number;
    requestParams: string;
    userId: string | number;
    username?: string;
  }): void;

  /**
   * 上报 API 错误
   */
  reportApiError(params: {
    error: AxiosError;
    requestUrl: string;
    requestMethod: string;
    statusCode: number;
    requestName: string;
    protoId: number | string;
    requestTime: string;
    requestTimestamp: number;
    duration: number;
    requestParams: string;
    responseData: string;
    userId: string | number;
    username?: string;
  }): void;

  /**
   * 上报业务错误
   */
  reportBusinessError(params: {
    businessCode: number;
    requestName: string;
    protoId: number | string;
    requestParams: string;
    errorMessage: string;
    responseData: string;
    userId: string | number;
    username?: string;
  }): void;
}

/**
 * 日志适配器接口
 */
export interface LoggerAdapter {
  log(...args: unknown[]): void;
  error(...args: unknown[]): void;
  warn(...args: unknown[]): void;
  info(...args: unknown[]): void;
}

/**
 * 请求选项
 */
export interface RequestOptions {
  /** 取消信号 */
  signal?: AbortSignal;
  /** 超时时间（毫秒） */
  timeout?: number;
}

/**
 * 缓存记录
 */
export interface CacheRecord<T> {
  data: T;
  expiry: number;
}
