# start

```js
#!/usr/bin/env node

const { spawn } = require("child_process");
const path = require("path");
const fs = require("fs");

/**
 * 启动指定的应用
 * 用法: pnpm start <appname>
 * 示例: pnpm start pwa
 */

const appName = process.argv[2];

if (!appName) {
  console.error("❌ 错误: 请指定要启动的应用名称");
  console.log("\n用法: pnpm start <appname>");
  console.log("示例: pnpm start pwa\n");
  console.log("可用的应用:");

  const appsDir = path.join(__dirname, "../apps");
  if (fs.existsSync(appsDir)) {
    const apps = fs.readdirSync(appsDir).filter((file) => {
      const appPath = path.join(appsDir, file);
      return (
        fs.statSync(appPath).isDirectory() &&
        fs.existsSync(path.join(appPath, "package.json"))
      );
    });
    apps.forEach((app) => console.log(`  - ${app}`));
  }

  process.exit(1);
}

const appPath = path.join(__dirname, "../apps", appName);

// 检查应用是否存在
if (!fs.existsSync(appPath)) {
  console.error(`❌ 错误: 应用 "${appName}" 不存在`);
  console.log(`\n应用路径: ${appPath}`);

  const appsDir = path.join(__dirname, "../apps");
  if (fs.existsSync(appsDir)) {
    const apps = fs.readdirSync(appsDir).filter((file) => {
      const appPath = path.join(appsDir, file);
      return (
        fs.statSync(appPath).isDirectory() &&
        fs.existsSync(path.join(appPath, "package.json"))
      );
    });
    console.log("\n可用的应用:");
    apps.forEach((app) => console.log(`  - ${app}`));
  }

  process.exit(1);
}

// 检查 package.json 是否存在
const packageJsonPath = path.join(appPath, "package.json");
if (!fs.existsSync(packageJsonPath)) {
  console.error(`❌ 错误: 应用 "${appName}" 的 package.json 不存在`);
  process.exit(1);
}

// 读取 package.json 检查是否有 dev 脚本
try {
  const packageJson = JSON.parse(fs.readFileSync(packageJsonPath, "utf-8"));
  if (!packageJson.scripts || !packageJson.scripts.dev) {
    console.error(`❌ 错误: 应用 "${appName}" 没有定义 "dev" 脚本`);
    process.exit(1);
  }
} catch (error) {
  console.error(`❌ 错误: 读取 package.json 失败: ${error.message}`);
  process.exit(1);
}

console.log(`🚀 启动应用: ${appName}`);
console.log(`📁 应用路径: ${appPath}\n`);

// 使用 pnpm 运行 dev 命令
const child = spawn("pnpm", ["--filter", `@sitin/${appName}`, "dev"], {
  stdio: "inherit",
  shell: true,
  cwd: path.join(__dirname, ".."),
});

child.on("error", (error) => {
  console.error(`❌ 启动失败: ${error.message}`);
  process.exit(1);
});

child.on("exit", (code) => {
  if (code !== 0) {
    console.error(`\n❌ 应用退出，退出码: ${code}`);
    process.exit(code);
  }
});

// 处理 Ctrl+C
process.on("SIGINT", () => {
  console.log("\n\n👋 正在停止应用...");
  child.kill("SIGINT");
  process.exit(0);
});
```
