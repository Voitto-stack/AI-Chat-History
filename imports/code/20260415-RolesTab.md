# RolesTab

```tsx
import { PlusOutlined } from '@ant-design/icons';
import {
  ModalForm,
  ProFormText,
  ProFormTextArea,
  ProTable,
  type ActionType,
  type ProColumns,
} from '@ant-design/pro-components';
import { useRequest } from 'ahooks';
import { Button, Drawer, List, message, Popconfirm, Space, Tag } from 'antd';
import dayjs from 'dayjs';
import { useMemo, useRef, useState } from 'react';
import { useAccess } from '@umijs/max';
import {
  assignRolePermissions,
  createRole,
  deleteRole,
  getRole,
  listPermissionsGrouped,
  listRoles,
  updateRole,
  type RoleDetail,
} from '@/api';
import PermissionSelector from './components/PermissionSelector';
import type { RoleItem } from './config';

const RolesTab = () => {
  // Hooks
  const access = useAccess();
  const actionRef = useRef<ActionType>(null);
  const [createOpen, setCreateOpen] = useState(false);
  const [editOpen, setEditOpen] = useState(false);
  const [assignOpen, setAssignOpen] = useState(false);
  const [detailOpen, setDetailOpen] = useState(false);
  const [currentRole, setCurrentRole] = useState<RoleItem | null>(null);
  const [detailRole, setDetailRole] = useState<RoleDetail | null>(null);

  const { runAsync: loadRoles } = useRequest(listRoles, { manual: true });
  const { runAsync: loadRoleDetail } = useRequest(getRole, { manual: true });
  const { runAsync: loadGroupedPermissions, data: groupedPermissions } = useRequest(
    listPermissionsGrouped,
    { manual: true },
  );

  const requestTable = async () => {
    try {
      const roles = await loadRoles();
      return {
        data: roles || [],
        success: true,
        total: roles?.length || 0,
      };
    } catch (error) {
      message.error('角色列表加载失败');
      return { data: [], success: false, total: 0 };
    }
  };

  const handleOpenDetail = async (record: RoleItem) => {
    setDetailOpen(true);
    setDetailRole(null);
    setCurrentRole(record);
    try {
      const detail = await loadRoleDetail(record.id);
      setDetailRole(detail);
    } catch (error) {
      message.error('角色详情加载失败');
    }
  };

  const columns: ProColumns<RoleItem>[] = [
    {
      title: '角色名',
      dataIndex: 'name',
      key: 'name',
    },
    {
      title: '描述',
      dataIndex: 'description',
      key: 'description',
      render: (_, record) => record.description || '-',
    },
    {
      title: '用户数',
      dataIndex: 'userCount',
      key: 'userCount',
    },
    {
      title: '权限数',
      key: 'permissions',
      render: (_, record) => record.permissions?.length || 0,
    },
    {
      title: '创建时间',
      dataIndex: 'createdAt',
      key: 'createdAt',
      render: (_, record) => dayjs(record.createdAt).format('YYYY-MM-DD HH:mm:ss'),
    },
    {
      title: '操作',
      key: 'actions',
      valueType: 'option',
      render: (_, record) => {
        const canRead = access.can('rbac.role.read');
        const canWrite = access.can('rbac.role.write');
        const canDelete = access.can('rbac.role.delete');
        const hasUsers = record.userCount > 0;

        return (
          <Space size={8}>
            {canRead && (
              <Button type="link" onClick={() => handleOpenDetail(record)}>
                查看
              </Button>
            )}
            {canWrite && (
              <Button
                type="link"
                onClick={() => {
                  setCurrentRole(record);
                  setEditOpen(true);
                }}
              >
                编辑
              </Button>
            )}
            {canWrite && (
              <Button
                type="link"
                onClick={() => {
                  setCurrentRole(record);
                  setAssignOpen(true);
                  if (!groupedPermissions) {
                    loadGroupedPermissions();
                  }
                }}
              >
                分配权限
              </Button>
            )}
            {canDelete && (
              <Popconfirm
                title={
                  hasUsers
                    ? '角色仍绑定用户，需先移除用户'
                    : '确认删除该角色？'
                }
                onConfirm={async () => {
                  if (hasUsers) return;
                  await deleteRole(record.id);
                  message.success('删除成功');
                  actionRef.current?.reload();
                }}
                okButtonProps={{ disabled: hasUsers }}
              >
                <Button type="link" danger disabled={hasUsers}>
                  删除
                </Button>
              </Popconfirm>
            )}
          </Space>
        );
      },
    },
  ];

  const groupedReady = useMemo(() => groupedPermissions || {}, [groupedPermissions]);

  return (
    <>
      <ProTable<RoleItem>
        headerTitle="角色管理"
        actionRef={actionRef}
        columns={columns}
        request={requestTable as any}
        rowKey="id"
        search={false}
        options={false}
        scroll={{ x: 'max-content' }}
        columnsState={{
          persistenceKey: 'rbac-roles-table-columns',
          persistenceType: 'localStorage',
        }}
        pagination={{
          defaultPageSize: 10,
          showSizeChanger: true,
          showQuickJumper: true,
        }}
        cardBordered
        toolBarRender={() =>
          access.can('rbac.role.write')
            ? [
                <Button
                  key="create"
                  type="primary"
                  icon={<PlusOutlined />}
                  onClick={() => {
                    setCreateOpen(true);
                    if (!groupedPermissions) {
                      loadGroupedPermissions();
                    }
                  }}
                >
                  新建角色
                </Button>,
              ]
            : []
        }
      />

      <ModalForm
        title="新建角色"
        open={createOpen}
        modalProps={{ destroyOnHidden: true, onCancel: () => setCreateOpen(false) }}
        onFinish={async (values) => {
          await createRole(values);
          message.success('创建成功');
          setCreateOpen(false);
          actionRef.current?.reload();
          return true;
        }}
      >
        <ProFormText
          name="name"
          label="角色名"
          rules={[
            { required: true, message: '请输入角色名' },
            { max: 50, message: '最多 50 字' },
          ]}
        />
        <ProFormTextArea
          name="description"
          label="描述"
          fieldProps={{ maxLength: 255, showCount: true }}
        />
        {groupedPermissions && (
          <PermissionSelector
            name="permissionIds"
            label="权限"
            grouped={groupedReady}
          />
        )}
      </ModalForm>

      <ModalForm
        key={`edit-${currentRole?.id || 'empty'}`}
        title="编辑角色"
        open={editOpen}
        modalProps={{ destroyOnHidden: true, onCancel: () => setEditOpen(false) }}
        initialValues={{
          name: currentRole?.name,
          description: currentRole?.description || '',
        }}
        onFinish={async (values) => {
          if (!currentRole) return false;
          await updateRole(currentRole.id, values);
          message.success('更新成功');
          setEditOpen(false);
          actionRef.current?.reload();
          return true;
        }}
      >
        <ProFormText
          name="name"
          label="角色名"
          rules={[{ max: 50, message: '最多 50 字' }]}
        />
        <ProFormTextArea
          name="description"
          label="描述"
          fieldProps={{ maxLength: 255, showCount: true }}
        />
      </ModalForm>

      <ModalForm
        key={`assign-${currentRole?.id || 'empty'}`}
        title="分配权限"
        open={assignOpen}
        modalProps={{ destroyOnHidden: true, onCancel: () => setAssignOpen(false) }}
        initialValues={{
          permissionIds: currentRole?.permissions?.map((perm) => perm.id) || [],
        }}
        onFinish={async (values) => {
          if (!currentRole) return false;
          await assignRolePermissions(currentRole.id, {
            permissionIds: values.permissionIds || [],
          });
          message.success('分配成功');
          setAssignOpen(false);
          actionRef.current?.reload();
          return true;
        }}
      >
        {groupedPermissions && (
          <PermissionSelector
            name="permissionIds"
            label="权限"
            grouped={groupedReady}
            required
          />
        )}
      </ModalForm>

      <Drawer
        title="角色详情"
        width={600}
        open={detailOpen}
        onClose={() => setDetailOpen(false)}
      >
        <List
          header="绑定用户"
          dataSource={detailRole?.users || []}
          renderItem={(user) => (
            <List.Item key={user.id}>
              <Space size={8}>
                <Tag>{user.sub}</Tag>
                <span>{user.displayName || '-'}</span>
              </Space>
            </List.Item>
          )}
        />
      </Drawer>
    </>
  );
};

export default RolesTab;
```
