```ts
import Router from '@koa/router';
import {
  assignRolePermissionsRequestSchema,
  createRoleRequestSchema,
  updateRoleRequestSchema,
} from '@sitin/minerva-schemas';
import { prisma } from '../../utils/prisma.js';
import { success } from '../../utils/response.js';
import { AppError } from '../../utils/AppError.js';
import { authMiddleware, requirePermission } from '../../middlewares/auth.js';

export const rolesRouter = new Router();

// Apply auth middleware to all routes
rolesRouter.use(authMiddleware);

/**
 * GET /api/rbac/roles
 * List all roles
 */
rolesRouter.get('/', requirePermission('rbac.role.read'), async (ctx) => {
  const roles = await prisma.role.findMany({
    include: {
      rolePermissions: {
        include: {
          permission: true,
        },
      },
      _count: {
        select: {
          userRoles: true,
        },
      },
    },
    orderBy: { name: 'asc' },
  });

  const result = roles.map((role) => ({
    id: role.id,
    name: role.name,
    description: role.description,
    permissions: role.rolePermissions.map((rp) => ({
      id: rp.permission.id,
      key: rp.permission.key,
      description: rp.permission.description,
    })),
    userCount: role._count.userRoles,
    createdAt: role.createdAt,
  }));

  success(ctx, result);
});

/**
 * GET /api/rbac/roles/:id
 * Get role by ID
 */
rolesRouter.get('/:id', requirePermission('rbac.role.read'), async (ctx) => {
  const { id } = ctx.params;

  const role = await prisma.role.findUnique({
    where: { id },
    include: {
      rolePermissions: {
        include: {
          permission: true,
        },
      },
      userRoles: {
        include: {
          user: {
            select: {
              id: true,
              sub: true,
              displayName: true,
            },
          },
        },
      },
    },
  });

  if (!role) {
    throw AppError.notFound('Role not found');
  }

  success(ctx, {
    id: role.id,
    name: role.name,
    description: role.description,
    permissions: role.rolePermissions.map((rp) => ({
      id: rp.permission.id,
      key: rp.permission.key,
      description: rp.permission.description,
    })),
    users: role.userRoles.map((ur) => ur.user),
    createdAt: role.createdAt,
  });
});

/**
 * POST /api/rbac/roles
 * Create a new role
 */
rolesRouter.post('/', requirePermission('rbac.role.write'), async (ctx) => {
  const body = createRoleRequestSchema.parse(ctx.request.body);

  // Check if role name already exists
  const existing = await prisma.role.findUnique({
    where: { name: body.name },
  });

  if (existing) {
    throw AppError.conflict('Role name already exists', 'ROLE_EXISTS');
  }

  // Create role with permissions
  const role = await prisma.role.create({
    data: {
      name: body.name,
      description: body.description,
      rolePermissions: body.permissionIds
        ? {
            create: body.permissionIds.map((permissionId) => ({
              permissionId,
            })),
          }
        : undefined,
    },
    include: {
      rolePermissions: {
        include: {
          permission: true,
        },
      },
    },
  });

  success(
    ctx,
    {
      id: role.id,
      name: role.name,
      description: role.description,
      permissions: role.rolePermissions.map((rp) => ({
        id: rp.permission.id,
        key: rp.permission.key,
      })),
    },
    201
  );
});

/**
 * PATCH /api/rbac/roles/:id
 * Update a role
 */
rolesRouter.patch('/:id', requirePermission('rbac.role.write'), async (ctx) => {
  const { id } = ctx.params;
  const body = updateRoleRequestSchema.parse(ctx.request.body);

  // Check if new name conflicts with existing role
  if (body.name) {
    const existing = await prisma.role.findFirst({
      where: {
        name: body.name,
        NOT: { id },
      },
    });
    if (existing) {
      throw AppError.conflict('Role name already exists', 'ROLE_EXISTS');
    }
  }

  const role = await prisma.role.update({
    where: { id },
    data: body,
  });

  success(ctx, {
    id: role.id,
    name: role.name,
    description: role.description,
  });
});

/**
 * DELETE /api/rbac/roles/:id
 * Delete a role
 */
rolesRouter.delete(
  '/:id',
  requirePermission('rbac.role.delete'),
  async (ctx) => {
    const { id } = ctx.params;

    // Check if role has users assigned
    const roleWithUsers = await prisma.role.findUnique({
      where: { id },
      include: {
        _count: {
          select: { userRoles: true },
        },
      },
    });

    if (!roleWithUsers) {
      throw AppError.notFound('Role not found');
    }

    if (roleWithUsers._count.userRoles > 0) {
      throw AppError.badRequest(
        `Cannot delete role with ${roleWithUsers._count.userRoles} assigned users. Remove users first.`
      );
    }

    // Delete role permissions and role
    await prisma.$transaction([
      prisma.rolePermission.deleteMany({ where: { roleId: id } }),
      prisma.role.delete({ where: { id } }),
    ]);

    success(ctx, { message: 'Role deleted successfully' });
  }
);

/**
 * PUT /api/rbac/roles/:id/permissions
 * Assign permissions to role (replaces existing permissions)
 */
rolesRouter.put(
  '/:id/permissions',
  requirePermission('rbac.role.write'),
  async (ctx) => {
    const { id } = ctx.params;
    const body = assignRolePermissionsRequestSchema.parse(ctx.request.body);

    // Verify role exists
    const role = await prisma.role.findUnique({ where: { id } });
    if (!role) {
      throw AppError.notFound('Role not found');
    }

    // Replace permissions in a transaction
    await prisma.$transaction([
      prisma.rolePermission.deleteMany({ where: { roleId: id } }),
      prisma.rolePermission.createMany({
        data: body.permissionIds.map((permissionId) => ({
          roleId: id,
          permissionId,
        })),
      }),
    ]);

    // Fetch updated role with permissions
    const updatedRole = await prisma.role.findUnique({
      where: { id },
      include: {
        rolePermissions: {
          include: {
            permission: true,
          },
        },
      },
    });

    success(ctx, {
      id: updatedRole!.id,
      name: updatedRole!.name,
      permissions: updatedRole!.rolePermissions.map((rp) => ({
        id: rp.permission.id,
        key: rp.permission.key,
      })),
    });
  }
);
```
