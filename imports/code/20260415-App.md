```tsx
// 运行时配置
import { getCurrentUser } from '@/api';
import requestConfig from '@/request';
import type { InitialState } from '@/types/access';
import { LogoutOutlined } from '@ant-design/icons';
import '@ant-design/v5-patch-for-react-19';
import type { RequestConfig } from '@umijs/max';
import { history } from '@umijs/max';
import { Button, message } from 'antd';
import {
  getSessionToken,
  removeSessionCookie,
  setFeishuAuth,
} from './utils/session';

// 全局初始化数据配置，用于 Layout 用户信息和权限初始化
// 更多信息见文档：https://umijs.org/docs/api/runtime-config#getinitialstate
export async function getInitialState(): Promise<InitialState> {
  const token = getSessionToken();

  // 如果没有 token，返回未登录状态
  if (!token) {
    return { name: 'Minerva', token: null };
  }

  // 尝试获取当前用户信息（含角色和权限）
  try {
    const currentUser = await getCurrentUser();
    console.log('[getInitialState] 当前用户信息:', currentUser);
    if (currentUser.displayName) {
      setFeishuAuth(encodeURIComponent(currentUser.displayName));
    }
    return {
      name: currentUser.displayName || currentUser.sub,
      token,
      currentUser,
    };
  } catch (error) {
    console.error('[getInitialState] 获取当前用户信息失败:', error);
    return { name: 'Minerva', token };
  }
}

// 路由守卫 - 检查登录状态
export function onRouteChange({ location }: any) {
  const token = getSessionToken();
  const isLoginPage = location.pathname === '/login';
  const isFeishuCallback = location.pathname === '/auth/feishu/callback';

  if (!token && !isLoginPage && !isFeishuCallback) {
    history.push('/login');
  }

  if (token && isLoginPage) {
    history.push('/home');
  }
}

export const request: RequestConfig = requestConfig;

export const layout = {
  logo: 'https://app.sitin.ai/assets/image/a7da8bfb7782978bf2ba08189990a202.webp',
  rightContentRender: () => {
    return (
      <Button
        type="text"
        icon={<LogoutOutlined />}
        onClick={() => {
          removeSessionCookie();
          message.success('已退出登录');
          setTimeout(() => {
            history.push('/login');
          }, 500);
        }}
      >
        退出登录
      </Button>
    );
  },
};

```
