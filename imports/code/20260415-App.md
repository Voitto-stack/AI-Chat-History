import { useEffect, useRef } from "react";
import * as Sentry from "@sentry/react";
import { BrowserRouter, useLocation } from "react-router-dom";
import AppRouter from "./router";
import { useSafeArea } from "@/hooks/useSafeArea";
import { FaceIdProvider } from "@/contexts/FaceIdContext";
import GlobalLoading from "@/components/GlobalLoading";
import { GlobalModal } from "@/components/GlobalModal";
import FloatingCameraPreview from "@/components/FloatingCameraPreview";
import Toast from "@/components/Toast";
import { useUser } from "@/hooks/useUser";
import { useUserInit } from "@/hooks/useUserInit";
import { useOnlineStatusReport } from "@/hooks/useOnlineStatusReport";
import { useOfflineEarnings } from "@/hooks/useOfflineEarnings";
import { isApp } from "@/utils/bridge";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";
import "./styles/index.css";

/** H5 fullscreen 模式顶部安全区兜底值（单位：px） */
const H5_FULLSCREEN_SAFE_AREA_TOP_FALLBACK = 24;

function AppContent() {
  const { isReady, userState, userInfo } = useUser();
  const { initUser, cleanUser } = useUserInit();
  const prevUserIdRef = useRef<number | undefined>(undefined);
  const location = useLocation();

  // 页面浏览埋点
  useEffect(() => {
    bpTrack(EventName.pwa_page_view, { path: location.pathname });
  }, [location.pathname]);

  // 在线状态上报（登录后激活）
  useOnlineStatusReport(isReady && !!userInfo?.userId);

  // 离线收益功能
  useOfflineEarnings();

  useEffect(() => {
    if (!isReady) return;

    const userId = userInfo?.userId;
    const prevUserId = prevUserIdRef.current;
    if (userId === prevUserId) return;
    if (prevUserId !== undefined) {
      cleanUser();
    }
    // 更新记录
    prevUserIdRef.current = userId;

    if (userId) {
      // 初始化用户数据
      initUser(userId, userState);
    }
  }, [isReady, userState, userInfo?.userId, initUser, cleanUser]);

  return (
    <>
      {/* <PullToRefresh>{(refreshKey) => <AppRouter key={refreshKey} />}</PullToRefresh> */}
      <AppRouter />
      <FloatingCameraPreview />
      <GlobalLoading />
      <GlobalModal />
      <Toast />
    </>
  );
}

function App() {
  const safeArea = useSafeArea();
  const safeAreaTopRef = useRef(0);

  useEffect(() => {
    safeAreaTopRef.current = safeArea.top;
  }, [safeArea.top]);

  // 安全区域设置
  useEffect(() => {
    document.documentElement.style.setProperty("--safe-area-top", `${safeArea.top}px`);
    document.documentElement.style.setProperty("--safe-area-right", `${safeArea.right}px`);
    document.documentElement.style.setProperty("--safe-area-bottom", `${safeArea.bottom}px`);
    document.documentElement.style.setProperty("--safe-area-left", `${safeArea.left}px`);
  }, [safeArea]);

  // 全屏模式：键盘未弹出时点击进入全屏，键盘弹出时退出全屏
  useEffect(() => {
    const isStandalone = window.matchMedia("(display-mode: standalone)").matches;
    if (isStandalone) return;
    if (isApp()) return; // APK WebView 内不需要浏览器全屏，由原生控制
    if (location.hostname === "localhost" || location.hostname === "127.0.0.1") return;

    let savedSat = "";
    let keyboardVisible = false;

    let fullscreenTimer: ReturnType<typeof setTimeout> | null = null;

    // 这些页面有输入框，全屏会导致键盘闪烁，跳过
    const noFullscreenPaths = ["/onboardingcontainer", "/chat-detail"];

    const enterFullscreen = (e: MouseEvent) => {
      if (keyboardVisible || document.fullscreenElement) return;
      if (noFullscreenPaths.some((p) => window.location.pathname.startsWith(p))) return;
      const target = e.target as HTMLElement;
      if (target.tagName === "INPUT" || target.tagName === "TEXTAREA" || target.isContentEditable) return;
      if (target.closest("input, textarea, [contenteditable]")) return;
      // 延迟进入全屏，如果期间键盘弹出则取消
      if (fullscreenTimer) clearTimeout(fullscreenTimer);
      fullscreenTimer = setTimeout(() => {
        if (!keyboardVisible && !document.fullscreenElement) {
          document.documentElement.requestFullscreen?.().catch(() => {});
        }
      }, 300);
    };

    const exitFullscreen = () => {
      if (document.fullscreenElement) {
        document.exitFullscreen?.().catch(() => {});
      }
    };

    // 通过 visualViewport 检测键盘弹出
    const onViewportResize = () => {
      const viewport = window.visualViewport;
      if (!viewport) return;
      const heightDiff = window.innerHeight - viewport.height;
      const isKeyboardUp = heightDiff > 150;

      if (isKeyboardUp && !keyboardVisible) {
        keyboardVisible = true;
        if (fullscreenTimer) {
          clearTimeout(fullscreenTimer);
          fullscreenTimer = null;
        }
        exitFullscreen();
      } else if (!isKeyboardUp && keyboardVisible) {
        keyboardVisible = false;
      }
    };

    // 全屏切换时处理顶部 safe area padding
    const onFullscreenChange = () => {
      if (document.fullscreenElement) {
        savedSat = getComputedStyle(document.documentElement).getPropertyValue("--sat");
        const fullscreenSafeAreaTop = safeAreaTopRef.current || H5_FULLSCREEN_SAFE_AREA_TOP_FALLBACK;
        document.documentElement.style.setProperty("--sat", `${fullscreenSafeAreaTop}px`);
      } else if (savedSat) {
        document.documentElement.style.setProperty("--sat", savedSat);
      }
    };

    // 输入框获得焦点时立即退出全屏，避免键盘工具栏在全屏→非全屏切换时位置错乱
    const onFocusIn = (e: FocusEvent) => {
      const target = e.target as HTMLElement;
      if (target.tagName === "INPUT" || target.tagName === "TEXTAREA" || target.isContentEditable) {
        if (fullscreenTimer) {
          clearTimeout(fullscreenTimer);
          fullscreenTimer = null;
        }
        exitFullscreen();
      }
    };

    document.addEventListener("click", enterFullscreen);
    document.addEventListener("focusin", onFocusIn);
    document.addEventListener("fullscreenchange", onFullscreenChange);
    window.visualViewport?.addEventListener("resize", onViewportResize);

    return () => {
      if (fullscreenTimer) clearTimeout(fullscreenTimer);
      document.removeEventListener("click", enterFullscreen);
      document.removeEventListener("focusin", onFocusIn);
      document.removeEventListener("fullscreenchange", onFullscreenChange);
      window.visualViewport?.removeEventListener("resize", onViewportResize);
    };
  }, []);

  return (
    <Sentry.ErrorBoundary
      fallback={<></>}
      onError={(error) => bpTrack(EventName.pwa_error, { message: error instanceof Error ? error.message : "unknown" })}
    >
      <BrowserRouter>
        <FaceIdProvider>
          <AppContent />
        </FaceIdProvider>
      </BrowserRouter>
    </Sentry.ErrorBoundary>
  );
}

export default App;
