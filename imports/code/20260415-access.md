# access

```ts
import type { AccessFunctions, InitialState } from '@/types/access';

/**
 * 权限管理配置
 * 基于 RBAC 系统，通过 permissions 数组进行权限控制
 * 参考文档 https://umijs.org/docs/max/access
 */
export default (initialState: InitialState): AccessFunctions => {
  const { currentUser } = initialState || {};
  const permissions = currentUser?.permissions || [];
  const roles = currentUser?.roles || [];
  const isLoggedIn = !!initialState?.token && !!currentUser;

  return {
    /**
     * 检查是否有指定权限
     * @param permission 权限 key，如 'rbac.user.read'
     */
    can: (permission: string) => {
      return permissions.includes(permission);
    },

    /**
     * 检查是否有任一权限
     * @param requiredPermissions 权限 key 数组
     */
    canAny: (requiredPermissions: string[]) => {
      return requiredPermissions.some((p) => permissions.includes(p));
    },

    /**
     * 检查是否有全部权限
     * @param requiredPermissions 权限 key 数组
     */
    canAll: (requiredPermissions: string[]) => {
      return requiredPermissions.every((p) => permissions.includes(p));
    },

    /**
     * 检查是否有指定角色
     * @param role 角色名称
     */
    hasRole: (role: string) => {
      return roles.includes(role);
    },

    /**
     * 检查是否有任一角色
     * @param requiredRoles 角色名称数组
     */
    hasAnyRole: (requiredRoles: string[]) => {
      return requiredRoles.some((r) => roles.includes(r));
    },

    /**
     * 是否已登录
     */
    isLoggedIn,

    /**
     * 是否可以访问 RBAC 管理页面
     * 拥有任一 rbac read 权限即可
     */
    canManageRbac:
      isLoggedIn &&
      (permissions.includes('rbac.user.read') ||
        permissions.includes('rbac.role.read') ||
        permissions.includes('rbac.permission.read')),

    /**
     * 是否可以访问 Lovia 故事管理模块
     * 拥有任一 lovia.story 权限即可
     */
    canAccessLovia:
      isLoggedIn &&
      (permissions.includes('lovia.story.create') ||
        permissions.includes('lovia.story.read') ||
        permissions.includes('lovia.story.update') ||
        permissions.includes('lovia.story.delete')),

    /**
     * 是否可以创建故事
     */
    canCreateStory: isLoggedIn && permissions.includes('lovia.story.create'),

    /**
     * 是否可以查看故事
     */
    canReadStory: isLoggedIn && permissions.includes('lovia.story.read'),

    /**
     * 是否可以更新故事
     */
    canUpdateStory: isLoggedIn && permissions.includes('lovia.story.update'),

    /**
     * 是否可以删除故事
     */
    canDeleteStory: isLoggedIn && permissions.includes('lovia.story.delete'),

    // 以下是为了兼容旧代码，后续可以移除
    canSeeAdmin: isLoggedIn,
  };
};

```
