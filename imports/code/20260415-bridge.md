/**
 * 通过 Bridge 从原生 App 获取安全区域信息
 * @returns 安全区域 insets（单位 px），失败时返回 undefined
 */
export async function getSafeAreaInsets(): Promise<
  { top: number; right: number; bottom: number; left: number } | undefined
> {
  try {
    const response = await invokeBridge("getSafeAreaInsets", "");
    if (response.code === 0 && response.payload && typeof response.payload === "object") {
      const p = response.payload as Record<string, number>;
      return {
        top: p.top ?? 0,
        right: p.right ?? 0,
        bottom: p.bottom ?? 0,
        left: p.left ?? 0,
      };
    }
    return undefined;
  } catch (error) {
    console.error(TAG, "获取安全区域失败:", error);
    return undefined;
  }
}

/**
 * 请求原生权限（仅在 APK 中可用）
 * @param permission 权限类型
 * @returns {Promise<boolean>} 是否授予权限
 */
export async function requestPermission(permission: Permissions): Promise<boolean> {
  if (!isApp()) {
    console.warn(TAG, "requestPermission 仅在 APK 中可用");
    return false;
  }

  console.log(TAG, `开始请求权限: ${permission}`);
  try {
    // 添加 10 秒超时
    const timeoutPromise = new Promise<ResponseData>((_, reject) => {
      setTimeout(() => reject(new Error("Permission request timeout after 10s")), 10000);
    });

    const bridgePromise = invokeBridge("requestPermission", permission);
    const response = await Promise.race([bridgePromise, timeoutPromise]);

    console.log(TAG, `权限 ${permission} 返回结果:`, response);
    const granted = response.code === 0 && response.payload === true;
    console.log(TAG, `权限 ${permission} ${granted ? "已授予" : "被拒绝"}`);
    return granted;
  } catch (error) {
    console.error(TAG, `请求权限 ${permission} 失败:`, error);
    return false;
  }
}

/**
 * 初始化 Bridge
 * 等待 PwaBridgeReady 事件后调用 bridge.init()，再注册所有 native→JS handler
 * 必须在应用启动时调用一次
 */
export function initBridge(): Promise<void> {
  return new Promise<void>((resolve) => {
    if (!window.connectWebViewJavascriptBridge) {
      // 不在 APK 环境，无需初始化
      resolve();
      return;
    }

    window.connectWebViewJavascriptBridge((bridge) => {
      // 初始化 bridge
      bridge.init?.((_message, responseCallback) => {
        if (responseCallback) {
          responseCallback({ "Javascript Responds": "not implemented" });
        }
      });

      // 注册键盘高度变化通知（APK 原生→JS）
      bridge.registerHandler?.("notifyKeyboardChanged", (data: string, responseCallback) => {
        try {
          const params = JSON.parse(data);
          const height = params.show ? params.height : 0;
          // 缩小 body 可视区域，让内容被键盘推上去
          document.body.style.paddingBottom = `${height}px`;
        } catch (e) {
          console.error(TAG, "notifyKeyboardChanged 处理失败:", e);
        }
        responseCallback("success");
      });

      console.log(TAG, "Bridge 初始化完成");
      resolve();
    });
  });
}

function getOs(): string {
  const userAgent = navigator.userAgent.toLowerCase();
  if (userAgent.includes("android")) return "android";
  if (userAgent.includes("iphone") || userAgent.includes("ipad")) return "ios";
  return "web";
}

/**
 * 构建 mailto URL
 */
function buildMailtoUrl(email: string, subject?: string, body?: string): string {
  const params = new URLSearchParams();
  if (subject) params.append("subject", subject);
  if (body) params.append("body", body);
  return `mailto:${email}${params.toString() ? `?${params.toString()}` : ""}`;
}

/**
 * 打开邮件客户端
 * 会在邮箱地址中添加用户信息用于追踪
 *
 * @param email - 邮箱地址
 * @param subject - 邮件主题（可选）
 * @param body - 邮件正文（可选）
 */
export async function openEmailClient(email: string, subject?: string, body?: string): Promise<void> {
  // 获取用户 ID
  const { useUserStore } = await import("@/stores/userStore");
  const userId = useUserStore.getState().userInfo?.userId || "unknown";
  const APP_VERSION = "1.0.0";

  // 在邮箱地址中添加用户追踪信息
  // 例如：contact@gracechat.com -> contact+userId_version_os_locale@gracechat.com
  const enhancedEmail = email.replace("contact", `contact+${userId}_${APP_VERSION}_${getOs()}_${getLocale()}`);

  // 如果不在 App 环境，直接使用 mailto 打开
  if (!isApp()) {
    window.location.href = buildMailtoUrl(enhancedEmail, subject, body);
    return;
  }

  // 在 App 环境，通过 Bridge 调用原生方法
  try {
    await invokeBridge(
      "openEmailClient",
      JSON.stringify({
        email: enhancedEmail,
        subject: subject || "",
        body: body || "",
      }),
    );
  } catch (error) {
    console.error(TAG, "通过 Bridge 打开邮件客户端失败:", error);
    // 降级到 mailto
    window.location.href = buildMailtoUrl(enhancedEmail, subject, body);
  }
}

/**
 * 注册 Bridge Handler（供原生端调用 PWA）
 * @param handlerName 处理器名称
 * @param handler 处理函数
 */
export function registerBridgeHandler(
  handlerName: string,
  handler: (data: unknown, responseCallback?: (response: string) => void) => void,
): void {
  if (!isApp()) {
    console.warn(TAG, `registerBridgeHandler: pwaBridge 不可用, 跳过注册 ${handlerName}`);
    return;
  }

  try {
    window.pwaBridge.registerHandler!(handlerName, (data: string, responseCallback: (response: string) => void) => {
      try {
        const parsedData = typeof data === "string" ? JSON.parse(data) : data;
        handler(parsedData, responseCallback);
      } catch (error) {
        console.error(TAG, `Handler ${handlerName} 执行失败:`, error);
        responseCallback?.("error");
      }
    });
    console.log(TAG, `已注册 Bridge Handler: ${handlerName}`);
  } catch (error) {
    console.error(TAG, `注册 Bridge Handler ${handlerName} 失败:`, error);
  }
}

/**
 * 调用 Bridge 方法
 */
async function invokeBridge(method: string, params?: unknown): Promise<ResponseData> {
  return new Promise<ResponseData>((resolve) => {
    if (window.pwaBridge) {
      try {
        window.pwaBridge.callHandler(method, params, (responseData: string) => {
          try {
            const parsedData: unknown = JSON.parse(responseData);
            if (!isResponseData(parsedData)) {
              resolve({
                code: -503,
                reason: "invalid response format",
                payload: null,
              });
              return;
            }
            resolve(parsedData);
          } catch (error) {
            console.error(TAG, `解析方法 ${method} 返回值失败:`, error);
            resolve({
              code: -502,
              reason: "parse response failed",
              payload: null,
            });
          }
        });
      } catch (error) {
        console.error(TAG, `调用方法 ${method} 失败:`, error);
        resolve({
          code: -500,
          reason: "invoke method failed",
          payload: null,
        });
      }
    } else {
      resolve({
        code: -501,
        reason: "no pwaBridge",
        payload: null,
      });
    }
  });
}

function isResponseData(data: unknown): data is ResponseData {
  if (typeof data !== "object" || data === null) {
    return false;
  }

  const response = data as Partial<ResponseData>;
  return typeof response.code === "number" && typeof response.reason === "string" && "payload" in response;
}
