# gtag

```ts
// Google Analytics (gtag) 追踪
// https://developers.google.com/analytics/devguides/collection/gtagjs

/**
 * Google Analytics (gtag) 上报
 * @param eventName 事件名称
 * @param params 事件参数
 */
export async function gtagTrack(eventName: string, params: Record<string, unknown> = {}): Promise<void> {
  try {
    if (typeof window.gtag === "function") {
      window.gtag("event", eventName, params);
      console.log(`[Google Analytics] Event: ${eventName}`, params);
    } else {
      console.warn("[Google Analytics] gtag not loaded, event dropped:", eventName);
    }
  } catch (e) {
    console.error("[Google Analytics] Track error:", e);
  }
}

/**
 * 检查 Google Analytics 是否已加载
 */
export function isGtagLoaded(): boolean {
  return typeof window.gtag === "function";
}

```
