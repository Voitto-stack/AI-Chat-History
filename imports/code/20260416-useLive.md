# useLive

```ts
/**
 * 直播状态管理 Hook
 */

import { useLiveStore, LiveState } from "@/stores/liveStore";
import { changeDispatchStatus } from "@/http/liveApi";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";
import { STORAGE_KEYS } from "@/constants/storageKeys";

export { LiveState };

export function useLive() {
  const {
    liveState,
    isExitPopupVisible,
    isFloatingVisible,
    hasDispatchClosed,
    setLiveState,
    setExitPopupVisible,
    setFloatingVisible,
    setHasDispatchClosed,
  } = useLiveStore();

  // 切换直播状态，非 Action 时联动隐藏浮窗
  const changeLiveState = (state: LiveState) => {
    // 埋点：Live 状态变化（UV: 93,996 - 最高！）
    bpTrack(EventName.pwa_live_state, {
      from_state: liveState,
      to_state: state,
      live_state: state,
    });

    if (state !== LiveState.Action) {
      setFloatingVisible(false);
    }
    // 进入 Action 状态时重置标记，允许下次关闭时再调接口
    if (state === LiveState.Action) {
      setHasDispatchClosed(false);
    }
    setLiveState(state);
  };

  // 结束直播（只有首次关闭才调接口设为不分发）
  const endLive = () => {
    setExitPopupVisible(false);
    changeLiveState(LiveState.Preview);
    if (!hasDispatchClosed) {
      setHasDispatchClosed(true);
      changeDispatchStatus(false);
    }

    changeDispatchStatus(false);
    // 埋点：Live 结束
    const startTime = Number(localStorage.getItem(STORAGE_KEYS.LIVE_START_UTC_TIME) || 0);
    const call_duration = startTime ? Math.floor((Date.now() - startTime) / 1000) : 0;
    bpTrack(EventName.pwa_live_stop, { call_duration });
    bpTrack(EventName.pwa_stop_live);
  };

  // 关闭退出弹窗（继续直播）
  const keepEarning = () => {
    setExitPopupVisible(false);
    // 埋点：继续赚钱
    bpTrack(EventName.pwa_live_keep_earning);
  };

  // 请求退出直播（显示确认弹窗）
  const requestExit = () => {
    setExitPopupVisible(true);
    // 埋点：请求关闭 Live
    const startTime = Number(localStorage.getItem(STORAGE_KEYS.LIVE_START_UTC_TIME) || 0);
    const call_duration = startTime ? Math.floor((Date.now() - startTime) / 1000) : 0;
    bpTrack(EventName.pwa_live_close, { call_duration, from: "live_page" });
  };

  return {
    // 状态
    liveState,
    isExitPopupVisible,
    isFloatingVisible,

    // 业务方法
    changeLiveState,
    setFloatingVisible,
    endLive,
    keepEarning,
    requestExit,
    setLiveState,
  };
}

export const getLiveStoreState = () => useLiveStore.getState();
```
