# bffStore

```ts
/**
 * BFF 裂变功能 Zustand Store
 * 分为两层：overview（轻量）和 dashboard（完整）
 */

import { create } from "zustand";
import type { BffNotification, BffOverview, BffDashboard } from "@/types/bff";

// ==================== Overview 状态 ====================

interface BffOverviewState {
  overview: BffOverview | null;
  loading: boolean;
  error: string | null;

  // Actions
  setOverview: (data: BffOverview) => void;
  setLoading: (loading: boolean) => void;
  setError: (error: string | null) => void;
  reset: () => void;
}

export const useBffOverviewStore = create<BffOverviewState>((set) => ({
  overview: null,
  loading: false,
  error: null,

  setOverview: (data) =>
    set({
      overview: data,
      error: null,
    }),

  setLoading: (loading) =>
    set({
      loading,
    }),

  setError: (error) =>
    set({
      error,
    }),

  reset: () =>
    set({
      overview: null,
      loading: false,
      error: null,
    }),
}));

// ==================== Dashboard 状态 ====================

interface BffDashboardState {
  dashboard: BffDashboard | null;
  loading: boolean;
  error: string | null;
  lastUpdated: number;

  // Actions
  setDashboard: (data: BffDashboard) => void;
  setLoading: (loading: boolean) => void;
  setError: (error: string | null) => void;
  updateLastUpdated: () => void;
  reset: () => void;
}

export const useBffDashboardStore = create<BffDashboardState>((set) => ({
  dashboard: null,
  loading: false,
  error: null,
  lastUpdated: 0,

  setDashboard: (data) =>
    set({
      dashboard: data,
      error: null,
      lastUpdated: Date.now(),
    }),

  setLoading: (loading) =>
    set({
      loading,
    }),

  setError: (error) =>
    set({
      error,
    }),

  updateLastUpdated: () =>
    set({
      lastUpdated: Date.now(),
    }),

  reset: () =>
    set({
      dashboard: null,
      loading: false,
      error: null,
      lastUpdated: 0,
    }),
}));

// ==================== 全局通知状态 ====================

interface BffNotificationState {
  notification: BffNotification | null;
  hasUnread: boolean;

  // Actions
  setNotification: (data: BffNotification) => void;
  markAsRead: () => void;
  clear: () => void;
}

export const useBffNotificationStore = create<BffNotificationState>((set) => ({
  notification: null,
  hasUnread: false,

  setNotification: (data) =>
    set({
      notification: data,
      hasUnread: data.hasNewEvent,
    }),

  markAsRead: () =>
    set({
      hasUnread: false,
    }),

  clear: () =>
    set({
      notification: null,
      hasUnread: false,
    }),
}));

```
