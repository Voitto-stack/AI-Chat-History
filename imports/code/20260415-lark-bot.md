/**

const { execSync } = require("child_process");

/**
 * 日志函数
 * @param {string} type - 日志类型: info, success, error, warn
 * @param {string} message - 日志消息
 */
function log(type, message) {
  const prefix =
    {
      info: "[INFO]",
      success: "[SUCCESS]",
      error: "[ERROR]",
      warn: "[WARN]",
    }[type] || "[LOG]";
  console.log(`${prefix} ${message}`);
}

/**
 * 发送飞书通知
 * @param {string|object} messageOrPayload - 文本消息字符串 或 完整的飞书消息 payload 对象
 * @param {string} webhook - 飞书 Webhook URL
 */
async function sendFeishuNotification(messageOrPayload, webhook) {
  if (!webhook) {
    log("warn", "未配置 FEISHU_WEBHOOK，跳过通知");
    return;
  }

  const payload =
    typeof messageOrPayload === "string"
      ? { msg_type: "text", content: { text: messageOrPayload } }
      : messageOrPayload;

  try {
    const https = require("https");
    const url = require("url");
    const parsedUrl = url.parse(webhook);

    return new Promise((resolve, reject) => {
      const req = https.request(
        {
          hostname: parsedUrl.hostname,
          port: parsedUrl.port || 443,
          path: parsedUrl.path,
          method: "POST",
          headers: {
            "Content-Type": "application/json",
          },
        },
        (res) => {
          let data = "";
          res.on("data", (chunk) => (data += chunk));
          res.on("end", () => {
            if (res.statusCode === 200) {
              try {
                const result = JSON.parse(data);
                if (result.code && result.code !== 0) {
                  log(
                    "error",
                    `飞书通知业务错误: code=${result.code}, msg=${result.msg}`,
                  );
                  reject(
                    new Error(`Feishu error: ${result.code} - ${result.msg}`),
                  );
                  return;
                }
              } catch (_) {}
              log("info", "✅ 飞书通知发送成功");
              resolve(data);
            } else {
              log("error", `飞书通知发送失败: ${res.statusCode} - ${data}`);
              reject(new Error(`HTTP ${res.statusCode}`));
            }
          });
        },
      );

      req.on("error", (error) => {
        log("error", `飞书通知发送失败: ${error.message}`);
        reject(error);
      });

      req.write(JSON.stringify(payload));
      req.end();
    });
  } catch (error) {
    log("error", `发送飞书通知异常: ${error.message}`);
  }
}

/**
 * 获取 Git 信息（从 Jenkins 环境变量读取）
 * @returns {{branch: string, commit: string, commitAuthor: string, commitMessage: string, repoUrl: string}}
 */
function getGitInfo() {
  const branch = process.env.GIT_BRANCH || "unknown";

  const commit = (process.env.GIT_COMMIT || "unknown").substring(0, 7);

  let commitAuthor = "";
  let commitMessage = "";

  try {
    const raw = execSync(`git log -n 1 ${commit}`, { encoding: "utf-8" });
    const lines = raw.split("\n");
    commitAuthor = (lines[1] || "").replace(/^\s*Author:\s*/, "");
  } catch (_) {}

  try {
    const raw = execSync(`git log -1 --format=%B ${commit}`, {
      encoding: "utf-8",
    }).trim();
    commitMessage = raw.replace(/"/g, "");
  } catch (_) {}

  // 获取远程仓库 URL 并转为 HTTPS 浏览链接
  let repoUrl = "";
  try {
    const remoteUrl = execSync("git remote get-url origin", {
      encoding: "utf-8",
    }).trim();
    // 将 git@github.com:org/repo.git 转为 https://github.com/org/repo
    repoUrl = remoteUrl
      .replace(/^git@([^:]+):/, "https://$1/")
      .replace(/\.git$/, "");
  } catch (_) {}

  return { branch, commit, commitAuthor, commitMessage, repoUrl };
}

/**
 * 构建飞书 Interactive Card 消息体
 * @param {object} opts - 卡片选项
 * @returns {object} 飞书 interactive 消息 payload
 */
function buildDeployCard(opts) {
  const {
    status,
    projectName,
    branch,
    branchUrl,
    commit,
    commitAuthor,
    commitMessage,
    environment,
    duration,
    imageTag,
    bucketName,
    errorMessage,
    appName,
    atUsers, // [{openId: string, name: string}]
  } = opts;

  const isSuccess = status === "success";
  const minutes = Math.floor(duration / 60);
  const seconds = duration % 60;

  const elements = [];

  // 分支 & 提交
  const branchText = branchUrl ? `[${branch}](${branchUrl})` : branch;
  const gitLines = [`**分支:** ${branchText}  |  **提交:** ${commit}`];
  if (commitAuthor) gitLines.push(`**提交作者:** ${commitAuthor}`);
  if (commitMessage) gitLines.push(`**提交信息:** ${commitMessage}`);
  elements.push({
    tag: "div",
    text: {
      tag: "lark_md",
      content: gitLines.join("\n"),
    },
  });

  elements.push({ tag: "hr" });

  // 核心字段（纵向排列）
  const lines = [
    `**状态:** ${isSuccess ? "✅ 成功" : "❌ 失败"}`,
    `**环境:** ${environment || "-"}`,
    `**耗时:** ${minutes}m ${seconds}s`,
  ];
  if (appName) {
    lines.push(`**应用:** ${appName}`);
  }
  elements.push({
    tag: "div",
    text: { tag: "lark_md", content: lines.join("\n") },
  });

  // 镜像 / Bucket 补充信息
  if (imageTag || bucketName) {
    elements.push({ tag: "hr" });
    const extra = [];
    if (imageTag)
      extra.push({
        is_short: true,
        text: { tag: "lark_md", content: `**镜像:**\n${imageTag}` },
      });
    if (bucketName)
      extra.push({
        is_short: true,
        text: { tag: "lark_md", content: `**Bucket:**\n${bucketName}` },
      });
    elements.push({ tag: "div", fields: extra });
  }

  // 错误信息
  if (!isSuccess && errorMessage) {
    elements.push({ tag: "hr" });
    elements.push({
      tag: "div",
      text: { tag: "lark_md", content: `**错误信息:**\n${errorMessage}` },
    });
  }

  // @ 提醒（prod 环境）
  if (atUsers && atUsers.length > 0) {
    const atText = atUsers
      .map((u) => `<at id=${u.openId}>${u.name}</at>`)
      .join(" ");
    elements.push({ tag: "hr" });
    elements.push({
      tag: "div",
      text: { tag: "lark_md", content: atText },
    });
  }

  // 时间戳
  elements.push({
    tag: "note",
    elements: [
      {
        tag: "plain_text",
        content: `📅 ${new Date().toLocaleString("zh-CN")}`,
      },
    ],
  });

  return {
    msg_type: "interactive",
    card: {
      header: {
        title: {
          tag: "plain_text",
          content: `${projectName || "前端"} 构建&部署`,
        },
        template: isSuccess ? "green" : "red",
      },
      elements,
    },
  };
}

/**
 * 统一处理部署成功，输出成功信息并退出
 * @param {object} options - 配置选项
 * @param {number} options.startTime - 开始时间戳
 * @param {string} [options.hostName] - 部署域名（GCS 部署时使用）
 * @param {string} [options.bucketName] - GCS Bucket 名称（GCS 部署时使用）
 * @param {string} [options.buildNumber] - 构建版本号（GCS 部署时使用）
 * @param {string} [options.imageTag] - Docker 镜像标签（Docker 部署时使用）
 * @param {string} [options.environment] - 部署环境（Docker 部署时使用）
 * @param {string} [options.projectName] - 项目名称（Docker 部署时使用）
 * @param {string} [options.feishuWebhook] - 飞书 Webhook URL（Docker 部署时可选）
 */
async function handleDeploySuccess(options) {
  const {
    startTime,
    hostName,
    bucketName,
    buildNumber,
    imageTag,
    environment,
    projectName,
    feishuWebhook = "https://open.feishu.cn/open-apis/bot/v2/hook/34c2771c-e8b9-4bcd-9e96-8df9e9eb7d53",
  } = options;

  // 计算耗时
  const duration = Math.floor((Date.now() - startTime) / 1000);
  const minutes = Math.floor(duration / 60);
  const seconds = duration % 60;

  // 判断部署类型
  const isDockerDeploy = !!imageTag;

  // 发送飞书卡片通知
  if (feishuWebhook) {
    const gitInfo = getGitInfo();
    const branchUrl = gitInfo.repoUrl
      ? `${gitInfo.repoUrl}/tree/${gitInfo.branch}`
      : "";

    // prod 环境 @ 张峥 和构建人
    const isProd = /^prod(uction)?$/i.test(environment || "");
    let atUsers;
    if (isProd) {
      const masterId = process.env.MASTER_OPEN_ID || "";
      const builderOpenId = process.env.BUILD_USER_OPEN_ID || "";
      const buildUser = process.env.BUILD_USER_NAME || "unknown";
      atUsers = [];
      if (masterId) atUsers.push({ openId: masterId, name: "张峥" });
      if (builderOpenId)
        atUsers.push({ openId: builderOpenId, name: buildUser });
    }

    const card = buildDeployCard({
      status: "success",
      projectName,
      branch: gitInfo.branch,
      branchUrl,
      commit: gitInfo.commit,
      commitAuthor: gitInfo.commitAuthor,
      commitMessage: gitInfo.commitMessage,
      environment,
      duration,
      imageTag,
      bucketName,
      atUsers,
    });
    await sendFeishuNotification(card, feishuWebhook);
  }

  // 输出成功信息
  log("success", "========================================");
  log(
    "success",
    isDockerDeploy
      ? "🎉 Build & Deploy Completed!"
      : "Build & Deploy Completed!",
  );
  log("success", "========================================");

  if (isDockerDeploy) {
    // Docker 部署信息
    log("info", `📦 镜像: ${imageTag}`);
    log("info", `🌍 环境: ${environment}`);
    log("info", `⏱️  总耗时: ${minutes}m ${seconds}s`);
  } else {
    // GCS 部署信息
    log("info", `URL: https://${hostName}/`);
    log("info", `Bucket: gs://${bucketName}/`);
    log("info", `Version: ${buildNumber}`);
    log("info", `总耗时: ${minutes}m ${seconds}s`);
    log("success", "========================================");
    log("success", `Open https://${hostName}/ in browser and enjoy!`);
  }

  log("success", "========================================");
  process.exit(0);
}

/**
 * 统一处理部署失败，输出错误信息并退出
 * @param {object} options - 配置选项
 * @param {number} options.startTime - 开始时间戳
 * @param {Error} options.error - 错误对象
 * @param {string} [options.environment] - 部署环境（Docker 部署时使用）
 * @param {string} [options.projectName] - 项目名称（Docker 部署时使用）
 * @param {string} [options.feishuWebhook] - 飞书 Webhook URL（Docker 部署时可选）
 */
async function handleDeployError(options) {
  const {
    startTime,
    error,
    environment,
    projectName,
    feishuWebhook = "https://open.feishu.cn/open-apis/bot/v2/hook/34c2771c-e8b9-4bcd-9e96-8df9e9eb7d53",
  } = options;

  // 计算耗时
  const duration = Math.floor((Date.now() - startTime) / 1000);
  const minutes = Math.floor(duration / 60);
  const seconds = duration % 60;

  // 判断是否为 Docker 部署
  const isDockerDeploy = !!(environment && projectName);

  // 输出错误信息
  log("error", "========================================");
  log(
    "error",
    isDockerDeploy ? "❌ Build & Deploy Failed!" : "Build & Deploy Failed!",
  );
  log("error", "========================================");
  log("error", error.message);
  if (error.stack) {
    console.error(error.stack);
  }
  log("error", "========================================");
  log("error", `失败耗时: ${minutes}m ${seconds}s`);

  // Docker 部署：发送飞书失败卡片通知
  if (isDockerDeploy && feishuWebhook) {
    const gitInfo = getGitInfo();
    const branchUrl = gitInfo.repoUrl
      ? `${gitInfo.repoUrl}/tree/${gitInfo.branch}`
      : "";

    // prod 环境 @ 张峥 和构建人
    const isProd = /^prod(uction)?$/i.test(environment || "");
    let atUsers;
    if (isProd) {
      const masterId = process.env.MASTER_OPEN_ID || "";
      const builderOpenId = process.env.BUILD_USER_OPEN_ID || "";
      const buildUser = process.env.BUILD_USER_NAME || "unknown";
      atUsers = [];
      if (masterId) atUsers.push({ openId: masterId, name: "张峥" });
      if (builderOpenId)
        atUsers.push({ openId: builderOpenId, name: buildUser });
    }

    const card = buildDeployCard({
      status: "error",
      projectName,
      branch: gitInfo.branch,
      branchUrl,
      commit: gitInfo.commit,
      commitAuthor: gitInfo.commitAuthor,
      commitMessage: gitInfo.commitMessage,
      environment,
      duration,
      errorMessage: error.message,
      atUsers,
    });
    await sendFeishuNotification(card, feishuWebhook);
  }

  process.exit(1);
}

module.exports = {
  sendFeishuNotification,
  getGitInfo,
  buildDeployCard,
  handleDeploySuccess,
  handleDeployError,
};
