# Button

```tsx
import { ReactNode } from "react";
import Spinner from "@/components/Spinner";

interface ButtonProps {
  onClick: () => void;
  disabled?: boolean;
  loading?: boolean;
  children: ReactNode;
  variant?: "gradient" | "primary" | "secondary" | "text";
  className?: string;
}

export default function Button({
  onClick,
  disabled = false,
  loading = false,
  children,
  variant = "gradient",
  className = "",
}: ButtonProps) {
  const variantStyles = {
    gradient:
      "h-12 rounded-[999px] bg-gradient-to-b from-[#5aa9ff] to-[#007afe] font-normal text-base text-white active:scale-[0.97] active:opacity-90 transition-all duration-150",
    primary:
      "h-14 rounded-[28px] bg-brand font-medium text-[15px] text-white active:scale-[0.97] active:opacity-80 transition-all duration-150",
    secondary:
      "h-14 rounded-[28px] bg-[#f2f2f7] font-medium text-[15px] text-black active:scale-[0.97] active:opacity-80 transition-all duration-150",
    text: "h-auto bg-transparent font-normal text-[15px] text-[rgba(60,60,67,0.6)] underline leading-5 tracking-[-0.23px] active:opacity-60 transition-opacity duration-150",
  };

  return (
    <button
      className={`flex items-center justify-center w-full border-none text-center ${variantStyles[variant]} ${
        disabled || loading ? "opacity-60" : ""
      } ${className}`}
      onClick={onClick}
      disabled={disabled || loading}
    >
      {loading ? <Spinner size={20} color="white" /> : children}
    </button>
  );
}

```
