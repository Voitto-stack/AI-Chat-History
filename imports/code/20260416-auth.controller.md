# auth.controller

```ts
import { type FastifyInstance } from "fastify";
import { authLogin, authLogout, authMe, authChangePassword } from "./auth.service.js";

export function authController(app: FastifyInstance) {
  app.post("/auth/login", { config: { rateLimit: { max: 5, timeWindow: 60_000 } } }, authLogin);
  app.post("/auth/logout", authLogout);
  app.post("/auth/me", authMe);
  app.post("/auth/change-password", authChangePassword);
}

```
