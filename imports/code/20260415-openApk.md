/**
 * APK 跳转与登录信息传递工具
 */

import { getUserStoreState } from "@/hooks/useUser";
import { STORAGE_KEYS } from "@/constants/storageKeys";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";

const APK_PACKAGE_NAME = "com.gracechat.prod";

/**
 * 将登录信息复制到剪贴板，供 APK 启动时读取
 * 格式: { userid, timsig, token, host }
 */
export function copyLoginInfoToClipboard(): void {
  const { userInfo, timUsersig } = getUserStoreState();
  const token = localStorage.getItem(STORAGE_KEYS.HAVEN_TOKEN) || "";
  const data = {
    userid: userInfo?.userId ?? "",
    timsig: timUsersig,
    token,
    host: `https://${window.location.host}`,
  };
  navigator.clipboard.writeText(JSON.stringify(data)).catch((err) => {
    console.error("[openApk] Failed to copy login info:", err);
  });
}

/**
 * 通过 Android Intent URL 打开 APK 并携带登录信息
 * Scheme: {packageName}.pwatool://main
 *
 * @returns 是否成功跳转（5 秒后页面失去焦点视为成功）
 */
export async function openInToolApp(): Promise<boolean> {
  const { userInfo, timUsersig } = getUserStoreState();
  const token = localStorage.getItem(STORAGE_KEYS.HAVEN_TOKEN) || "";
  const userId = userInfo?.userId ?? "";

  const href = `intent://main#Intent;scheme=${APK_PACKAGE_NAME}.pwatool;S.userId=${userId};S.timSig=${timUsersig};S.token=${token};S.host=https://${window.location.host};end`;
  window.location.href = href;

  // 埋点：打开 APK 点击
  bpTrack(EventName.pwa_open_in_apk_click);

  // 等待 5 秒判断是否成功跳转（页面失去焦点 = APK 已打开）
  await new Promise((r) => setTimeout(r, 5000));
  return !document.hasFocus();
}
