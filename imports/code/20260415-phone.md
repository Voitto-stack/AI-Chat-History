# Phone

```tsx
import { useEffect, useRef, useState } from "react";
import { useOnboarding } from "@/hooks/useOnboarding";
import { showConfirmDialog } from "@/components/ConfirmDialog";
import { useLockFn } from "@/hooks/useLockFn";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";
import Input from "./Input";
import Button from "@/components/Button";
import { formatUsPhone, US_PHONE_FORMATTED_LENGTH } from "@/utils/phone";

interface OnboardingPhoneProps {
  onContinue: () => void;
}

export default function OnboardingPhone({ onContinue }: OnboardingPhoneProps) {
  const { inputPhoneNumber, setInputPhoneNumber } = useOnboarding();
  const [value, setValue] = useState(inputPhoneNumber);
  const [error, setError] = useState<string | null>(
    inputPhoneNumber.length >= US_PHONE_FORMATTED_LENGTH ? null : "Please enter a valid phone number",
  );
  const inputRef = useRef<HTMLInputElement>(null);

  useEffect(() => {
    inputRef.current?.focus();
    bpTrack(EventName.pwa_onboarding_phone_page_show);
    bpTrack(EventName.pwa_conv_phone_page_show); // 转化埋点
  }, []);

  const onChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    const formattedValue = formatUsPhone(event.target.value);
    setValue(formattedValue);
    setError(formattedValue.length >= US_PHONE_FORMATTED_LENGTH ? null : "Please enter a valid phone number");
  };

  const onClick = useLockFn(async () => {
    if (error) return;

    // 埋点：点击 Next（传递 phone_number 参数，与 react-haven 保持一致）
    bpTrack(EventName.pwa_onboarding_phone_page_next_click, {
      phone_number: value,
    });
    bpTrack(EventName.pwa_conv_phone_page_clickButton, {
      phone_number: value,
    }); // 转化埋点

    // 埋点：确认弹窗展示
    bpTrack(EventName.pwa_onboarding_phone_confirm_dialog_show);
    bpTrack(EventName.pwa_onboarding_phone_confirm_modal_show);

    const confirmed = await showConfirmDialog({
      render: () => (
        <>
          <span className="mt-2 text-black font-bold text-[20px] leading-[26px] text-center">Confirm Phone Number</span>
          <span className="mt-2 text-[#8E8E93] text-[14px] leading-[20px] text-center">
            Cash-out needs a valid phone; can't be changed after confirmation.
          </span>
          <div className="mt-5 w-full">
            <div className="w-full h-[56px] flex items-center justify-center rounded-xl border border-[#C7C7CC] bg-[#F5F5F5] shadow-[0_1px_4px_rgba(0,0,0,0.06)]">
              <span className="text-black text-[22px] font-medium" style={{ fontFamily: "TTFellows, sans-serif" }}>
                +1&nbsp; {value}
              </span>
            </div>
          </div>
        </>
      ),
      buttonLayout: "vertical",
      confirmText: "Confirm",
      cancelText: "Edit",
    });

    if (confirmed) {
      // 埋点：点击 Confirm（传递 phone_number 参数，与 react-haven 保持一致）
      bpTrack(EventName.pwa_onboarding_phone_confirm_dialog_confirm, {
        phone_number: value,
      });
      bpTrack(EventName.pwa_onboarding_phone_confirm_modal_confirm_click, {
        phone_number: value,
      });
      bpTrack(EventName.pwa_conv_phone_pop_confirm, {
        phone_number: value,
      }); // 转化埋点

      // 埋点：手机号确认完成（不传参数，与 react-haven 保持一致）
      bpTrack(EventName.pwa_onboarding_phone_page_confirm);

      setInputPhoneNumber(value);
      onContinue();
    } else {
      // 埋点：点击 Edit
      bpTrack(EventName.pwa_onboarding_phone_confirm_dialog_edit);
      bpTrack(EventName.pwa_onboarding_phone_confirm_modal_edit_click);
      bpTrack(EventName.pwa_conv_phone_pop_cancel); // 转化埋点
    }
  });

  const isButtonDisabled = !!error;

  return (
    <>
      <Input title="Your Phone Number">
        <div className="flex items-center justify-center w-[80%] h-14 mt-10 p-0 border-0 border-b-[0.5px] border-solid border-b-[#e5e5ea] rounded-none bg-transparent overflow-hidden">
          <span className="text-black/30 font-medium text-[22px] leading-7 mr-2 flex-shrink-0">US+1</span>
          <input
            ref={inputRef}
            className="flex-1 min-w-0 h-full p-0 border-0 bg-transparent text-black font-medium text-[28px] leading-7 text-left focus:outline-none autofill:bg-transparent autofill:shadow-[inset_0_0_0_1000px_transparent]"
            type="text"
            inputMode="numeric"
            pattern="[0-9]*"
            value={value}
            onChange={onChange}
            autoComplete="off"
          />
        </div>
      </Input>

      <Button onClick={onClick} disabled={isButtonDisabled}>
        Next
      </Button>
    </>
  );
}
```
