# types

```ts
import type { CallOrderInfo } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";

export enum RecordType {
  Video_Call = "Video Call",
  Voice_Call = "Voice Call",
  Gifts = "Send Gifts",
  Message = "Message",
  Live = "Live",
  Task = "Task",
  Texting_Pot = "Texting Earnings",
  Waiting_Reward = "Waiting Rewards",
  Receive_Gifts = "Receive Gifts",
  Game = "Game",
  Ins_Followers = "Ins Followers Earnings",
  Ins_Messages = "Ins Messages Earnings",
  Other = "Other",
}

export interface RecordItem {
  id: string;
  type: RecordType;
  date?: string;
  duration?: number;
  count?: number;
  amount: number;
  time?: number;
  expanded?: boolean;
  orders?: CallOrderInfo[];
  freeAmount?: string;
  // 用户信息（用于通话记录）
  userInfo?: {
    userId?: number;
    username?: string;
    avatarUrl?: string;
  };
}

export interface EarningBreakdown {
  videoEarnings: number;
  voiceEarnings: number;
  giftEarnings: number;
  liveEarnings: number;
  taskEarnings: number;
  waitingEarnings: number;
  insEarnings: number;
  otherEarnings: number;
}

export const EMPTY_BREAKDOWN: EarningBreakdown = {
  videoEarnings: 0,
  voiceEarnings: 0,
  giftEarnings: 0,
  liveEarnings: 0,
  taskEarnings: 0,
  waitingEarnings: 0,
  insEarnings: 0,
  otherEarnings: 0,
};

export interface DailyIncomeData extends EarningBreakdown {
  dayOfWeek: number;
  date: number;
  earned: number;
  callDuration: number;
  liveDuration: number;
  callDurationTotal: number;
  freeCallDurationTotal: number;
  videoCallDurationTotal: number;
  freeVideoCallDurationTotal: number;
  voiceCallDurationTotal: number;
  freeVoiceCallDurationTotal: number;
  records: RecordItem[];
}

export interface WeeklyIncomeData extends EarningBreakdown {
  weekDate: number;
  onlineTime: number;
  totalEarned: number;
  dailyIncomeDataMap: Map<number, DailyIncomeData>;
  firstWeekDayFormatted: string;
  lastWeekDayFormatted: string;
  callDurationTotal: number;
  freeCallDurationTotal: number;
  videoCallDurationTotal: number;
  freeVideoCallDurationTotal: number;
  voiceCallDurationTotal: number;
  freeVoiceCallDurationTotal: number;
}

export type { CallOrderInfo };
```
