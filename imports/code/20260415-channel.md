# channel

// 渠道管理工具函数

let originalChannel: string | null = null;
let adChannel: string | null = null;
let ttclid: string | null = null;
let fbclid: string | null = null;

/**
 * 获取原始渠道（首次访问时设置，不可变）
 */
export function getOriginalChannel(): string {
  if (!originalChannel) {
    try {
      originalChannel = localStorage.getItem("original_channel") || "";
    } catch (e) {
      console.error("[Channel] Failed to get original_channel:", e);
      return "";
    }
  }
  return originalChannel;
}

/**
 * 设置原始渠道
 * @param channel 渠道标识 (tt, fb, google, etc.)
 */
export function setOriginalChannel(channel: string): void {
  if (!channel) {
    console.warn("[Channel] Empty channel provided, skipping");
    return;
  }

  try {
    originalChannel = channel;
    localStorage.setItem("original_channel", channel);
    localStorage.setItem("ad_channel", channel);

    // TikTok 渠道：保存 ttclid
    if (channel === "tt") {
      const searchParams = new URLSearchParams(window.location.search);
      const id = searchParams.get("ttclid");
      if (id) setTtclid(id);
    }

    // Facebook 渠道：保存 fbclid
    if (channel === "fb") {
      const searchParams = new URLSearchParams(window.location.search);
      const id = searchParams.get("fbclid");
      if (id) setFbclid(id);
    }

    console.log("[Channel] Set original_channel and ad_channel to:", channel);
  } catch (e) {
    console.error("[Channel] Failed to set original_channel:", e);
  }
}

/**
 * 获取广告渠道（可更新）
 */
export function getAdChannel(): string | null {
  if (!adChannel) {
    try {
      adChannel = localStorage.getItem("ad_channel");
    } catch (e) {
      console.error("[Channel] Failed to get ad_channel:", e);
      return null;
    }
  }
  return adChannel;
}

/**
 * 设置广告渠道
 */
export function setAdChannel(channel: string | null): void {
  adChannel = channel;
  try {
    if (channel) {
      localStorage.setItem("ad_channel", channel);
      console.log("[Channel] Set ad_channel to:", channel);
    } else {
      localStorage.removeItem("ad_channel");
      console.log("[Channel] Removed ad_channel");
    }
  } catch (e) {
    console.error("[Channel] Failed to set ad_channel:", e);
  }
}

/**
 * 获取 TikTok 点击 ID
 */
export function getTtclid(): string | null {
  if (!ttclid) {
    try {
      ttclid = localStorage.getItem("ttclid");
    } catch (e) {
      console.error("[Channel] Failed to get ttclid:", e);
      return null;
    }
  }
  return ttclid;
}

/**
 * 设置 TikTok 点击 ID
 */
export function setTtclid(id: string): void {
  ttclid = id;
  try {
    localStorage.setItem("ttclid", id);
    console.log("[Channel] Set ttclid to:", id);
  } catch (e) {
    console.error("[Channel] Failed to set ttclid:", e);
  }
}

/**
 * 获取 Facebook 点击 ID
 */
export function getFbclid(): string | null {
  if (!fbclid) {
    try {
      fbclid = localStorage.getItem("fbclid");
    } catch (e) {
      console.error("[Channel] Failed to get fbclid:", e);
      return null;
    }
  }
  return fbclid;
}

/**
 * 设置 Facebook 点击 ID
 */
export function setFbclid(id: string): void {
  fbclid = id;
  try {
    localStorage.setItem("fbclid", id);
    console.log("[Channel] Set fbclid to:", id);
  } catch (e) {
    console.error("[Channel] Failed to set fbclid:", e);
  }
}

/**
 * 清除所有渠道信息
 */
export function clearChannelInfo(): void {
  originalChannel = null;
  adChannel = null;
  ttclid = null;
  fbclid = null;

  try {
    localStorage.removeItem("original_channel");
    localStorage.removeItem("ad_channel");
    localStorage.removeItem("ttclid");
    localStorage.removeItem("fbclid");
    console.log("[Channel] All channel info cleared");
  } catch (e) {
    console.error("[Channel] Failed to clear channel info:", e);
  }
}

/**
 * 获取所有渠道参数（用于埋点）
 */
export function getChannelParams() {
  return {
    channel: getOriginalChannel(),
    adChannel: getAdChannel(),
    ttclid: getTtclid(),
    fbclid: getFbclid(),
  };
}
