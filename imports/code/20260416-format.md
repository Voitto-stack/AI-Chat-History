# format

```ts
/**
 * 通用格式化工具函数
 */

/**
 * 智能格式化数字，自动去除尾部无意义的0
 * @param num 数字
 * @param maxDecimals 最大保留小数位数
 * @returns 格式化后的字符串
 */
function formatNumberSmart(num: number, maxDecimals: number): string {
  // 如果是整数，直接返回
  if (num % 1 === 0) {
    return num.toString();
  }

  // 有小数，保留指定位数后去除尾部的0
  return num.toFixed(maxDecimals).replace(/\.?0+$/, "");
}

/**
 * 格式化数字
 * @param num 数字
 * @param decimals 保留小数位数，默认2位
 * @returns 格式化后的字符串，自动去除尾部无意义的0
 */
export function formatNumber(num: number, decimals: number = 2): string {
  return formatNumberSmart(num, decimals);
}

/**
 * 格式化奖励金额
 * @param reward 奖励金额字符串（可能带 $ 符号）
 * @returns 格式化后的数字字符串
 */
export function formatReward(reward: string): string {
  if (!reward) return "0";

  const rewardStr = reward.startsWith("$") ? reward.slice(1) : reward;
  const num = Number(rewardStr);

  if (isNaN(num)) return rewardStr;

  return formatNumberSmart(num, 1);
}

/**
 * 格式化时间（分钟）
 * @param mins 分钟数
 * @returns 格式化后的时间字符串，如 "5m" 或 "2.5m"
 */
export function formatTime(mins: number): string {
  return `${formatNumberSmart(mins, 1)}m`;
}

/**
 * 格式化金额
 * @param amount 金额数字
 * @returns 格式化后的金额字符串，如 "$10.00"
 */
export function formatAmount(amount: number): string {
  return `$${amount.toFixed(2)}`;
}

```
