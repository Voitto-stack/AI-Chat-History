# TabBar

```tsx
import { useNavigate, useLocation } from "react-router-dom";
import { useMemo } from "react";
import icTabHome from "@/assets/images/tab/ic_tab_home.webp";
import icTabHomeSelected from "@/assets/images/tab/ic_tab_home_selected.webp";
import icTabEarnings from "@/assets/images/tab/ic_tab_earnings.webp";
import icTabEarningsSelected from "@/assets/images/tab/ic_tab_earnings_selected.webp";
import icTabLive from "@/assets/images/tab/ic_tab_live.webp";
import icTabChat from "@/assets/images/tab/ic_tab_chat.webp";
import icTabChatSelected from "@/assets/images/tab/ic_tab_chat_selected.webp";
import icTabMe from "@/assets/images/tab/ic_tab_me.webp";
import icTabMeSelected from "@/assets/images/tab/ic_tab_me_selected.webp";
import { BadgeView } from "@/components/BadgeView";
import { useChatListStore } from "@/stores/chatListStore";
import { useMissCall } from "@/hooks/useMissCall";

interface TabItem {
  path: string;
  label: string;
  icon?: string;
  iconSelected?: string;
}

const TAB_STYLE = {
  fontSize: "12px",
  lineHeight: "16px",
  fontWeight: 400,
  letterSpacing: "-0.23px",
} as const;

const TABS: TabItem[] = [
  { path: "/", label: "Home", icon: icTabHome, iconSelected: icTabHomeSelected },
  { path: "/earnings", label: "Earnings", icon: icTabEarnings, iconSelected: icTabEarningsSelected },
  { path: "/live", label: "", icon: icTabLive, iconSelected: icTabLive },
  { path: "/chats", label: "Chats", icon: icTabChat, iconSelected: icTabChatSelected },
  { path: "/me", label: "Me", icon: icTabMe, iconSelected: icTabMeSelected },
];

const MID_INDEX = Math.floor(TABS.length / 2);

export default function TabBar() {
  const navigate = useNavigate();
  const location = useLocation();

  // 从已过滤的会话列表计算总未读数（含未接来电数量）
  const chatList = useChatListStore((s) => s.chatList);
  const { userTotal: missCallTotal } = useMissCall();
  const chatUnreadCount = useMemo(
    () => chatList.reduce((sum, c) => sum + (c.unreadCount || 0), 0) + missCallTotal,
    [chatList, missCallTotal],
  );

  const isActive = (path: string) => (path === "/" ? location.pathname === "/" : location.pathname.startsWith(path));

  return (
    <nav
      className="flex items-start justify-around w-full mx-auto bg-white z-50 pb-[var(--sab,0px)]"
      style={{ height: "84px", boxShadow: "0px -20px 25.6px rgba(0, 0, 0, 0.02)" }}
    >
      {TABS.map((tab, index) => {
        const active = isActive(tab.path);
        const isLive = index === MID_INDEX;
        const isChats = tab.path === "/chats";
        const iconSrc = active ? tab.iconSelected : tab.icon;

        return (
          <button
            key={tab.path}
            onClick={() => navigate(tab.path)}
            className="relative flex flex-col items-center justify-center w-full gap-1 press-feedback"
            style={{ padding: "8px 6px" }}
          >
            {isLive ? (
              <img
                src={iconSrc}
                alt="live"
                className="rounded-full bg-white"
                style={{ width: "50px", height: "50px", marginTop: "-20px", padding: "8px", boxSizing: "content-box" }}
              />
            ) : (
              <img src={iconSrc} alt={tab.label} className="rounded-full" style={{ width: "24px", height: "24px" }} />
            )}
            {isChats && (
              <BadgeView unreadCount={chatUnreadCount} size="sm" className="absolute top-1 right-[calc(50%-16px)]" />
            )}
            {!isLive && (
              <span style={{ color: active ? "#47AEEF" : "rgba(60, 60, 67, 0.6)", textAlign: "center", ...TAB_STYLE }}>
                {tab.label}
              </span>
            )}
          </button>
        );
      })}
    </nav>
  );
}
```
