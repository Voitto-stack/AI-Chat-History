```tsx
import { useState, useRef, useCallback, useEffect } from "react";
import { useOnboarding } from "@/hooks/useOnboarding";
import { useUser } from "@/hooks/useUser";
import { useLoading } from "@/hooks/useLoading";
import { useLockFn } from "@/hooks/useLockFn";
import { faceScore } from "@/http/api";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";
import Button from "@/components/Button";
import { toast } from "@/utils/toast";
import { compressImage, readFileAsDataUrl } from "@/utils/fileUtils";
import CropPhotoModal from "@/components/CropPhotoModal";
import PhotoChangeModal from "@/components/PhotoChangeModal";

interface OnboardingPhotoProps {
  onContinue: () => void;
}

const MAX_AGE = 46;

/**
 * OnboardingPhoto - 上传头像页面（单张照片）
 * 支持从相册选择或拍照，包含裁剪和颜值校验
 */
export default function OnboardingPhoto({ onContinue }: OnboardingPhotoProps) {
  const { avatarUri, setAvatarUri, inputAge } = useOnboarding();
  const { userInfo, updateAvatar } = useUser();
  const { showLoading, hideLoading } = useLoading();
  const [imageSrc, setImageSrc] = useState<string>(avatarUri);
  const [imageFile, setImageFile] = useState<File>();
  const [showCropPhoto, setShowCropPhoto] = useState(false);
  const [imgSize, setImageSize] = useState({ width: 0, height: 0 });
  const [showModal, setShowModal] = useState(false);
  const fileInputRef = useRef<HTMLInputElement>(null);
  const imgRef = useRef<HTMLImageElement>(null);

  const hasPhoto = !!imageFile || !!avatarUri;

  // 页面展示埋点
  useEffect(() => {
    bpTrack(EventName.pwa_onboarding_photo_page_show);
    bpTrack(EventName.pwa_conv_upload_photo_show); // 转化埋点
  }, []);

  /** 处理图片选择 */
  const onChange = async (event: React.ChangeEvent<HTMLInputElement>) => {
    const fileList = event.target.files;
    if (!fileList?.length) {
      event.target.value = "";
      return;
    }

    const file = fileList[0];

    // 埋点：选择照片
    bpTrack(EventName.pwa_reg_photo_select, {
      selected_count: 1,
      is_cancel: false,
    });

    // 埋点：选择照片成功
    bpTrack(EventName.pwa_onboarding_photo_page_select_photo_success, {
      photo_count: 1,
      is_camera: event.target.hasAttribute("capture"),
      result: "success",
    });

    try {
      const url = await readFileAsDataUrl(file);

      // 加载图片尺寸
      const img = new Image();
      img.onload = () => {
        setImageSize({ width: img.naturalWidth, height: img.naturalHeight });
      };
      img.src = url;

      setImageSrc(url);
      setImageFile(file);
      setShowCropPhoto(true);
    } catch (error) {
      toast.error("Failed to load image, please try again");
    } finally {
      event.target.value = "";
    }
  };

  /** 触发文件选择 */
  const triggerFileInput = (capture: boolean) => {
    // 埋点：打开相册（不传参数，与 react-haven 保持一致）
    bpTrack(EventName.pwa_reg_photo_open_album);

    // 转化埋点：区分相机和相册
    if (capture) {
      bpTrack(EventName.pwa_conv_upload_photo_camera_click);
    } else {
      bpTrack(EventName.pwa_conv_upload_photo_album_click);
    }

    const input = fileInputRef.current;
    if (!input) return;

    if (capture) {
      input.setAttribute("capture", "environment");
    } else {
      input.removeAttribute("capture");
    }
    input.click();
    setShowModal(false);
  };

  /** 图片加载完成后获取尺寸 */
  const handleImageLoad = () => {
    if (imgRef.current) {
      setImageSize({
        width: imgRef.current.naturalWidth,
        height: imgRef.current.naturalHeight,
      });
    }
  };

  /** 裁剪完成回调 */
  const onCropPhoto = useCallback(({ imageUrl, imageFile: croppedFile }: { imageUrl: string; imageFile: File }) => {
    setImageSrc(imageUrl);
    setImageFile(croppedFile);
    setShowCropPhoto(false);
  }, []);

  /** 点击继续按钮 */
  const onClick = useLockFn(async () => {
    if (!hasPhoto) {
      toast.error("Please upload a clear photo to continue");
      return;
    }

    // 已缓存头像且未重新选择，直接跳过上传
    if (!imageFile && avatarUri) {
      onContinue();
      return;
    }

    if (!imageFile) {
      toast.error("Please select a photo");
      return;
    }

    if (!userInfo?.userId) {
      toast.error("Please sign in to continue");
      return;
    }

    // 检查年龄是否符合要求
    if (!inputAge || inputAge > MAX_AGE) {
      toast.error("Photo doesn't meet requirements. Please upload a clear, well-lit photo of your face.");
      return;
    }

    // 埋点：点击保存照片
    bpTrack(EventName.pwa_reg_photo_save_success_click, {
      photo_count: 1,
      avatar_index: 0,
    });
    bpTrack(EventName.pwa_reg_photo_click_next);

    showLoading();
    try {
      // 为颜值打分创建超小尺寸版本（降低 base64 长度）
      const faceScoreFile = await compressImage(imageFile, 0.05, 300);
      const faceScoreBase64 = await readFileAsDataUrl(faceScoreFile);

      // 颜值打分（先校验）
      const faceScoreResponse = await faceScore({
        userId: userInfo.userId,
        imageUrl: faceScoreBase64,
      });

      const ageQualified = inputAge && inputAge <= MAX_AGE;
      const appearanceQualified =
        faceScoreResponse.beautyScore && (faceScoreResponse.beautyScore.appearance ?? 0) >= 60;

      // 埋点：审核结果（统一事件）
      bpTrack(EventName.pwa_reg_photo_audit_result, {
        beauty_score: faceScoreResponse.beautyScore?.appearance,
        age_qualified: ageQualified,
        pass: appearanceQualified && ageQualified,
      });

      if (!appearanceQualified) {
        toast.error("Photo doesn't meet requirements. Please upload a clear, well-lit photo of your face.");
        hideLoading();
        return;
      }

      // 压缩头像用于上传（2MB，最大宽高 1200px）
      const processedFile = await compressImage(imageFile, 2, 1200);

      // 颜值校验通过后才上传头像（自动刷新用户信息）
      try {
        await updateAvatar({
          userId: userInfo.userId,
          avatarFile: processedFile,
        });

        // 埋点：服务器上传成功
        bpTrack(EventName.pwa_onboarding_upload_photo_page_server_success);
        bpTrack(EventName.pwa_conv_upload_photo_success); // 转化埋点
      } catch (uploadError) {
        // 埋点：服务器上传失败
        bpTrack(EventName.pwa_onboarding_upload_photo_page_server_failure, {
          success: false,
          message: uploadError instanceof Error ? uploadError.message : "unknown",
        });
        throw uploadError;
      }

      // 埋点：照片确认（审核通过后）
      bpTrack(EventName.pwa_onboarding_photo_page_confirm);
      bpTrack(EventName.pwa_conv_upload_photo_clickButton); // 转化埋点

      // 保存 avatarUri
      setAvatarUri(imageSrc);
      hideLoading();
      onContinue();
    } catch (error) {
      console.error("Upload avatar failed:", error);
      const errorMessage =
        error instanceof Error ? error.message : "Image file type not supported, please select another one";

      toast.error(errorMessage);
      hideLoading();
    }
  });

  return (
    <>
      <div className="flex-1 flex flex-col items-center mt-3  gap-4 min-h-0 overflow-hidden">
        <h2 className="text-black font-medium text-2xl text-center shrink-0">Upload Profile Photos</h2>

        <p className="text-[#9e9e9e] font-normal text-base tracking-[-0.23px] text-center shrink-0">
          Choose 1 clear face photos - let your personality shine!
        </p>

        {/* 单张照片上传区域 */}
        <div
          className="mt-4 flex items-center justify-center w-4/5 aspect-[3/4] overflow-hidden border border-dashed border-[#87ceeb] rounded-2xl cursor-pointer"
          onClick={() => setShowModal(true)}
        >
          {hasPhoto ? (
            <img
              src={imageSrc}
              alt="Uploaded photo"
              className="w-full h-full object-cover"
              onLoad={handleImageLoad}
              ref={imgRef}
            />
          ) : (
            <div className="flex flex-col items-center justify-center gap-2.5">
              <svg width="18" height="18" viewBox="0 0 18 18" fill="none">
                <path d="M9 1V17" stroke="#48B6FF" strokeWidth="2" strokeLinecap="round" />
                <path d="M1 9H17" stroke="#48B6FF" strokeWidth="2" strokeLinecap="round" />
              </svg>
              <span className="text-[#48b6ff] font-normal text-base  text-center">Add photo</span>
            </div>
          )}
        </div>

        <input ref={fileInputRef} type="file" accept="image/*" className="hidden" onChange={onChange} />
      </div>
      <Button onClick={onClick} disabled={!hasPhoto} variant="primary">
        Claim 0.3 $
      </Button>

      {/* 图片选择弹窗（含照片要求示例） */}
      <PhotoChangeModal
        open={showModal}
        onClose={() => setShowModal(false)}
        onChooseFromAlbum={() => triggerFileInput(false)}
        onTakePhoto={() => triggerFileInput(true)}
      />

      {/* 裁剪图片弹窗 */}
      {imageFile && (
        <CropPhotoModal
          open={showCropPhoto}
          onClose={() => setShowCropPhoto(false)}
          avatarUri={imageSrc}
          avatarFile={imageFile}
          onCropPhoto={onCropPhoto}
          initImgSize={imgSize}
          images={[{ imageUrl: imageSrc, imageFile }]}
          initialIndex={0}
        />
      )}
    </>
  );
}
```
