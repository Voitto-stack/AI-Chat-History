# saved-sql.model

```ts
import { getPgDb } from "@heyhru/server-plugin-pg";
import * as Q from "./saved-sql.sql.js";

export function listSavedSqlsByUser(userId: string) {
  return getPgDb().query(Q.LIST_BY_USER, [userId]);
}

interface CreateSavedSqlOpts {
  userId: string;
  name: string;
  sqlText: string;
  dataSourceId: string | null;
  database: string | null;
}

export function createSavedSqlRow(opts: CreateSavedSqlOpts) {
  return getPgDb().queryOne(Q.CREATE, [
    opts.userId,
    opts.name,
    opts.sqlText,
    opts.dataSourceId,
    opts.database,
  ]);
}

export async function updateSavedSqlRow(
  id: string,
  userId: string,
  data: { name?: string; sqlText?: string }
) {
  const fields: string[] = [];
  const values: unknown[] = [];
  if (data.name !== undefined) {
    fields.push("name = ?");
    values.push(data.name);
  }
  if (data.sqlText !== undefined) {
    fields.push("sql_text = ?");
    values.push(data.sqlText);
  }
  if (!fields.length) throw new Error("No fields to update");
  fields.push("updated_at = NOW()");
  values.push(id, userId);
  return getPgDb().queryOne(
    `UPDATE saved_sqls SET ${fields.join(", ")} WHERE id = ? AND user_id = ? RETURNING id, name, sql_text, data_source_id, database, created_at, updated_at`,
    values
  );
}

export function deleteSavedSqlRow(id: string, userId: string) {
  return getPgDb().run(Q.DELETE, [id, userId]);
}
```
