# index.test

```ts
import { describe, it, expect } from "vitest";
import { createLogger } from "./index";

describe("createLogger", () => {
  it("should create a logger with default options", () => {
    const logger = createLogger();
    expect(logger).toBeDefined();
    expect(logger.info).toBeTypeOf("function");
    expect(logger.error).toBeTypeOf("function");
    expect(logger.warn).toBeTypeOf("function");
    expect(logger.debug).toBeTypeOf("function");
    expect(logger.trace).toBeTypeOf("function");
    expect(logger.fatal).toBeTypeOf("function");
  });

  it("should create a logger with a name", () => {
    const logger = createLogger({ name: "test" });
    expect(logger).toBeDefined();
  });

  it("should respect custom log level", () => {
    const logger = createLogger({ level: "warn" });
    expect(logger.level).toBe("warn");
  });

  it("should support child loggers", () => {
    const logger = createLogger({ name: "parent" });
    const child = logger.child({ requestId: "abc-123" });
    expect(child).toBeDefined();
    expect(child.info).toBeTypeOf("function");
  });
});
```
