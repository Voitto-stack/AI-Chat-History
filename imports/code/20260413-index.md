// ==================== 滚动数字 Hook ====================

const useRollingNumber = (from: number, to: number, play: boolean, duration: number = 1200) => {
  const [display, setDisplay] = useState(from);

  useEffect(() => {
    if (!play) {
      setDisplay(from);
      return;
    }
    if (from === to) return;

    const startTime = performance.now();
    let raf: number;

    const tick = (now: number) => {
      const progress = Math.min((now - startTime) / duration, 1);
      const ease = 1 - Math.pow(1 - progress, 3);
      setDisplay(Math.round(from + (to - from) * ease));
      if (progress < 1) {
        raf = requestAnimationFrame(tick);
      }
    };
    raf = requestAnimationFrame(tick);
    return () => cancelAnimationFrame(raf);
  }, [from, to, play, duration]);

  return display;
};

const LockedBonusCard = ({
  dashboard,
  playAnimation,
  rewardDelta,
  moneyAnimationData,
  onAnimationEnd,
  onCashOut,
}: {
  dashboard: BffDashboardType;
  playAnimation: boolean;
  rewardDelta: number;
  moneyAnimationData: Record<string, unknown> | null;
  onAnimationEnd: () => void;
  onCashOut: () => void;
}) => {
  const totalPotential = dashboard.squadList.reduce(
    (sum, m) => sum + (m.stageProgress?.reduce((s, p) => s + p.rewardAmount, 0) ?? 0),
    0,
  );

  const lockedRef = useRef<HTMLDivElement>(null);
  const availableRef = useRef<HTMLDivElement>(null);
  const cardRef = useRef<HTMLDivElement>(null);
  const [flyStyle, setFlyStyle] = useState<React.CSSProperties | null>(null);

  const [availablePlay, setAvailablePlay] = useState(false);

  const displayLocked = useRollingNumber(
    dashboard.lockedAmount + rewardDelta,
    dashboard.lockedAmount,
    playAnimation,
    1000,
  );
  const displayAvailable = useRollingNumber(
    dashboard.availableAmount - rewardDelta,
    dashboard.availableAmount,
    availablePlay,
    600,
  );

  useEffect(() => {
    if (!playAnimation || rewardDelta <= 0) return;

    // Locked 减少 + 飞钱同时启动
    requestAnimationFrame(() => {
      if (lockedRef.current && availableRef.current && cardRef.current) {
        const cardRect = cardRef.current.getBoundingClientRect();
        const fromRect = lockedRef.current.getBoundingClientRect();
        const toRect = availableRef.current.getBoundingClientRect();

        setFlyStyle({
          "--start-x": `${fromRect.left - cardRect.left + fromRect.width / 2}px`,
          "--start-y": `${fromRect.top - cardRect.top + fromRect.height / 2}px`,
          "--end-x": `${toRect.left - cardRect.left + toRect.width / 2}px`,
          "--end-y": `${toRect.top - cardRect.top + toRect.height / 2}px`,
        } as React.CSSProperties);
      }
    });

    // 飞钱飞到一半时，Available 开始增加
    const t1 = setTimeout(() => {
      setAvailablePlay(true);
    }, 700);

    // 全部结束
    const t2 = setTimeout(() => {
      setFlyStyle(null);
      onAnimationEnd();
    }, 1800);

    return () => {
      clearTimeout(t1);
      clearTimeout(t2);
    };
  }, [playAnimation, rewardDelta, onAnimationEnd]);

  return (
    <div ref={cardRef} className="mx-4 bg-[#f4fafe] rounded-[16px] p-4 relative overflow-hidden">
      {/* 上半部分：Locked 金额 + 钱包插画 */}
      <div className="flex justify-between items-start">
        <div>
          <p className="text-[13px] text-[#16613F] leading-[18px] tracking-[-0.23px] font-['Saans_TRIAL']">
            Locked Bonus
          </p>
          <div ref={lockedRef} className="flex items-center mt-1">
            <span className="text-[40px] font-extrabold text-[#16613F] font-[Pangram]">${displayLocked}</span>
            <img src={icLockWarning} alt="" className="w-10 h-10 ml-1.5" />
          </div>
        </div>
        <img src={icWallet} alt="" className="w-[89px] h-[89px] object-contain -mt-1" />
      </div>

      {/* 虚线分隔 */}
      <div className="border-t border-dashed border-gray-200 my-3" />

      <p className="text-[13px] text-[#16613F] leading-[18px] tracking-[-0.23px] mb-3 font-['Saans_TRIAL']">
        History always presents ${dashboard.historicalWithdrawAmount ?? 0}
      </p>

      {/* 下半部分：Available + Cash Out */}
      <div className="flex items-end justify-between bg-[#F3F3F3] rounded-[12px] p-3">
        <div ref={availableRef} className="font-['Saans_TRIAL']">
          <p className="text-[12px] leading-[16px] text-[#16613F80]">Available</p>
          <p className="text-[22px] font-medium text-[#16613F] leading-none mt-0.5">${displayAvailable}</p>
        </div>
        <button
          className="h-[34px] px-5 bg-[#399C44] rounded-full text-white text-[14px] font-semibold font-['Saans_TRIAL']"
          onClick={onCashOut}
        >
          Cash Out
        </button>
      </div>

      {/* 飞钱 Lottie 从 Locked → Available */}
      {flyStyle && moneyAnimationData && (
        <>
          <div
            className="absolute pointer-events-none z-10"
            style={{
              left: `var(--start-x)`,
              top: `var(--start-y)`,
              ...flyStyle,
              animation: `flyMoney 1.3s ease-in-out forwards`,
            }}
          >
            <Lottie
              animationData={moneyAnimationData}
              autoplay={true}
              loop={false}
              style={{ width: 80, height: 80, marginLeft: -40, marginTop: -40 }}
            />
          </div>
          <style>{`
            @keyframes flyMoney {
              0% {
                transform: translate(0, 0) scale(1.2);
                opacity: 1;
              }
              50% {
                transform: translate(
                  calc((var(--end-x) - var(--start-x)) * 0.5),
                  calc((var(--end-y) - var(--start-y)) * 0.5)
                );
                opacity: 1;
              }
              100% {
                transform: translate(
                  calc(var(--end-x) - var(--start-x)),
                  calc(var(--end-y) - var(--start-y))
                );
                opacity: 0;
              }
            }
          `}</style>
        </>
      )}
    </div>
  );
};

// ==================== Squad 成员卡片 ====================

const SquadMemberCard = ({
  member,
  inviteUrl,
  bgImage,
  btnColor,
  stageColor,
  secondStageCircleColor,
}: {
  member: SquadMember;
  inviteUrl: string;
  bgImage: string;
  btnColor: string;
  stageColor: string;
  secondStageCircleColor: string;
}) => {
  // 根据当前阶段获取文案配置
  const stageCopy = STAGE_COPY_CONFIG[member.currentStage as "W2" | "W3" | "W4" | "W5"];

  // 根据 currentStage 动态生成 stageProgress 数组
  const stageRewardMap: Record<BffStage, number> = {
    W1: 0,
    W2: 0,
    W3: 5,
    W4: 15,
    W5: 30,
    DONE: 0,
  };

  const generateStageProgress = (): StageProgressItem[] => {
    const stages: BffStage[] = ["W1", "W2", "W3", "W4", "W5"];
    const currentStageIndex = member.currentStage === "DONE" ? stages.length : stages.indexOf(member.currentStage as BffStage);

    return stages.map((stage, index) => ({
      stage,
      completed: index < currentStageIndex,
      rewardAmount: stageRewardMap[stage] ?? 0,
      rewardType: stage === "W2" ? "BOOST" : "CASH",
    }));
  };

  const stageProgressList = member.stageProgress?.length
    ? member.stageProgress.map((item) => ({
        ...item,
        rewardAmount: item.rewardAmount ?? stageRewardMap[item.stage] ?? 0,
      }))
    : generateStageProgress();

  // 如果当前阶段没有配置（如 W1 或 DONE），使用后端返回的文案作为降级
  const taskText = stageCopy?.taskText || member.taskText;
  const btnText = stageCopy?.btnText || member.mainBtnText?.replace(/^👉\s*/, "") || "Poke Her";
  const shareText = stageCopy?.shareText || member.shareText;

  // 根据后端 activeStatus 判断状态
  const getActivityStatus = () => {
    const status = member.activeStatus || "active";
    const copy = member.activeStatusCopy || "";

    switch (status) {
      case "very_inactive":
        return { show: true, icon: icStatusLost, text: copy || "Lost > 7d", color: "#DF2727" };
      case "inactive":
        return { show: true, icon: icStatusUrgent, text: copy || "Inactive", color: "#DF2727" };
      case "idle":
        return { show: true, icon: icStatusInactive, text: copy || "Idle", color: "#FB881C" };
      case "active":
      default:
        return { show: true, icon: icStatusActive, text: copy || "Active", color: "#3EB828" };
    }
  };

  const activityStatus = getActivityStatus();

  // 处理按钮点击 - 触发分享
  const handleButtonClick = async () => {
    await shareBffLink({
      text: shareText,
      url: inviteUrl,
    });
    bpTrack(EventName.bff_poke_sent, {
      relationId: member.relationId,
      stage: member.currentStage,
    });
  };

  return (
    <div
      className="rounded-[20px] overflow-hidden relative h-[190px]"
      style={{
        backgroundImage: `url(${bgImage})`,
        backgroundSize: "cover",
        backgroundPosition: "center",
      }}
    >
      {/* 内容区 */}
      <div className="absolute left-4 right-4 top-4 flex flex-col gap-[15px]">
        {/* 顶部：头像 + 名字/任务 + Active 状态 */}
