  };

  return await httpClient.requestPost2(UploadUserInfoRequest, request, UploadUserInfoResponse);
}

// 更新用户头像
export async function updateAvatar(params: { userId: number; avatarFile: File }): Promise<UpdateAvatarResponse> {
  const arrayBuffer = await params.avatarFile.arrayBuffer();
  const fileUnit8Array = new Uint8Array(arrayBuffer);

  // 生成标准化的文件名: avatar_<MD5(userId)>.<扩展名>
  const extendName = getFileExtension(params.avatarFile);
  const fileName = `avatar_${MD5(params.userId.toString())}${extendName}`;

  const uploadPic: UploadPicture = {
    file: fileUnit8Array,
    name: fileName,
  };

  const request: UpdateAvatarRequest = {
    userId: params.userId,
    uploadAvatar: uploadPic,
  };

  return await httpClient.requestPost2(UpdateAvatarRequest, request, UpdateAvatarResponse);
}
