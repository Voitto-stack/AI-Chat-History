# main

```tsx
import { createRoot } from "react-dom/client";
import { registerSW } from "virtual:pwa-register";
import { initPwaInstall } from "@/services/pwaInstall";
import { initTracking } from "@/tracking";
import { initConsoleControl } from "@/utils/consoleControl";
import { redirectIfChina } from "@/utils/regionGuard";
import "@/utils/globalBridge"; // APK WebView 登录态注入（window.reloadUser）
import { initBridge } from "@/utils/bridge";
import App from "./App.tsx";

// 🔥 风险缓解：Sentry 加载前的早期错误缓存
interface EarlyError {
  message: string;
  stack?: string;
  timestamp: number;
}
const earlyErrors: EarlyError[] = [];
const originalErrorHandler = window.onerror;

// 拦截早期错误（Sentry 加载前）
window.onerror = function (message, source, lineno, colno, error) {
  earlyErrors.push({
    message: typeof message === "string" ? message : String(message),
    stack: error?.stack,
    timestamp: Date.now(),
  });

  // 保留原有的错误处理
  if (originalErrorHandler) {
    return originalErrorHandler.call(this, message, source, lineno, colno, error);
  }
  return false;
};

// 运行时 console 控制（生产环境默认静默，debug 页可开启）
initConsoleControl();

// 中文语言/中国时区 → 跳转官网（仅生产环境），跳转后不再初始化
if (!redirectIfChina()) {
  initTracking();

  // APK Bridge 初始化（注册 notifyKeyboardChanged 等 native→JS handler）
  initBridge();

  const updateSW = registerSW({
    onNeedRefresh() {
      updateSW(true);
    },
    onOfflineReady() {
      console.log("PWA: 离线就绪");
    },
    onRegisteredSW(swUrl, registration) {
      console.log("PWA: Service Worker 已注册:", swUrl);
      if (registration) {
        setInterval(() => registration.update(), 30 * 60 * 1000);
      }
    },
    onRegisterError(error) {
      console.error("PWA: Service Worker 注册失败:", error);
    },
  });

  initPwaInstall();

  // vConsole：读取 localStorage 开关，dev 默认开启，prod 默认关闭
  const vcFlag = localStorage.getItem("debug_vconsole");
  const enableVConsole = vcFlag !== null ? vcFlag === "1" : import.meta.env.DEV;
  if (enableVConsole) {
    import("vconsole").then(({ default: VConsole }) => {
      new VConsole();
    });
  }

  // 先渲染 UI
  createRoot(document.getElementById("root")!).render(<App />);

  // 🔥 优化：Sentry 延迟初始化（首屏渲染后异步加载，省 ~200KB）
  const loadSentry = () => {
    import("@/sentry")
      .then(() => {
        // Sentry 加载完成，恢复原有错误处理
        window.onerror = originalErrorHandler;

        // 🔥 风险缓解：补报早期错误
        if (earlyErrors.length > 0) {
          console.log(`[Sentry] Reporting ${earlyErrors.length} early errors`);
          // 动态导入 Sentry 以访问 captureException
          import("@sentry/react").then((Sentry) => {
            earlyErrors.forEach((err) => {
              Sentry.captureException(new Error(err.message), {
                contexts: {
                  early_error: {
                    original_stack: err.stack,
                    captured_at: err.timestamp,
                    reported_at: Date.now(),
                  },
                },
              });
            });
            earlyErrors.length = 0; // 清空缓存
          });
        }
      })
      .catch((err) => {
        console.error("Failed to load Sentry:", err);
        window.onerror = originalErrorHandler; // 恢复原有处理
      });
  };

  if ("requestIdleCallback" in window) {
    requestIdleCallback(loadSentry);
  } else {
    setTimeout(loadSentry, 1000);
  }
}
```
