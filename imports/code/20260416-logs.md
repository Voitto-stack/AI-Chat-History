# logs

```ts
import { client } from "./client";

export interface AuditLog {
  id: string;
  user_id: string;
  username: string;
  data_source_id: string | null;
  action: string;
  sql_text: string | null;
  result_summary: string | null;
  ip_address: string | null;
  created_at: string;
}

export interface LogFilters {
  userId?: string;
  dataSourceId?: string;
  limit?: number;
  offset?: number;
}

export const listLogs = (filters?: LogFilters) => client.post<AuditLog[]>("/logs/list", filters);

```
