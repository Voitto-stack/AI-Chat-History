/**
 * 帖子相关 API
 */

import {
  CreatePostRequest,
  CreatePostResponse,
  GetPostsByUserIdRequest,
  GetPostsByUserIdResponse,
  PostPicture,
  PostSuggestedContentRequest,
  PostSuggestedContentResponse,
} from "@heyhru/business-pwa-proto/gen/archat_api/post_api";
import httpClient from "./httpClient";

/** 获取用户帖子列表（cursor-based 分页） */
export async function getPostsByUserId(params: {
  userId: number;
  cursor?: number;
  limit?: number;
}): Promise<GetPostsByUserIdResponse> {
  const request: GetPostsByUserIdRequest = {
    userId: params.userId,
    cursor: params.cursor ?? 0,
    limit: params.limit,
  };

  return await httpClient.requestPost2(GetPostsByUserIdRequest, request, GetPostsByUserIdResponse);
}

/** 创建帖子 */
export async function createPost(params: { content?: string; pictures: PostPicture[] }): Promise<CreatePostResponse> {
  const request: CreatePostRequest = {
    clientToken: crypto.randomUUID(),
    content: params.content || "",
    pictures: params.pictures,
  };

  return await httpClient.requestPost2(CreatePostRequest, request, CreatePostResponse);
}

/** 发帖图片自动文案生成 */
export async function postSuggestedContent(params: {
  pictures: PostPicture[];
  timezoneOffsetMinutes: number;
}): Promise<PostSuggestedContentResponse> {
  const request: PostSuggestedContentRequest = {
    pictures: params.pictures,
    timezoneOffsetMinutes: params.timezoneOffsetMinutes,
  };

  return await httpClient.requestPost2(PostSuggestedContentRequest, request, PostSuggestedContentResponse);
}
