# bff

```ts
/**
 * BFF 裂变功能 API 服务
 */

import httpClient from "@/http/httpClient";
import {
  BffGetDashboardRequest,
  BffGetDashboardResponse,
  BffGetOverviewRequest,
  BffGetOverviewResponse,
} from "@heyhru/business-pwa-proto/gen/archat_api/user_api";
import type { BffSquadMember } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";
import type { BffOverview, BffDashboard, SquadMember, BffRiskFlag } from "@/types/bff";
import { numberToBffStage } from "@/utils/bffCalculations";

const getOverviewDollars = (res: BffGetOverviewResponse, key: "lockedAmountDollars" | "availableAmountDollars") => {
  return (res as BffGetOverviewResponse & Partial<Record<typeof key, number>>)[key] ?? 0;
};

const getDashboardDollars = (
  res: BffGetDashboardResponse,
  key: "lockedAmountDollars" | "availableAmountDollars" | "historicalWithdrawAmountDollars" | "buttonText",
) => {
  return (res as BffGetDashboardResponse & Partial<Record<typeof key, number | string>>)[key];
};

const isActiveStatus = (value: string | undefined): value is NonNullable<SquadMember["activeStatus"]> =>
  ["active", "idle", "inactive", "very_inactive"].includes(value ?? "");

const mapDashboardMember = (member: BffSquadMember, index: number): SquadMember => ({
  relationId: String(member.relationshipId ?? `member-${index}`),
  nickname: member.nickname || "",
  avatar: member.avatar || "",
  currentStage: numberToBffStage(member.currentStage),
  stageProgress: [],
  inactiveHours: member.lastActiveTime
    ? Math.max(0, Math.floor((Date.now() / 1000 - member.lastActiveTime) / 3600))
    : 0,
  pokeAvailable: true,
  riskFlag: (member.riskStatus as BffRiskFlag) || "NONE",
  taskText: "",
  mainBtnText: "Poke Her",
  stageReminderCopy: member.stageReminderCopy || "",
  activeStatus: isActiveStatus(member.activeStatus) ? member.activeStatus : "active",
  activeStatusCopy: member.activeStatusCopy || "",
});

/**
 * 获取 BFF Overview 数据
 * POST /bffApi/getOverview, protoId: 4904/4905
 */
export const getBffOverview = async (): Promise<BffOverview> => {
  try {
    const res = await httpClient.requestPost2(BffGetOverviewRequest, {}, BffGetOverviewResponse);
    console.log("[BFF API] getOverview response:", res);
    // 适配 proto 响应到前端类型
    // 注意: Overview 接口不包含 locked/available 金额，这些数据从 Dashboard 接口获取
    return {
      lockedAmount: getOverviewDollars(res, "lockedAmountDollars"),
      availableAmount: getOverviewDollars(res, "availableAmountDollars"),
      entryStatus: (res.entryStatus as BffOverview["entryStatus"]) ?? "NO_INVITE",
      totalInviteCount: res.totalInviteCount ?? 0,
      inviteCode: res.inviteCode || "",
      // 新增字段映射
      bannerGiftAmount: res.inviteeSignupRewardDollars ?? 0,
      bannerRewardAmount: res.inviterSignupRewardDollars ?? 0,
      inviteeSignupReward: res.inviteeSignupRewardDollars ?? 0,
      inviterSignupReward: res.inviterSignupRewardDollars ?? 0,
      // content 字段映射
      content: res.content
        ? {
            title: res.content.title,
            subTitle: res.content.subTitle,
          }
        : undefined,
      // 可选字段暂时不映射
      squadActivityStatus: undefined,
      recentEarnerNickname: undefined,
      recentEarnerAmount: undefined,
    };
  } catch (err) {
    console.error("[BFF API] getOverview error:", err);
    throw err;
  }
};

/**
 * 获取 BFF Dashboard 完整数据
 * POST /bffApi/getDashboard, protoId: 4906/4907
 */
export const getBffDashboard = async (): Promise<BffDashboard> => {
  try {
    const res = await httpClient.requestPost2(BffGetDashboardRequest, {}, BffGetDashboardResponse);
    console.log("[BFF API] getDashboard response:", res);

    return {
      lockedAmount: Number(getDashboardDollars(res, "lockedAmountDollars") ?? 0),
      availableAmount: Number(getDashboardDollars(res, "availableAmountDollars") ?? 0),
      inviteUrl: "",
      shareText: res.shareText || "",
      hasPendingCoinAnimation: res.hasPendingCoinAnimation ?? false,
      animationRewardDelta: undefined,
      squadList: (res.squadList || []).map(mapDashboardMember),
      inviteCode: res.inviteCode || "",
      defaultCopy: res.defaultContent || "",
      defaultReward: res.defaultRewards?.[0]
        ? {
            taskName: res.defaultRewards[0].title || "",
            rewardAmount: Number((res.defaultRewards[0].subTitle || "").replace(/[^\d.]/g, "")) || 60,
          }
        : undefined,
      historicalWithdrawAmount: Number(getDashboardDollars(res, "historicalWithdrawAmountDollars") ?? 0),
      defaultRewards: (res.defaultRewards || []).map((item) => ({
        title: item.title || "",
        subTitle: item.subTitle || "",
      })),
      buttonText: String(getDashboardDollars(res, "buttonText") ?? ""),
    };
  } catch (err) {
    console.error("[BFF API] getDashboard error:", err);
    throw err;
  }
};
```
