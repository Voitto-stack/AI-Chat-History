export const formatPercentage = (value: string): React.ReactNode => {
  if (!value || value === '0') return '-';
  const numValue = parseFloat(value);
  if (isNaN(numValue)) return value;

  const percentage = numValue.toFixed(2);
  let color = 'blue';

  return <Tag color={color}>{percentage}%</Tag>;

// 解析百分比字符串为数字
const parsePercentage = (value: string | number) => {
  if (typeof value === 'number') return value;
  if (!value) return 0;
  // 移除百分号并转换为数字
  return parseFloat(String(value).replace('%', '')) || 0;
};

// 渲染 R-Score Rate 列(带差值对比)
export const renderRScoreRateWithDiff = (
  record: PwaDailyDataItem,
  summaryData: PwaDailyDataItem | null,
): React.ReactNode => {
  if (!summaryData) {
    return <Tag color="blue">-</Tag>;
  }

  const current = parsePercentage(record.remainingRadio);
  const total = parsePercentage(summaryData.remainingRadio);
  const diff = Number((current - total).toFixed(2));

  if (diff === 0) {
    return (
      <div>
        <span style={{ marginRight: '8px' }}>{record.remainingRadio}</span>
        <Tag color="blue">0%</Tag>
      </div>
    );
  }

  const color = diff > 0 ? 'green' : 'red';
  const sign = diff > 0 ? '+' : '';

  return (
    <div>
      <span style={{ marginRight: '8px' }}>{record.remainingRadio}</span>
      <Tag color={color}>
        {sign}
        {diff}%
      </Tag>
    </div>
  );
};

// 根据 summaryData 获取带有自定义渲染的 columns
export const getColumnsWithSummary = (
  summaryData: PwaDailyDataItem | null,
): ProColumns<PwaDailyDataItem>[] => {
  return columns.map((col) => {
    if (col.key === 'remainingRadio') {
      return {
        ...col,
        render: (_: any, record: PwaDailyDataItem) =>
          renderRScoreRateWithDiff(record, summaryData),
      };
    }
    return col;
  });
};
