# feishu

```ts
/**
 * Feishu OAuth API
 * 飞书扫码登录相关接口
 */
import {
  feishuAuthorizeUrlResponseSchema,
  feishuConfigResponseSchema,
  feishuLoginRequestSchema,
  feishuLoginResponseSchema,
} from '@sitin/minerva-schemas';
import type {
  FeishuAuthorizeUrlResponse,
  FeishuConfigResponse,
  FeishuLoginRequest,
  FeishuLoginResponse,
} from '@sitin/minerva-schemas';
import { request } from '@/request/client';

// Re-export types for external use
export type {
  FeishuConfigResponse,
  FeishuAuthorizeUrlResponse,
  FeishuLoginRequest,
  FeishuLoginResponse,
};

/**
 * 获取飞书登录配置状态
 */
export async function getFeishuConfig(): Promise<FeishuConfigResponse> {
  return request<FeishuConfigResponse>(
    '/api/auth/feishu/config',
    { skipAuth: true },
    feishuConfigResponseSchema,
  );
}

/**
 * 获取飞书 OAuth 授权 URL（用于 QR SDK 的 goto 参数）
 */
export async function getFeishuAuthorizeUrl(): Promise<FeishuAuthorizeUrlResponse> {
  return request<FeishuAuthorizeUrlResponse>(
    '/api/auth/feishu/authorize-url',
    { skipAuth: true },
    feishuAuthorizeUrlResponseSchema,
  );
}

/**
 * 飞书扫码登录 - 用授权码换取 JWT token
 */
export async function feishuLogin(
  params: FeishuLoginRequest,
): Promise<FeishuLoginResponse> {
  const body = feishuLoginRequestSchema.parse(params);
  return request<FeishuLoginResponse>(
    '/api/auth/feishu/login',
    {
      method: 'POST',
      body,
      skipAuth: true,
    },
    feishuLoginResponseSchema,
  );
}
```
