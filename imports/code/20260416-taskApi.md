# taskApi

```ts
/**
 * 任务相关 API
 */

import {
  ListAllCommonTaskRequest,
  ListAllCommonTaskResponse,
  ListUserCommonTaskRequest,
  ListUserCommonTaskResponse,
  FinishUserCommonTaskRequest,
  FinishUserCommonTaskResponse,
} from "@heyhru/business-pwa-proto/gen/archat_api/user_api";
import httpClient from "./httpClient";

// 获取所有任务列表
export async function listAllCommonTasks(): Promise<ListAllCommonTaskResponse> {
  const request: ListAllCommonTaskRequest = {};

  return await httpClient.requestPost2(ListAllCommonTaskRequest, request, ListAllCommonTaskResponse);
}

// 获取用户已完成的任务列表
export async function getCommonFinishedTasks(): Promise<ListUserCommonTaskResponse> {
  const request: ListUserCommonTaskRequest = {};

  return await httpClient.requestPost2(ListUserCommonTaskRequest, request, ListUserCommonTaskResponse);
}

// 完成任务，提交任务完成请求并获取奖励
export async function finishUserCommonTask(taskId: number): Promise<FinishUserCommonTaskResponse> {
  const request: FinishUserCommonTaskRequest = {
    taskId,
  };

  return await httpClient.requestPost2(FinishUserCommonTaskRequest, request, FinishUserCommonTaskResponse);
}
```
