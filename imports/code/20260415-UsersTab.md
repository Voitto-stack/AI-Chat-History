# UsersTab

```tsx
import { PlusOutlined } from '@ant-design/icons';
import {
  ModalForm,
  ProFormSelect,
  ProFormText,
  ProTable,
  type ActionType,
  type ProColumns,
} from '@ant-design/pro-components';
import { useRequest } from 'ahooks';
import { Button, Descriptions, Drawer, List, message, Popconfirm, Space, Tag } from 'antd';
import { useMemo, useRef, useState } from 'react';
import { useAccess, useModel } from '@umijs/max';
import {
  assignUserRoles,
  createUser,
  deleteUser,
  getUser,
  listRoles,
  listUsers,
  resetUserPassword,
  updateUser,
  type AdminUserDetail,
} from '@/api';
import { USER_TYPE_MAP, type UserItem } from './config';

const UsersTab = () => {
  // Hooks
  const access = useAccess();
  const { initialState } = useModel('@@initialState');
  const actionRef = useRef<ActionType>(null);
  const [createOpen, setCreateOpen] = useState(false);
  const [editOpen, setEditOpen] = useState(false);
  const [assignOpen, setAssignOpen] = useState(false);
  const [resetOpen, setResetOpen] = useState(false);
  const [detailOpen, setDetailOpen] = useState(false);
  const [currentUser, setCurrentUser] = useState<UserItem | null>(null);
  const [detailUser, setDetailUser] = useState<AdminUserDetail | null>(null);

  const { runAsync: loadUsers } = useRequest(listUsers, { manual: true });
  const { runAsync: loadRoles, data: rolesData } = useRequest(listRoles, { manual: true });
  const { runAsync: loadUserDetail } = useRequest(getUser, { manual: true });

  const roleOptions = useMemo(
    () =>
      (rolesData || []).map((role) => ({
        label: role.name,
        value: role.id,
      })),
    [rolesData],
  );

  // API 请求函数
  const requestTable = async () => {
    try {
      const users = await loadUsers();
      return {
        data: users || [],
        success: true,
        total: users?.length || 0,
      };
    } catch (error) {
      message.error('用户列表加载失败');
      return { data: [], success: false, total: 0 };
    }
  };

  const handleOpenDetail = async (record: UserItem) => {
    setDetailOpen(true);
    setDetailUser(null);
    setCurrentUser(record);
    try {
      const detail = await loadUserDetail(record.id);
      setDetailUser(detail);
    } catch (error) {
      message.error('用户详情加载失败');
    }
  };

  const columns: ProColumns<UserItem>[] = [
    {
      title: '用户名',
      dataIndex: 'sub',
      key: 'sub',
    },
    {
      title: '显示名',
      dataIndex: 'displayName',
      key: 'displayName',
      render: (_, record) => record.displayName || '-',
    },
    {
      title: '用户类型',
      dataIndex: 'userType',
      key: 'userType',
      render: (_, record) => USER_TYPE_MAP[record.userType] || record.userType,
    },
    {
      title: '手机号',
      dataIndex: 'mobile',
      key: 'mobile',
      render: (_, record) => record.mobile || '-',
    },
    {
      title: '租户',
      dataIndex: 'tenantKey',
      key: 'tenantKey',
      render: (_, record) => record.tenantKey || '-',
    },
    {
      title: '角色',
      dataIndex: 'roles',
      key: 'roles',
      render: (_, record) =>
        record.roles?.length ? (
          record.roles.map((role) => (
            <Tag key={role.id}>{role.name}</Tag>
          ))
        ) : (
          '-'
        ),
    },
    {
      title: '操作',
      key: 'actions',
      valueType: 'option',
      render: (_, record) => {
        const canWrite = access.can('rbac.user.write');
        const canDelete = access.can('rbac.user.delete');
        const canRead = access.can('rbac.user.read');
        const isSelf = record.id === initialState?.currentUser?.id;

        return (
          <Space size={8}>
            {canRead && (
              <Button
                type="link"
                onClick={() => handleOpenDetail(record)}
              >
                查看
              </Button>
            )}
            {canWrite && (
              <Button
                type="link"
                onClick={() => {
                  setCurrentUser(record);
                  setEditOpen(true);
                }}
              >
                编辑
              </Button>
            )}
            {canWrite && (
              <Button
                type="link"
                onClick={() => {
                  setCurrentUser(record);
                  setAssignOpen(true);
                  if (!rolesData) {
                    loadRoles();
                  }
                }}
              >
                分配角色
              </Button>
            )}
            {canWrite && (
              <Button
                type="link"
                onClick={() => {
                  setCurrentUser(record);
                  setResetOpen(true);
                }}
              >
                重置密码
              </Button>
            )}
            {canDelete && (
              <Popconfirm
                title={isSelf ? '不能删除自己' : '确认删除该用户？'}
                onConfirm={async () => {
                  if (isSelf) {
                    return;
                  }
                  await deleteUser(record.id);
                  message.success('删除成功');
                  actionRef.current?.reload();
                }}
                okButtonProps={{ disabled: isSelf }}
              >
                <Button type="link" danger disabled={isSelf}>
                  删除
                </Button>
              </Popconfirm>
            )}
          </Space>
        );
      },
    },
  ];

  return (
    <>
      <ProTable<UserItem>
        headerTitle="用户管理"
        actionRef={actionRef}
        columns={columns}
        request={requestTable as any}
        rowKey="id"
        search={false}
        options={false}
        scroll={{ x: 'max-content' }}
        columnsState={{
          persistenceKey: 'rbac-users-table-columns',
          persistenceType: 'localStorage',
        }}
        pagination={{
          defaultPageSize: 10,
          showSizeChanger: true,
          showQuickJumper: true,
        }}
        cardBordered
        toolBarRender={() =>
          access.can('rbac.user.write')
            ? [
                <Button
                  key="create"
                  type="primary"
                  icon={<PlusOutlined />}
                  onClick={() => {
                    setCreateOpen(true);
                    if (!rolesData) {
                      loadRoles();
                    }
                  }}
                >
                  新建用户
                </Button>,
              ]
            : []
        }
      />

      <ModalForm
        title="新建用户"
        open={createOpen}
        modalProps={{ destroyOnHidden: true, onCancel: () => setCreateOpen(false) }}
        onFinish={async (values) => {
          await createUser(values);
          message.success('创建成功');
          setCreateOpen(false);
          actionRef.current?.reload();
          return true;
        }}
      >
        <ProFormText
          name="sub"
          label="用户名"
          rules={[
            { required: true, message: '请输入用户名' },
            { min: 3, max: 100, message: '长度需在 3-100 之间' },
          ]}
        />
        <ProFormText.Password
          name="password"
          label="密码"
          rules={[
            { required: true, message: '请输入密码' },
            { min: 6, message: '密码至少 6 位' },
          ]}
        />
        <ProFormText
          name="displayName"
          label="显示名"
          rules={[{ max: 100, message: '最多 100 字' }]}
        />
        <ProFormSelect
          name="roleIds"
          label="角色"
          options={roleOptions}
          mode="multiple"
          placeholder="请选择角色"
        />
      </ModalForm>

      <ModalForm
        key={`edit-${currentUser?.id || 'empty'}`}
        title="编辑用户"
        open={editOpen}
        modalProps={{ destroyOnHidden: true, onCancel: () => setEditOpen(false) }}
        initialValues={{
          displayName: currentUser?.displayName || '',
          mobile: currentUser?.mobile || '',
        }}
        onFinish={async (values) => {
          if (!currentUser) return false;
          await updateUser(currentUser.id, values);
          message.success('更新成功');
          setEditOpen(false);
          actionRef.current?.reload();
          return true;
        }}
      >
        <ProFormText
          name="displayName"
          label="显示名"
          rules={[{ max: 100, message: '最多 100 字' }]}
        />
        <ProFormText
          name="mobile"
          label="手机号"
          rules={[{ max: 50, message: '最多 50 字' }]}
        />
      </ModalForm>

      <ModalForm
        key={`assign-${currentUser?.id || 'empty'}`}
        title="分配角色"
        open={assignOpen}
        modalProps={{ destroyOnHidden: true, onCancel: () => setAssignOpen(false) }}
        initialValues={{
          roleIds: currentUser?.roles?.map((role) => role.id) || [],
        }}
        onFinish={async (values) => {
          if (!currentUser) return false;
          await assignUserRoles(currentUser.id, {
            roleIds: values.roleIds || [],
          });
          message.success('分配成功');
          setAssignOpen(false);
          actionRef.current?.reload();
          return true;
        }}
      >
        <ProFormSelect
          name="roleIds"
          label="角色"
          options={roleOptions}
          mode="multiple"
          rules={[{ required: true, message: '请选择角色' }]}
          placeholder="请选择角色"
        />
      </ModalForm>

      <ModalForm
        key={`reset-${currentUser?.id || 'empty'}`}
        title="重置密码"
        open={resetOpen}
        modalProps={{ destroyOnHidden: true, onCancel: () => setResetOpen(false) }}
        onFinish={async (values) => {
          if (!currentUser) return false;
          await resetUserPassword(currentUser.id, {
            newPassword: values.newPassword,
          });
          message.success('重置成功');
          setResetOpen(false);
          return true;
        }}
      >
        <ProFormText.Password
          name="newPassword"
          label="新密码"
          rules={[
            { required: true, message: '请输入新密码' },
            { min: 6, message: '密码至少 6 位' },
          ]}
        />
      </ModalForm>

      <Drawer
        title="用户详情"
        width={600}
        open={detailOpen}
        onClose={() => setDetailOpen(false)}
      >
        <Descriptions column={2} size="small">
          <Descriptions.Item label="用户名">
            {detailUser?.sub || currentUser?.sub || '-'}
          </Descriptions.Item>
          <Descriptions.Item label="显示名">
            {detailUser?.displayName || currentUser?.displayName || '-'}
          </Descriptions.Item>
          <Descriptions.Item label="用户类型">
            {detailUser?.userType
              ? USER_TYPE_MAP[detailUser.userType] || detailUser.userType
              : currentUser?.userType
              ? USER_TYPE_MAP[currentUser.userType] || currentUser.userType
              : '-'}
          </Descriptions.Item>
          <Descriptions.Item label="手机号">
            {detailUser?.mobile || currentUser?.mobile || '-'}
          </Descriptions.Item>
          <Descriptions.Item label="租户">
            {detailUser?.tenantKey || currentUser?.tenantKey || '-'}
          </Descriptions.Item>
        </Descriptions>

        <List
          style={{ marginTop: 16 }}
          header="角色与权限"
          dataSource={detailUser?.roles || []}
          renderItem={(role) => (
            <List.Item key={role.id}>
              <Space direction="vertical" size={4}>
                <strong>{role.name}</strong>
                <div>
                  {role.permissions?.length
                    ? role.permissions.map((permission) => (
                        <Tag key={permission}>{permission}</Tag>
                      ))
                    : '-'}
                </div>
              </Space>
            </List.Item>
          )}
        />
      </Drawer>
    </>
  );
};

export default UsersTab;
```
