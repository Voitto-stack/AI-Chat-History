# ChatItem

```tsx
import { memo, useCallback } from "react";
import { Conversation } from "@tencentcloud/chat";
import { getAvatarUrl, CustomAvatarType, DEFAULT_AVATAR, isRealHuman } from "@/utils/userUtil";
import { getTimeStamp } from "@/utils/chatUtils";
import { BadgeView } from "@/components/BadgeView";
import { ChatListItem, useChatUserInfo } from "@/hooks/chatList";
import { createChatMessage } from "@/types/chatMessage";
import icChatHaoganRed from "@/assets/images/ic_chat_haogan_red.svg";
import { bpTrack } from "@/tracking";
import { EventName } from "@/tracking/events";
interface ChatItemProps {
  item: ChatListItem;
  onClick?: (item: Conversation) => void;
}

// 获取消息预览文本
function getMessagePreview(item: Conversation): string {
  if (!item.lastMessage) return "[unknown message]";

  const { lastMessage } = item;

  // 使用 createChatMessage 解析消息并调用 content() 方法
  try {
    const message = createChatMessage(lastMessage);
    return message.content();
  } catch {
    // 降级处理
    switch (lastMessage.type) {
      case "TIMImageElem":
        return "[image]";
      case "TIMSoundElem":
        return "[Voice Message]";
      case "TIMVideoFileElem":
        return "[Video]";
      case "TIMTextElem":
        return lastMessage.payload?.text || "";
      default:
        return "";
    }
  }
}

const ChatItem = memo<ChatItemProps>(({ item, onClick }) => {
  const { userInfo } = useChatUserInfo(item.conversationID);
  const intimacyNum = item.intimacyNum;

  const handleClick = useCallback(() => {
    // 埋点：点击会话
    bpTrack(EventName.pwa_chat_conversation_click, {
      conversation_id: item.conversationID,
      unread_count: item.unreadCount || 0,
      intimacy_num: item.intimacyNum || 0,
      target_user_id: userInfo?.userId,
      target_user_type: isRealHuman(userInfo) ? "real" : "dh",
    });
    onClick?.(item);
  }, [item, onClick]);

  const avatarUrl = getAvatarUrl(userInfo, CustomAvatarType.Min);
  const messagePreview = getMessagePreview(item);
  const lastMessageTime = item.lastMessage?.lastTime ? item.lastMessage.lastTime * 1000 : Date.now();
  const formattedTime = getTimeStamp(lastMessageTime);

  return (
    <div
      className="flex h-[80px] w-full shrink-0 cursor-pointer items-center gap-2 border-b border-[#f0f0f0] active:bg-black/[0.03] transition-colors duration-150"
      onClick={handleClick}
    >
      {/* 头像 */}
      <div className="relative h-[60px] w-[60px] shrink-0">
        <img
          src={avatarUrl || DEFAULT_AVATAR}
          alt="avatar"
          className="h-full w-full rounded-full object-cover"
          onError={(e) => {
            e.currentTarget.src = DEFAULT_AVATAR;
          }}
        />
        <BadgeView unreadCount={item.unreadCount} className="absolute -right-1 -top-1" />
      </div>

      {/* 名称 + 消息预览 */}
      <div className="flex min-w-0 flex-1 flex-col items-start gap-[6px]">
        <span className="w-fit font-[Pangram] text-[15px] font-medium leading-[20px] tracking-[-0.23px] text-[#000000]">
          {userInfo?.username || "User"}
        </span>
        <span className="block w-full overflow-hidden text-ellipsis whitespace-nowrap font-[Pangram] text-[13px] font-normal leading-[18px] tracking-[-0.23px] text-[#3c3c4399]">
          {messagePreview}
        </span>
      </div>

      {/* 时间 + 亲密度 */}
      <div className="flex w-[95px] shrink-0 flex-col items-end gap-[6px]">
        <div className="font-[Pangram] text-[12px] font-normal leading-[16px] tracking-[-0.23px] text-[#3c3c434d]">
          {formattedTime}
        </div>
        {intimacyNum > 0 && (
          <div className="flex items-center gap-[4px]">
            <span className="font-[Pangram] text-[12px] font-normal leading-[16px] text-[#ff3b30]">{intimacyNum}</span>
            <img className="h-[14px] w-[14px]" src={icChatHaoganRed} alt="intimacy" />
          </div>
        )}
      </div>
    </div>
  );
});

export default ChatItem;
```
