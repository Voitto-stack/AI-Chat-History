# firebase

```ts
/**
 * Firebase 推送通知服务
 * 仅用于后台/离线时的消息推送，在线时不推送
 * APK 环境下跳过初始化（由原生处理推送）
 */

import { initializeApp, FirebaseApp } from "firebase/app";
import { getMessaging, getToken, isSupported, Messaging } from "firebase/messaging";
import { isApp } from "@/utils/bridge";
import httpClient from "@/http/httpClient";
import { UploadFcmTokenRequest, UploadFcmTokenResponse } from "@heyhru/business-pwa-proto/gen/archat_api/user_api";

const TAG = "Firebase";

let app: FirebaseApp | null = null;
let messaging: Messaging | null = null;

/**
 * 从环境变量读取 Firebase 配置
 */
function getFirebaseConfig() {
  return {
    apiKey: import.meta.env.VITE_FIREBASE_API_KEY,
    authDomain: import.meta.env.VITE_FIREBASE_AUTH_DOMAIN,
    projectId: import.meta.env.VITE_FIREBASE_PROJECT_ID,
    storageBucket: import.meta.env.VITE_FIREBASE_STORAGE_BUCKET,
    messagingSenderId: import.meta.env.VITE_FIREBASE_MESSAGING_SENDER_ID,
    appId: import.meta.env.VITE_FIREBASE_APP_ID,
    measurementId: import.meta.env.VITE_FIREBASE_MEASUREMENT_ID,
  };
}

/**
 * 初始化 Firebase（APK 环境下跳过）
 * @returns {Promise<boolean>} 是否初始化成功
 */
export async function initFirebase(): Promise<boolean> {
  // APK 环境由原生处理推送
  if (isApp()) {
    console.log(TAG, "APK 环境，跳过 Firebase 初始化");
    return false;
  }

  if (messaging) return true;

  try {
    const supported = await isSupported();
    if (!supported) {
      console.warn(TAG, "当前浏览器不支持 Firebase Messaging");
      return false;
    }

    app = initializeApp(getFirebaseConfig());
    messaging = getMessaging(app);
    console.log(TAG, "Firebase 初始化成功");
    return true;
  } catch (error) {
    console.error(TAG, "Firebase 初始化失败:", error);
    return false;
  }
}

/**
 * 上传 FCM Token 到服务器
 */
async function uploadFcmToken(token: string): Promise<void> {
  try {
    const request: UploadFcmTokenRequest = { fcmToken: token };
    await httpClient.requestPost2(UploadFcmTokenRequest, request, UploadFcmTokenResponse);
    console.log(TAG, "FCM Token 上传成功");
  } catch (error) {
    console.error(TAG, "FCM Token 上传失败:", error);
  }
}

/**
 * 获取 FCM Token 并上传到服务器
 * @returns {Promise<string | null>} FCM Token 或 null
 */
export async function fetchFcmToken(): Promise<string | null> {
  if (!messaging) {
    console.warn(TAG, "Firebase 未初始化，无法获取 Token");
    return null;
  }

  const vapidKey = import.meta.env.VITE_FIREBASE_VAPID_KEY;
  if (!vapidKey) {
    console.error(TAG, "缺少 VAPID Key 配置");
    return null;
  }

  try {
    const token = await getToken(messaging, { vapidKey });
    if (token) {
      await uploadFcmToken(token);
      return token;
    }
    console.warn(TAG, "未获取到 FCM Token");
    return null;
  } catch (error) {
    console.error(TAG, "获取 FCM Token 失败:", error);
    return null;
  }
}
```
