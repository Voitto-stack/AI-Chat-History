# audit.model

```ts
import { getPgDb } from "@heyhru/server-plugin-pg";
import * as Q from "./audit.sql.js";

export function insertAuditLog(entry: {
  userId: string;
  dataSourceId?: string;
  action: string;
  sqlText?: string;
  resultSummary?: string;
  ipAddress?: string;
}) {
  return getPgDb().run(Q.INSERT, [
    entry.userId,
    entry.dataSourceId ?? null,
    entry.action,
    entry.sqlText ?? null,
    entry.resultSummary ?? null,
    entry.ipAddress ?? null,
  ]);
}

export async function queryAuditLogs(filters?: {
  userId?: string;
  dataSourceId?: string;
  limit?: number;
  offset?: number;
}) {
  let query = Q.LIST;
  const params: unknown[] = [];
  if (filters?.userId) {
    query += " AND al.user_id = ?";
    params.push(filters.userId);
  }
  if (filters?.dataSourceId) {
    query += " AND al.data_source_id = ?";
    params.push(filters.dataSourceId);
  }
  query += " ORDER BY al.created_at DESC LIMIT ? OFFSET ?";
  params.push(filters?.limit ?? 50, filters?.offset ?? 0);
  return getPgDb().query(query, params);
}
```
