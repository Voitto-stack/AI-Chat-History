# insStore

```ts
/**
 * Instagram 授权状态管理
 * 使用 Zustand 管理 Instagram 授权状态
 */

import { create } from "zustand";

interface InsState {
  insState: boolean; // Instagram 授权状态
  setInsState: (state: boolean) => void; // 设置授权状态
  resetInsState: () => void; // 重置授权状态（授权成功后关闭弹窗时调用）
}

export const useInsStore = create<InsState>((set) => ({
  insState: false,

  setInsState: (state) => {
    set({ insState: state });
  },

  resetInsState: () => {
    set({ insState: false });
  },
}));

```
