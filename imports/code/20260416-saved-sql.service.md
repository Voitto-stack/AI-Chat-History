# saved-sql.service

```ts
import { type FastifyRequest, type FastifyReply } from "fastify";
import { getRedis } from "@heyhru/server-plugin-redis";
import {
  listSavedSqlsByUser,
  createSavedSqlRow,
  updateSavedSqlRow,
  deleteSavedSqlRow,
} from "./saved-sql.model.js";

interface AuthenticatedRequest extends FastifyRequest {
  user: { id: string };
}

const CACHE_PREFIX = "saved_sqls:";
const CACHE_TTL = 3600; // 1 hour

function cacheKey(userId: string) {
  return `${CACHE_PREFIX}${userId}`;
}

async function invalidateCache(userId: string) {
  const redis = getRedis();
  if (redis) await redis.del(cacheKey(userId));
}

export async function savedSqlList(req: AuthenticatedRequest, reply: FastifyReply) {
  const userId = req.user.id;
  const redis = getRedis();

  if (redis) {
    const cached = await redis.get(cacheKey(userId));
    if (cached) return reply.send(JSON.parse(cached));
  }

  const rows = await listSavedSqlsByUser(userId);
  if (redis) await redis.set(cacheKey(userId), JSON.stringify(rows), "EX", CACHE_TTL);
  return reply.send(rows);
}

export async function savedSqlCreate(req: AuthenticatedRequest, reply: FastifyReply) {
  const { name, sqlText, dataSourceId, database } = (req.body ?? {}) as {
    name: string;
    sqlText: string;
    dataSourceId?: string;
    database?: string;
  };
  if (!name || !sqlText) {
    return reply.code(400).send({ error: "name and sqlText are required" });
  }
  const row = await createSavedSqlRow({
    userId: req.user.id,
    name,
    sqlText,
    dataSourceId: dataSourceId ?? null,
    database: database ?? null,
  });
  await invalidateCache(req.user.id);
  return reply.code(201).send(row);
}

export async function savedSqlUpdate(req: AuthenticatedRequest, reply: FastifyReply) {
  const { id, name, sqlText } = (req.body ?? {}) as {
    id: string;
    name?: string;
    sqlText?: string;
  };
  if (!id) return reply.code(400).send({ error: "id is required" });
  try {
    const row = await updateSavedSqlRow(id, req.user.id, { name, sqlText });
    if (!row) return reply.code(404).send({ error: "Not found" });
    await invalidateCache(req.user.id);
    return reply.send(row);
  } catch (err) {
    return reply.code(400).send({ error: err instanceof Error ? err.message : String(err) });
  }
}

export async function savedSqlDelete(req: AuthenticatedRequest, reply: FastifyReply) {
  const { id } = (req.body ?? {}) as { id: string };
  if (!id) return reply.code(400).send({ error: "id is required" });
  await deleteSavedSqlRow(id, req.user.id);
  await invalidateCache(req.user.id);
  return reply.code(204).send();
}
```
