# logger

```ts
import createWriteStream from "pino-roll";
import { createLogger, type LogStreamEntry } from "@heyhru/common-util-logger";
import { config } from "./config.js";

async function buildStreams(): Promise<LogStreamEntry[] | undefined> {
  if (config.nodeEnv !== "production" || !config.logDir) {
    return undefined;
  }
  const fileStream = await createWriteStream({
    file: `${config.logDir}/app`,
    frequency: "daily",
    dateFormat: "yyyy-MM-dd",
    size: "500m",
    symlink: true,
    limit: { count: 3 },
    mkdir: true,
  });
  return [{ stream: process.stdout }, { stream: fileStream }];
}

export const logger = createLogger({
  name: "dms-server",
  streams: await buildStreams(),
});
```
