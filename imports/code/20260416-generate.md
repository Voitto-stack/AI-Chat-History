# generate

```sh
#!/usr/bin/env bash
# Generate TypeScript from .proto files and build protoIdMapping.json
#
# Prerequisites:
#   brew install protobuf        # protoc >= v6.32.0
#   pnpm install                 # installs ts-proto (protoc-gen-ts_proto)
#
# Usage:
#   pnpm --filter @heyhru/business-pwa-proto proto:gen

set -euo pipefail
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
PKG_DIR="$(dirname "$SCRIPT_DIR")"
PROTO_DIR="$PKG_DIR/proto"
GEN_DIR="$PKG_DIR/src/gen"
PLUGIN="$PKG_DIR/node_modules/.bin/protoc-gen-ts_proto"
# Fallback: hoisted root node_modules (pnpm node-linker=hoisted)
if [ ! -f "$PLUGIN" ]; then
  PLUGIN="$(cd "$PKG_DIR/../.." && pwd)/node_modules/.bin/protoc-gen-ts_proto"
fi

# ─── Preflight checks ───────────────────────────────────────────────────────
if ! command -v protoc &>/dev/null; then
  echo "protoc not found. Install with: brew install protobuf"
  exit 1
fi
if [ ! -f "$PLUGIN" ]; then
  echo "protoc-gen-ts_proto not found. Run: pnpm install"
  exit 1
fi
echo "protoc $(protoc --version | awk '{print $2}')"
echo "ts-proto plugin: $PLUGIN"

# ─── 1. Clean old generated files ───────────────────────────────────────────
echo ""
echo "Cleaning $GEN_DIR ..."
rm -rf "$GEN_DIR"
mkdir -p "$GEN_DIR"

# ─── 2. Collect all .proto files ────────────────────────────────────────────
PROTO_FILES=$(find "$PROTO_DIR" -name "*.proto" -type f)
if [ -z "$PROTO_FILES" ]; then
  echo "No .proto files found in $PROTO_DIR"
  exit 1
fi
echo "Found $(echo "$PROTO_FILES" | wc -l | tr -d ' ') .proto files"

# ─── 3. Generate TypeScript with protoc + ts-proto ──────────────────────────
#
# Key ts-proto options (aligned with app-pwa object-literal usage):
#   esModuleInterop:          ES module compatible imports
#   forceLong=number:         use number instead of Long for int64
#   useExactTypes=false:      avoid Exact<DeepPartial<T>> restrictions on object literals
#   useOptionals=all:         generate optional keys for scalar/message fields
#   importSuffix=:            no .js suffix on imports (handled by bundler)
#   outputPartialMethods:     generate create/fromPartial methods
#   outputTypeAnnotations:    omit synthetic $type field from message interfaces
#
echo ""
echo "Running protoc ..."
protoc \
  --plugin="protoc-gen-ts_proto=$PLUGIN" \
  --ts_proto_out="$GEN_DIR" \
  --ts_proto_opt=esModuleInterop=true \
  --ts_proto_opt=forceLong=number \
  --ts_proto_opt=useExactTypes=false \
  --ts_proto_opt=useOptionals=all \
  --ts_proto_opt=importSuffix= \
  --ts_proto_opt=outputPartialMethods=true \
  --ts_proto_opt=outputTypeAnnotations=false \
  --ts_proto_opt=useProtobufTimestamp=false \
  --proto_path="$PROTO_DIR" \
  $PROTO_FILES

echo "TypeScript generated in $GEN_DIR"

# ─── 3.5. Prepend @ts-nocheck to all generated .ts files ───────────────────
echo ""
echo "Adding @ts-nocheck to generated files ..."
find "$GEN_DIR" -name "*.ts" -type f -exec sed -i '' '1s/^/\/\/ @ts-nocheck\n/' {} +

# ─── 3.6. Re-inject runtime/type-level $type for generated MessageFns ───────
# We keep message interfaces app-pwa-friendly (no synthetic $type fields on every
# message), but web-util-http still relies on MessageFns.$type to derive request
# names and proto IDs at runtime. So we patch generated MessageFns declarations
# and constants only.
echo ""
echo 'Patching generated MessageFns $type ...'
GEN_DIR="$GEN_DIR" python3 - <<'PY'
from pathlib import Path
import os
import re

root = Path(os.environ["GEN_DIR"])

for path in root.rglob("*.ts"):
    text = path.read_text(encoding="utf-8")
    pkg_match = re.search(r'export const protobufPackage = "([^"]+)";', text)
    if not pkg_match:
        continue
    protobuf_package = pkg_match.group(1)

    text = text.replace(
        "export interface MessageFns<T> {\n",
        "export interface MessageFns<T> {\n  readonly $type: string;\n",
        1,
    )

    interface_pattern = re.compile(r'^export interface ([A-Za-z0-9_]+) \{$', re.MULTILINE)

    def inject_interface_type(match: re.Match[str]) -> str:
        name = match.group(1)
        if name == "MessageFns":
            return match.group(0)
        return f"export interface {name} {{\n  $type?: string | undefined;"

    text = interface_pattern.sub(inject_interface_type, text)

    def inject_type(match: re.Match[str]) -> str:
        name = match.group(1)
        body = match.group(0)
        if '$type:' in body:
            return body
        return body.replace(
            "{\n",
            "{\n  $type: \"%s.%s\",\n" % (protobuf_package, name),
            1,
        )

    pattern = re.compile(
        r'export const ([A-Za-z0-9_]+): MessageFns<[^>]+> = \{\n',
        re.MULTILINE,
    )
    text = pattern.sub(inject_type, text)
    path.write_text(text, encoding="utf-8")
PY

# ─── 4. Generate protoIdMapping.json from ProtoConfig*.json ─────────────────
echo ""
echo "Generating protoIdMapping.json ..."
node "$SCRIPT_DIR/gen-proto-id-mapping.mjs"

echo ""
echo "Done! Run 'pnpm build' to compile."

```
