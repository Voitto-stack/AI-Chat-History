# GiftList

```tsx
import { memo, useCallback, useEffect, useMemo, useState } from "react";
import { GiftInfo } from "@sitin/api-proto/gen/archat_api/user_api";
import { getGiftList } from "@/http/chatApi";
import { extractFileName } from "@/utils/stringUtils";
import ModalContainer from "@/components/ModalContainer";
import EmptyView from "@/components/EmptyView";
import Button from "@/components/Button";
import Spinner from "@/components/Spinner";

type GiftTab = "normal" | "radiant" | "mystic" | "celestial";

const TAB_LABELS: { key: GiftTab; label: string }[] = [
  { key: "normal", label: "Normal" },
  { key: "radiant", label: "Radiant" },
  { key: "mystic", label: "Mystic" },
  { key: "celestial", label: "Celestial" },
];

// 礼物等级与 Tab 的映射
const LEVEL_TAB_MAP: Record<number, GiftTab> = {
  1: "normal",
  2: "radiant",
  3: "mystic",
  4: "celestial",
};

interface GiftListProps {
  onClose: () => void; // 关闭面板
  onSend: (gift: GiftInfo) => void; // 确认发送礼物
}

/**
 * GiftList - 礼物选择面板
 * 底部弹出，显示礼物列表供用户选择并发送
 */
const GiftList = memo<GiftListProps>(({ onClose, onSend }) => {
  const [gifts, setGifts] = useState<GiftInfo[]>([]);
  const [selectedTab, setSelectedTab] = useState<GiftTab>("normal");
  const [selectedGift, setSelectedGift] = useState<GiftInfo | null>(null);
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    setLoading(true);
    getGiftList()
      .then((resp) => {
        setGifts(resp.gifts ?? []);
      })
      .catch(() => {})
      .finally(() => setLoading(false));
  }, []);

  const filteredGifts = useMemo(
    () => gifts.filter((g) => g.giftLevel !== undefined && LEVEL_TAB_MAP[g.giftLevel] === selectedTab),
    [gifts, selectedTab],
  );

  const handleSend = useCallback(() => {
    if (!selectedGift) return;
    onSend(selectedGift);
    setSelectedGift(null);
  }, [selectedGift, onSend]);

  return (
    <ModalContainer open={true} onClose={onClose} variant="bottom-sheet">
      <div className="flex h-[350px] flex-col items-end overflow-hidden rounded-t-[15px] bg-white">
        {/* Tab 切换 */}
        <div className="flex w-full shrink-0 gap-5 border-b-[0.5px] border-[#e5e5ea] px-[10px]">
          {TAB_LABELS.map(({ key, label }) => (
            <button
              key={key}
              type="button"
              className={`shrink-0 border-none bg-transparent py-4 text-[15px] font-medium capitalize transition-colors ${
                selectedTab === key ? "text-[#000] border-b-[3px] border-[#1890ff]" : "text-[rgba(60,60,67,0.6)]"
              }`}
              onClick={() => {
                setSelectedTab(key);
                setSelectedGift(null);
              }}
            >
              {label}
            </button>
          ))}
        </div>

        {/* 礼物网格 */}
        <div className="grid w-full flex-1 grid-cols-4 gap-x-[1px] gap-y-[10px] overflow-y-auto p-[10px]">
          {loading && (
            <div className="col-span-4 flex items-center justify-center text-[#666]">
              <Spinner size={24} />
            </div>
          )}
          {!loading &&
            filteredGifts.map((gift) => (
              <GiftItem
                key={gift.giftId}
                gift={gift}
                isSelected={selectedGift?.giftId === gift.giftId}
                onClick={() => setSelectedGift(gift)}
              />
            ))}
          {!loading && filteredGifts.length === 0 && (
            <EmptyView emptyText="No gifts available" className="col-span-4" />
          )}
        </div>

        {/* 发送按钮 */}
        <Button
          variant="primary"
          disabled={!selectedGift}
          onClick={handleSend}
          className="!h-8 !w-20 m-[10px] shrink-0"
        >
          Send
        </Button>
      </div>
    </ModalContainer>
  );
});

// 礼物卡片组件
const GiftItem = memo<{ gift: GiftInfo; isSelected: boolean; onClick: () => void }>(({ gift, isSelected, onClick }) => {
  const displayName = extractFileName(gift.giftImageUrl) || gift.giftId || "";
  const priceInCents = gift.giftCostDollar ? Math.round(parseFloat(gift.giftCostDollar) * 100) : 0;
  return (
    <button
      type="button"
      className={`flex h-[118px] w-full flex-col items-center justify-center gap-1 rounded-lg ${
        isSelected ? "border border-[#47aeef] bg-[#ebf7ff]" : "bg-white"
      }`}
      onClick={onClick}
    >
      {gift.giftImageUrl && <img src={gift.giftImageUrl} alt={displayName} className="h-16 w-16 object-contain" />}
      <span className="text-[13px] leading-[18px] tracking-[-0.23px]">{displayName}</span>
      <span className="text-[11px] leading-[13px] tracking-[-0.23px] text-[rgba(60,60,67,0.6)]">{priceInCents}￠</span>
    </button>
  );
});

export default GiftList;
```
