/**
 * 获取旧版 API Base URL (protoId 网关)
 * @deprecated 逐步迁移到 getMinervaServerUrl
 */
export const getBaseUrl = (): string => {
  let url: string;

  console.log('(process.env.UMI_ENV)', process.env.UMI_ENV);

  switch (process.env.UMI_ENV) {
    case 'production':
      url = 'https://api-prod.memojiproperties.com/json';
      break;
    case 'development':
      url = 'https://api-dev.memojiproperties.com/json';
      break;
    case 'local':
      url = 'https://api-dev.memojiproperties.com/json';
      break;
    default:
      url = 'https://api-dev.memojiproperties.com/json';
  }

  return url;
};

/**
 * 获取 Minerva Server API Base URL (新后端)
 * 用于 RBAC、认证等功能
 */
export const getMinervaServerUrl = (): string => {
  let url: string;

  switch (process.env.UMI_ENV) {
    case 'production':
      url = 'https://admin-api-prod.sitin.ai';
      break;
    case 'development':
      url = 'https://admin-api-dev.sitin.ai';
      break;
    case 'local':
      url = 'http://localhost:29001';
      break;
    default:
      url = 'http://localhost:29001';
  }

  return url;
};

/**
 * 获取 Admin Console URL
 */
export const getAdminConsoleUrl = (): string => {
  switch (process.env.UMI_ENV) {
    case 'production':
      return 'https://admin.sitin.ai';
    case 'development':
    default:
      return 'https://cai-dev-admin-console.archat.us';
  }
};

export const getAdminUserProfileUrl = (userId: number) => {
  const baseUrl = getAdminConsoleUrl();
  const url = new URL('/user-console', baseUrl);
  url.searchParams.set('key', 'user-info-profile-search');
  url.searchParams.set('searchMode', 'userId');
  url.searchParams.set('searchString', String(userId));
  url.searchParams.set('userId', String(userId));

  return url.toString();
};
